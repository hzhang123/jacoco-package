package dtos

import io.growing.marketing.common.model.SmsConfig
import io.growing.micros.utils.conversion.GrowingConversions._
import play.api.libs.json.{ Json, Writes }

case class SmsConfigDto(smsConfig: SmsConfig)

object SmsConfigDto {
  implicit val writes: Writes[SmsConfigDto] = (dto: SmsConfigDto) => {
    val c = dto.smsConfig
    var base = Json.obj(
      "id" -> c.id.map(_.toUid),
      "creatorId" -> c.creatorId.toUid,
      "updaterId" -> c.updaterId.toUid,
      "smsProvider" -> c.smsProvider,
      "providerKey" -> c.providerKey,
      "signNames" -> c.signNames,
      "state" -> c.state,
      "sendPath" -> c.sendPath)

    c.providerValue.foreach(v => base ++= Json.obj("providerValue" -> v))
    c.numAttribute.foreach(a => base ++= Json.obj("numAttribute" -> a))
    base
  }
}
