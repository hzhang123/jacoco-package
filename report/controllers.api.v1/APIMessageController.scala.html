package controllers.api.v1

import java.util.Locale

import controllers.api.APIBaseController
import dao.ProductDao
import dtos.ApiErrorResponse
import form.MediaForm
import form.api.{ APIPushMessageForm, InAppMessageAPIForm }
import handlers.{ APIFormHandler, ApiPushTokenConverter, MessageContentHandler }
import io.growing.marketing.common.model.AudienceType.AudienceType
import io.growing.marketing.common.model.PushContent
import io.growing.micros.business.dao.ProjectDao
import io.growing.micros.business.model.Project
import io.growing.micros.utils.conversion.GrowingConversions._
import javax.inject.{ Inject, Singleton }
import play.api.libs.json.Json
import play.api.mvc.PlayBodyParsers
import play.api.mvc.Results.{ NoContent, Ok }
import service.auth.AuthenticationService
import service.{ PushApiService, PushMessageService }
import utils.TrackingManager

import scala.concurrent.ExecutionContext.Implicits.global
import scala.concurrent.Future

/**
 * APIPushMessageController
 *
 * @author damon lin
 *         2019/7/18
 */
@Singleton
class APIMessageController @Inject() (
  trackingManager: TrackingManager,
  apiFormHandler: APIFormHandler,
  apiPushTokenConverter: ApiPushTokenConverter,
  parse: PlayBodyParsers,
  projectDao: ProjectDao,
  authService: AuthenticationService,
  productDao: ProductDao,
  messageContentHandler: MessageContentHandler,
  pushMessageService: PushMessageService,
  pushApiService: PushApiService) extends APIBaseController(parse, projectDao, authService) {

  def createPushMessage(projectUid: String) = Actions(projectUid, parse.json).async(parse.tolerantJson) { implicit request =>
    request.body.validate[APIPushMessageForm].fold(
      errors => Future.successful(ApiErrorResponse.jsonBadRequest(errors)),
      form => {
        val productFuture = if (form.productId.nonEmpty) productDao.findByIds(request.projectId, Seq(form.productId.get.toId)).map(_.headOption)
        else productDao.findByPackageName(request.projectId, form.packageName)

        productFuture.flatMap {
          case Some(product) => pushApiService.createPushApiMessage(form, request, product)
          case _ => Future.successful(ApiErrorResponse.PRODUCT_NOT_FOUND.notFound)
        }
      })
  }

  def createInAppMessage(projectUid: String) = Actions(projectUid, parse.json).async(parse.tolerantJson) { implicit request =>
    request.body.validate[InAppMessageAPIForm].fold(
      errors => Future.successful(ApiErrorResponse.jsonBadRequest(errors)),
      form => {
        val realAudienceType = apiFormHandler.extraAudienceType(form.audience)
        if (realAudienceType.isEmpty) {
          Future.successful(ApiErrorResponse.EMPTY_AUDIENCE.badRequest)
        } else {
          val project = request.project
          val user = request.authentication.userId.getOrElse(-1)
          val futureMessageForm = convertForm(form, realAudienceType, project)

          for {
            messageForm <- futureMessageForm
            r <- pushMessageService.create(messageForm, project, user)
          } yield {
            if (r.message.isActive) {
              trackingManager.trackingPopupActive(r.message, r.rules.headOption, "直接上线")
            }
            Ok(Json.toJson(r.toPresenter))
          }
        }
      })
  }

  def updateInAppMessage(projectUid: String, messageUid: String) = Actions(projectUid, parse.json).async(parse.tolerantJson) { implicit request =>
    request.body.validate[InAppMessageAPIForm].fold(
      errors => Future.successful(ApiErrorResponse.jsonBadRequest(errors)),
      form => {
        val realAudienceType = apiFormHandler.extraAudienceType(form.audience)
        if (realAudienceType.isEmpty) {
          Future.successful(ApiErrorResponse.EMPTY_AUDIENCE.badRequest)
        } else {
          val project = request.project
          val user = request.authentication.userId.getOrElse(-1)
          val futureMessageForm = convertForm(form, realAudienceType, project)

          for {
            messageForm <- futureMessageForm
            r <- pushMessageService.update(messageForm, messageUid.toId, project, user)
          } yield {
            if (r.message.isActive) {
              trackingManager.trackingPopupActive(r.message, r.rules.headOption, r.from)
            }
            Ok(Json.toJson(r.toPresenter))
          }
        }
      })
  }

  def deleteInAppMessage(projectUid: String, id: String) = Actions(projectUid, parse.empty).async(parse.empty) { implicit request ⇒
    implicit val locale: Locale = request.locale
    val user = request.authentication.userId.getOrElse(-1)
    pushMessageService.deleteById(projectUid.toId, id.toId, user).map(_ => NoContent)
  }

  def indexInAppMessage(projectUid: String) = Actions(projectUid, parse.empty).async(parse.empty) { implicit request =>
    pushMessageService.findByProject(request.projectId).map { messages =>
      Ok(Json.toJson(messages.map(_.toPresenter)))
    }
  }

  // upload file
  def uploadMedias(projectUid: String) = Actions(projectUid, parse.json).async(parse.tolerantJson) { implicit request =>
    request.body.validate[MediaForm].fold(
      errors => Future.successful(ApiErrorResponse.jsonBadRequest(errors)),
      form => {
        messageContentHandler.handleUpload(request.project, form)
      })
  }

  private def convertForm(form: InAppMessageAPIForm, realAudienceType: Option[AudienceType], project: Project) = {
    val futureMessageForm = for {
      realAudienceId <- apiFormHandler.extraAudienceId(project.dbid, realAudienceType.get, form.audience)
      targets <- apiFormHandler.extraTargetMap(project.dbid, form.rule.targets)
    } yield {
      val realContent = PushContent.SimpleImgPushContent(form.content, targets)
      form.convertToPushMessageForm(realAudienceType, realAudienceId, realContent)
    }
    futureMessageForm
  }

}
