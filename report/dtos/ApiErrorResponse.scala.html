package dtos

import exception.MarketingApiException
import play.api.libs.json.{ JsPath, Json, JsonValidationError }
import play.api.mvc.Result
import play.api.mvc.Results.{ BadRequest, Forbidden, InternalServerError, NotFound }

/**
 * @author : xhldtc
 * @since : 2019/10/28
 */
case class ApiErrorResponse(code: Int, message: String) {

  implicit val fmt = Json.format[ApiErrorResponse]

  def toJsonString: String = {
    Json.toJson(this).toString()
  }

  def badRequest: Result = {
    BadRequest(toJsonString)
  }

  def forbidden: Result = {
    Forbidden(toJsonString)
  }

  def notFound: Result = {
    NotFound(toJsonString)
  }

  def internalServerError: Result = {
    InternalServerError(toJsonString)
  }
}

object ApiErrorResponse {

  def apply(marketingApiException: MarketingApiException): ApiErrorResponse =
    ApiErrorResponse(marketingApiException.code, marketingApiException.message)

  //请求body json校验不通过
  def jsonBadRequest(errors: Seq[(JsPath, Seq[JsonValidationError])]): Result = {
    val message = errors.flatMap(_._2).map(_.message).mkString(",")
    ApiErrorResponse(1001, message).badRequest
  }

  //service自定义error message封装
  def customBadRequest(message: String): Result = {
    ApiErrorResponse(1002, message).badRequest
  }

  //code 1001-1009预留，从1010开始往下添加
  val INTERNAL_SERVER_ERROR = ApiErrorResponse(1010, "系统内部异常，请稍后再试")
  val RATE_LIMIT_DENY = ApiErrorResponse(1011, "请求速度过快，请稍后再试")
  val INVALID_AUTH_TOKEN = ApiErrorResponse(1012, "API token校验不通过")
  val PROJECT_NOT_FOUND = ApiErrorResponse(1013, "未找到项目")
  val PROJECT_DISABLED = ApiErrorResponse(1014, "项目已被禁用")
  val PRODUCT_NOT_FOUND = ApiErrorResponse(1015, "未找到对应产品，请检查参数packageName")
  val DUPLICATE_REQUEST = ApiErrorResponse(1016, "重复请求，请检查参数")
  val EMPTY_AUDIENCE = ApiErrorResponse(1017, "查询目标用户为空，请检查audience参数")
  val PUSH_API_UNEXPECT_ERROR = ApiErrorResponse(1018, "未知异常，请更换cid后重试")
  val DUPLICATE_AUDIENCE = ApiErrorResponse(1019, "推送目标用户重复，请检查audience参数")
}
