package actors

import actors.flow.{ BuildPushTaskActor, CheckpointTaskActor, PublishPushTaskActor, UserInsightActor }
import actors.protocol.JobProtocols.{ CheckPointedTasks, PublishTaskSuccess, TasksNeedCheckpoint }
import akka.actor.SupervisorStrategy.Resume
import akka.actor.{ Actor, OneForOneStrategy, Props, SupervisorStrategy }
import com.google.common.base.Throwables
import com.google.inject.Inject
import com.typesafe.scalalogging.LazyLogging
import dtos.Messages.{ KafkaPushMessage, PushTaskRequest }
import dtos.{ JobContext, PushTaskRequestDto, UserInfo }
import io.growing.marketing.common.model.PushTask

/**
 * EndpointActor
 * 处理流程：根据分群查询得到用户列表 -> 组装得到推送任务 -> 持久化存储 -> 发布任务
 * segment -> Seq[Seq[[UserInfo]]] -> Seq[[PushTask]] -> Seq[[KafkaPushMessage]]
 *
 * @author damon lin
 *         2019/5/11
 */
class EndpointActor @Inject() (injector: Injector) extends Actor with LazyLogging {

  private val buildPushTasksActor = context.actorOf(BuildPushTaskActor.props(injector), "buildPushTasksActor")
  private val userInsightActor = context.actorOf(UserInsightActor.props(injector).withDispatcher("insight-client-dispatcher"), "userInsightActor")
  private val checkpointTaskActor = context.actorOf(CheckpointTaskActor.props(injector), "checkpointTaskActor")
  private val publishPushTaskActor = context.actorOf(PublishPushTaskActor.props(injector), "publishPushTaskActor")

  context.watch(buildPushTasksActor)
  context.watch(userInsightActor)
  context.watch(checkpointTaskActor)
  context.watch(publishPushTaskActor)

  override val supervisorStrategy: SupervisorStrategy = OneForOneStrategy() {
    case e: Exception =>
      logger.error(Throwables.getStackTraceAsString(e))
      Resume
  }

  def receive = {
    case pushTaskRequest: PushTaskRequest =>
      logger.info(s" ${this.getClass.getSimpleName} receive message : ${pushTaskRequest.getId} .")
      userInsightActor ! pushTaskRequest
    case pushTaskRequestDto: PushTaskRequestDto =>
      buildPushTasksActor ! pushTaskRequestDto
    case TasksNeedCheckpoint(jobContext: JobContext) =>
      checkpointTaskActor ! jobContext
    case CheckPointedTasks(jobContext: JobContext) =>
      publishPushTaskActor ! jobContext
    case PublishTaskSuccess(jobContext: JobContext) =>
      logger.info(s" ${this.getClass.getSimpleName} publish tasks successful message: ${jobContext.id} .")
      jobContext.maybeSender.foreach(sender_ => sender_ ! jobContext.job.messageId)
  }

}

object EndpointActor {

  def props(injector: Injector): Props = Props[EndpointActor](new EndpointActor(injector))

}
