abstract class BaseTable[M](tag: Tag, tableName: String) extends Table[M](tag, tableName) {

  def id = column[Int]("id", O.PrimaryKey, O.AutoInc)

  def projectId = column[Int]("project_id")

  def createdAt = column[DateTime]("created_at")

  def updatedAt = column[DateTime]("updated_at")

  def state = column[Status]("state")

  def creatorId = column[Int]("creator_id")

  def updaterId = column[Int]("updater_id")

  implicit val AudienceTypeE = MappedColumnType.base[AudienceType, String](
    e => e.toString,
    s => AudienceType.withName(s))

}

abstract class BaseDao[T <: BaseTable[M], M <: BaseModel[M]] extends HasDatabaseConfigProvider[GrowingPostgresDriver] {

  val table: TableQuery[T]
  lazy val insertQuery = table returning table.map(m => m.id) into ((model, id) => model.withId(id))

  def findById(projectId: Int, id: Int): Future[Option[M]] = {
    db.run(findAction(projectId, id).result).map(_.toList.headOption)
  }

  def findByProject(projectId: Int): Future[List[M]] = {
    db.run(table.filter(t => t.projectId === projectId && t.state =!= Status.archived).result).map(_.toList)
  }

  def insert(models: Seq[M]): Future[Seq[M]] = {
    db.run(insertQuery ++= models)
  }

  def insert(model: M): Future[M] = {
    db.run(insertQuery += model)
  }

  def update(projectId: Int, id: Int, model: M): Future[Int] = {
    db.run(findAction(projectId, id).update(model))
  }

  def deleteById(projectId: Int, id: Int): Future[Int] = {
    db.run(findAction(projectId, id).map(_.state).update(Status.archived))
  }

  def batchDelete(models: Seq[M]): Future[Int] = {
    val ids = models.map(_.id.get)
    db.run(table.filter(_.id inSet ids).delete)
  }

  private[this] def findAction(projectId: Int, id: Int) = {
    table.filter(t => t.projectId === projectId && t.id === id && !t.state.inSet(List(Status.archived, Status.deleted)))
  }

}
