package exception

import play.api.libs.ws.WSResponse

/**
 * @author :zhujingdi
 * @since : 2019/11/11
 */
case class WechatAppIdInvalidException(code: Int, message: String) extends IllegalAccessException

object WechatAppIdInvalidException {
  def apply(wsResponse: WSResponse): WechatAppIdInvalidException = {
    val errcode = (wsResponse.json \ "errcode").as[Int]
    val errmsg = "AppId 无效，请重新输入。"
    WechatAppIdInvalidException(errcode, errmsg)
  }
}
