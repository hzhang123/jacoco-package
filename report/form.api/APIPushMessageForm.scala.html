package form.api

case class PushOption(
  apnsProduction: Boolean)

object PushOption {
  implicit val fmt = Json.format[PushOption]
}

case class Notification(
  title: String,
  alert: String,
  extras: Map[String, String],
  actionType: String,
  actionTarget: String,
  actionParameters: Map[String, String]) {

  def genContent: PushContent = {
    PushContent(
      Json.obj(),
      List(Caption.pushTitle(title), Paragraph.pushContent(alert)),
      None)
  }

  def genPushRule(product: Product): PushRuleForm = {
    val target = PushRuleTarget(
      product.uid,
      actionType,
      actionTarget,
      actionParameters)
    PushRuleForm(
      Some(PushRuleType.Custom),
      action = Some("pageOpen"),
      Some(TriggerFilter("", Seq.empty)),
      targets = Some(List(target)),
      limit = Some(1),
      None,
      None,
      Some(Statics.RULE_DEFAULT_START_AT),
      Some(Statics.RULE_DEFAULT_END_AT),
      None,
      Some(86400),
      productIds = Some(List(product.uid)),
      None)
  }
}

object Notification {

  private[this] lazy val configParams = Default.injector.instanceOf[ConfigParams]
  private[this] lazy val captionLimit = configParams.pushMessageCaptionLimit
  private[this] lazy val paragraphLimit = configParams.pushMessageParagraphLimit

  implicit val reads: Reads[Notification] = (
    (JsPath \ "title").read[String](filter[String](JsonValidationError(s"标题不能为空且长度不能超过 $captionLimit 个字符"))(name => name.nonEmpty && name.length <= captionLimit)) and
    (JsPath \ "alert").read[String](filter[String](JsonValidationError(s"内容不能为空且长度不能超过 $paragraphLimit 个字符"))(name => name.nonEmpty && name.length <= paragraphLimit)) and
    (JsPath \ "extras").read[Map[String, String]] and
    (JsPath \ "actionType").read[String] and
    (JsPath \ "actionTarget").read[String] and
    (JsPath \ "actionParameters").read[Map[String, String]])(Notification.apply _)
}

case class APIPushMessageForm(
  cid: String,
  name: String,
  packageName: String,
  /*
  之前通过包名找product,在多产品包名一样的情况下可能会找到多个产品,新增productId,可准确定位到某一个product,
  为了保证接口兼容性，先设计成Option，后续如果API接口升版本了，将productId改成必传，废弃packageName字段
   */
  productId: Option[String],
  audience: List[String],
  options: PushOption,
  notification: Notification) extends APIForm {

  def convertToPushMessageForm(product: Product): PushMessageForm = {
    val ruleForm = notification.genPushRule(product)
    val content = notification.genContent
    PushMessageForm(
      name = name,
      `type` = MessageType.push,
      state = Some(Status.activated),
      platform = MessagePlatform.App,
      audienceType = Some(AudienceType.Segment),
      audienceId = None,
      campaign = None,
      priority = None,
      rule = Some(ruleForm),
      content = Some(content),
      audienceFilter = Some(PushFilterForm("", Seq())))
  }

}

object APIPushMessageForm {
  implicit val reads: Reads[APIPushMessageForm] = (
    (JsPath \ "cid").read[String](filter[String](JsonValidationError("cid 不能为空且长度不能超过 64 个字符"))(name => name.nonEmpty && name.length < 64)) and
    (JsPath \ "name").read[String](filter[String](JsonValidationError("消息名称不能为空且长度不能超过 200 个字符"))(name => name.nonEmpty && name.length < 200)) and
    (JsPath \ "packageName").read[String] and
    (JsPath \ "productId").readNullable[String] and
    (JsPath \ "audience").read[List[String]](filter[List[String]](JsonValidationError("推送目标列表不能为空且长度不能超过 2000"))(audience => audience.nonEmpty && audience.length <= 2000)) and
    (JsPath \ "options").read[PushOption] and
    (JsPath \ "notification").read[Notification])(APIPushMessageForm.apply _)
}
