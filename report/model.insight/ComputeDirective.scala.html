package model.insight

import play.api.libs.json.{ Json, Reads }

/**
 * @author : xhldtc
 * @since : 2019/11/18
 */
final case class ComputeDirective(
  alias: String,
  measurement: Measurement,
  op: Option[String],
  values: Option[Seq[String]],
  attribute: Option[String],
  aggregator: Option[String],
  timeRange: String,
  filter: Option[Filter],
  excluded: Boolean) {

  override def equals(that: Any): Boolean = that match {
    case that: ComputeDirective => (alias == that.alias && measurement == that.measurement && op == that.op
      && values.map(_.sorted) == that.values.map(_.sorted) && attribute == that.attribute && aggregator == that.aggregator
      && timeRange == that.timeRange && filter == that.filter && excluded == that.excluded)
    case _ => false
  }
}

object ComputeDirective {

  implicit val reads: Reads[ComputeDirective] = Json.reads[ComputeDirective]

  implicit val writes = Json.writes[ComputeDirective]
}
