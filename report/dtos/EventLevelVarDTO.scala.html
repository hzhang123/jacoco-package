package dtos

import io.growing.marketing.common.model.PushMetrics
import play.api.libs.json.{ Format, Json }

/**
 *
 * 事件级变量 rpc远程调用返回对象
 *
 * @author wusi
 * @since 2019-04-26 15:34
 */
case class EventLevelVarDTO(id: String, key: String, name: String, description: Option[String], `type`: String)

object EventLevelVarDTO {

  implicit val eventLevelVarDTOFormat: Format[EventLevelVarDTO] = Json.format[EventLevelVarDTO]

  def fromPushMetrics(pushMetrics: PushMetrics): Option[EventLevelVarDTO] = {
    if (pushMetrics.varKey.isEmpty) {
      None
    } else {
      Option(EventLevelVarDTO(pushMetrics.varKey.get, pushMetrics.varKey.get, pushMetrics.varName.get, None, pushMetrics.varType.get))
    }
  }

  def mock: EventLevelVarDTO = {
    EventLevelVarDTO(
      "id",
      "key",
      "name",
      None,
      "type")
  }
}
