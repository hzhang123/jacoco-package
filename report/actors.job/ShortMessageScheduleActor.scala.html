package actors.job

import actors.Injector
import actors.job.ShortMessageScheduleActor.ExpiredTickWay
import akka.actor.Props
import com.typesafe.scalalogging.LazyLogging
import io.growing.micros.business.model.Status
import io.growing.micros.utils.time.DateTimeUtil.UTCNow
import javax.inject.Inject
import infrastructure.lock.LockManager

import scala.concurrent.ExecutionContext.Implicits.global
import scala.concurrent.Future

class ShortMessageScheduleActor @Inject() (
  injector: Injector) extends AbstractTimerActor with LazyLogging {

  override val lockManager: LockManager = injector.lockManager
  override val TimerKey: Any = ExpiredTickWay
  override val delayMillis: Int = 500
  override val intervalSeconds: Int = 30

  override def initJob(): Unit = {}

  override def tickJob(): Future[Any] = {
    for {
      planningMessages <- injector.shortMessageDao.findPlanningShortMessages()
    } yield {
      if (planningMessages.nonEmpty) {
        logger.info(s"${this.getClass.getSimpleName} finds planning short messages: " + planningMessages.map(_.uid).mkString(","))
        planningMessages.map { msg =>
          injector.shortMessageHandler.sendActivatedShortMessages(msg, msg.projectKeyid).flatMap { sendResult =>
            val newMsg = msg.copy(sendResult = Some(sendResult), state = Status.activated, updatedAt = UTCNow)
            injector.shortMessageDao.updateShortMessage(newMsg.dbid, newMsg)
          }
        }
      }
    }
  }
}

object ShortMessageScheduleActor {

  private case object ExpiredTickWay

  def props(
    injector: Injector): Props = Props[ShortMessageScheduleActor](new ShortMessageScheduleActor(injector))
}
