package dao

import io.growing.marketing.common.model.AudienceType.AudienceType
import io.growing.marketing.common.model.{ ShortMessage, SmsProvider }
import io.growing.marketing.common.model.SmsProvider.SmsProvider
import io.growing.micros.business.model.Status
import io.growing.micros.play.slickpg.GrowingPostgresDriver
import io.growing.micros.play.slickpg.GrowingPostgresDriver.api._
import io.growing.micros.utils.time.DateTimeUtil.UTCNow
import javax.inject.Inject
import org.joda.time.DateTime
import play.api.db.slick.{ DatabaseConfigProvider, HasDatabaseConfigProvider }
import slick.lifted.{ ProvenShape, Rep, Tag }

import scala.concurrent.ExecutionContext.Implicits.global
import scala.concurrent.Future

class ShortMessageTable(tag: Tag) extends BaseTable[ShortMessage](tag, "short_messages") {

  implicit val SmsProviderE = MappedColumnType.base[SmsProvider, String](
    l => l.toString,
    r => SmsProvider.withName(r))

  def name: Rep[String] = column[String]("name")

  def projectKeyid: Rep[String] = column[String]("project_keyid")

  def configId: Rep[Int] = column[Int]("config_id")

  def audienceType: Rep[Option[AudienceType]] = column[Option[AudienceType]]("audience_type")

  def audienceId: Rep[Option[String]] = column[Option[String]]("audience_id")

  def signName: Rep[Option[String]] = column[Option[String]]("sign_name")

  def templateCode: Rep[Option[String]] = column[Option[String]]("template_code")

  def templateContent: Rep[Option[String]] = column[Option[String]]("template_content")

  def templateParams: Rep[List[String]] = column[List[String]]("template_params")

  def segAttrs: Rep[List[String]] = column[List[String]]("seg_attrs")

  def testValues: Rep[List[String]] = column[List[String]]("test_values")

  def testPhoneNum: Rep[Option[String]] = column[Option[String]]("test_num")

  def startAt: Rep[DateTime] = column[DateTime]("start_at")

  def sendResult: Rep[Option[String]] = column[Option[String]]("send_result")

  def smsProvider: Rep[SmsProvider] = column[SmsProvider]("sms_provider")

  override def * : ProvenShape[ShortMessage] = (id.?, projectId, projectKeyid, creatorId, updaterId, name, configId, audienceType,
    audienceId, state, signName, templateCode, templateContent, templateParams, segAttrs, testValues, testPhoneNum, startAt, sendResult, smsProvider, createdAt, updatedAt) <>
    ((ShortMessage.apply _).tupled, ShortMessage.unapply)
}

class ShortMessageDao @Inject() (protected val dbConfigProvider: DatabaseConfigProvider)
  extends BaseDao[ShortMessageTable, ShortMessage] with HasDatabaseConfigProvider[GrowingPostgresDriver] {

  override val table = TableQuery[ShortMessageTable]

  override lazy val insertQuery = table returning table.map(_.id) into ((ShortMsg, id) => ShortMsg.withId(id))

  override def insert(model: ShortMessage): Future[ShortMessage] = super.insert(model)

  def updateShortMessage(shortMsgId: Int, model: ShortMessage): Future[Int] = {
    db.run(table.filter(m => m.id === shortMsgId).update(model))
  }

  def findByShortMessageId(shortMsgId: Int): Future[Option[ShortMessage]] = {
    db.run(table.filter(m => m.id === shortMsgId && m.state =!= Status.archived).result).map(_.toList.headOption)
  }

  def findByProjectId(projectId: Int): Future[Seq[ShortMessage]] = {
    db.run(table.filter(m => m.projectId === projectId && m.state =!= Status.archived).sortBy(_.updatedAt.desc).result)
  }

  def deleteShortMessage(shortMsgId: Int): Future[Int] = {
    db.run(table.filter(m => m.id === shortMsgId).map(_.state).update(Status.archived))
  }

  def findPlanningShortMessages(): Future[Seq[ShortMessage]] = {
    db.run(table.filter(m => m.startAt <= UTCNow && m.state === Status.planning).result)
  }

  def findByName(name: String, projectId: Int): Future[Option[ShortMessage]] = {
    db.run(table.filter(m => m.name === name && m.projectId === projectId && m.state =!= Status.archived).result.headOption)
  }

  def findByNameInOthers(name: String, projectId: Int, id: Int): Future[Option[ShortMessage]] = {
    db.run(table.filter(m => m.id =!= id && m.projectId === projectId && m.name === name && m.state =!= Status.archived).result.headOption)
  }
}
