package handlers

import com.google.inject.ImplementedBy
import form.{ AbTestForm, AbTestMessageForm }
import handlers.impl.AbTestHandlerImpl
import io.growing.marketing.common.model.{ AbTest, AbTestMessage, PushMetrics, PushRule }
import io.growing.micros.business.model.{ Product, Project }

import scala.concurrent.Future

@ImplementedBy(classOf[AbTestHandlerImpl])
trait AbTestHandler {
  def buildRulesFromForm(form: AbTestMessageForm, products: Seq[Product], message: AbTestMessage): Seq[PushRule]

  def createAbTest(abTestForm: AbTestForm, project: Project, userId: Int, msgIds: List[Int]): Future[AbTest]

  def createAbTestMessage(form: AbTestMessageForm, project: Project, userId: Int, name: String): Future[AbTestMessage]

  def updateAbTest(form: AbTestForm, abTest: AbTest, abTestId: Int, project: Project, userId: Int,
    historyMsgs: List[Int], latestMsgIds: List[Int]): AbTest

  def updateAbTestMessage(message: AbTestMessage, form: AbTestMessageForm, userId: Int, url: Option[String] = None): AbTestMessage

  def createMetricsIfNecessary(project: Project, userId: Int): Future[PushMetrics]

  def copyMsgAndRuleIfNecessary(form: AbTestForm, abTest: AbTest, abTestMessages: Seq[AbTestMessage]): Future[Seq[AbTestMessage]]

  def updateExpireAbTestMessages(ids: Seq[Int]): Future[AnyRef]
}
