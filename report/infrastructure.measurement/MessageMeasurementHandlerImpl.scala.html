package infrastructure.measurement

import com.typesafe.scalalogging.LazyLogging
import form.BaseMessageForm
import infrastructure.measurement.metrics.MessageMetricsHandler
import io.growing.marketing.common.model.MessageType.MessageType
import io.growing.marketing.common.model.{ BaseMessage, PushMetrics }
import io.growing.micros.business.model.Project
import javax.inject.{ Inject, Singleton }

import scala.concurrent.Future

/**
 * MessageMeasurementHandlerImpl
 *
 * @author damon lin
 *         2019/6/28
 */
@Singleton
class MessageMeasurementHandlerImpl @Inject() (
  messageMetricsHandler: MessageMetricsHandler) extends MessageMeasurementHandler with LazyLogging {

  override def findById(id: Int): Future[Option[PushMetrics]] = {
    messageMetricsHandler.findById(id)
  }

  override def createMetricsIfNecessary(typ: MessageType, project: Project, userId: Int): Future[PushMetrics] = {
    messageMetricsHandler.createMetricsIfNecessary(typ, project, userId)
  }

  override def findInAppMetricByProject(projectId: Int): Future[Option[PushMetrics]] = {
    messageMetricsHandler.findInAppMetricByProject(projectId)
  }

  override def findPushMessageMetricByProject(projectId: Int): Future[Option[PushMetrics]] = {
    messageMetricsHandler.findPushMessageMetricByProject(projectId)
  }
}
