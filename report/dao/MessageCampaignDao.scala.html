package dao

import javax.inject.Inject
import io.growing.marketing.common.model.CampaignType.CampaignType
import io.growing.marketing.common.model.MessagePlatform.MessagePlatform
import slick.lifted.{ Rep, TableQuery, Tag }
import io.growing.marketing.common.model.{ CampaignType, MessageCampaign, MessagePlatform, Sequencing }
import io.growing.micros.business.model.Status
import io.growing.micros.play.slickpg.GrowingPostgresDriver
import io.growing.micros.play.slickpg.GrowingPostgresDriver.api._
import play.api.db.slick.{ DatabaseConfigProvider, HasDatabaseConfigProvider }
import play.api.libs.json.{ JsValue, Json }

import scala.concurrent.Future

/**
 * MessageCampaignDao
 *
 * @author damon lin
 *         2019/8/21
 */

class MessageCampaignTable(tag: Tag) extends BaseTable[MessageCampaign](tag, "message_campaigns") {

  implicit val CampaignTypeEnum = MappedColumnType.base[CampaignType, String](
    e ⇒ e.toString,
    s ⇒ CampaignType.withName(s))

  implicit val SequencingConverting = MappedColumnType.base[List[Sequencing], JsValue](
    e ⇒ Json.toJson(e),
    s ⇒ s.as[List[Sequencing]])

  implicit val MessagePlatformE = MappedColumnType.base[MessagePlatform, String](
    e => e.toString,
    s => MessagePlatform.withName(s))

  def name: Rep[String] = column[String]("name")

  def ai: Rep[String] = column[String]("ai")

  def platform: Rep[MessagePlatform] = column[MessagePlatform]("platform")

  def typ: Rep[CampaignType] = column[CampaignType]("type")

  def productIds: Rep[List[Int]] = column[List[Int]]("product_ids")

  def key: Rep[String] = column[String]("key")

  def sequencing: Rep[List[Sequencing]] = column[List[Sequencing]]("sequencing")

  override def * = (id.?, projectId, ai, name, platform, typ, productIds,
    key, sequencing, creatorId, updaterId, state, createdAt, updatedAt) <> ((MessageCampaign.apply _).tupled, MessageCampaign.unapply)

}

class MessageCampaignDao @Inject() (override protected val dbConfigProvider: DatabaseConfigProvider)
  extends BaseDao[MessageCampaignTable, MessageCampaign] with HasDatabaseConfigProvider[GrowingPostgresDriver] {

  override val table = TableQuery[MessageCampaignTable]

  def findByName(projectId: Int, name: String, other: Option[Int] = None): Future[Option[MessageCampaign]] = {
    if (other.isEmpty) {
      db.run(table.filter(c => c.projectId === projectId && c.name === name && c.state =!= Status.archived).result.headOption)
    } else {
      db.run(table.filter(c => c.projectId === projectId && c.name === name && c.state =!= Status.archived && c.id =!= other.get).result.headOption)
    }
  }

  override def findById(projectId: Int, id: Int): Future[Option[MessageCampaign]] = {
    db.run(table.filter(t => t.projectId === projectId && t.id === id && t.state =!= Status.archived).result.headOption)
  }

  override def update(projectId: Int, id: Int, model: MessageCampaign): Future[Int] = {
    db.run(table.filter(t => t.projectId === projectId && t.id === id && t.state =!= Status.archived).update(model))
  }

  def count(projectId: Int): Future[Int] = {
    db.run(table.filter(t => t.projectId === projectId && t.state === Status.activated).size.result)
  }

}
