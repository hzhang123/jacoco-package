package dao.impl

import io.growing.micros.play.slickpg.GrowingPostgresDriver
import io.growing.micros.business.model.{ Product, Status }
import dao.ProductDao

import scala.concurrent.Future
import dao.component.ProductComponent
import io.growing.micros.play.slickpg.GrowingPostgresDriver.api._
import javax.inject.{ Inject, Singleton }

import scala.concurrent.ExecutionContext.Implicits.global
import play.api.db.slick.{ DatabaseConfigProvider, HasDatabaseConfigProvider }

@Singleton
class ProductDaoImpl @Inject() (protected val dbConfigProvider: DatabaseConfigProvider)
  extends ProductDao with ProductComponent with HasDatabaseConfigProvider[GrowingPostgresDriver] {

  override def findById(id: Int): Future[Option[Product]] = {
    db.run(products.filter(_.id === id).result.headOption)
  }

  override def findByPackageName(projectId: Int, packageName: String): Future[Option[Product]] = {
    db.run(products.filter(p => p.projectId === projectId && p.spn === packageName).result.headOption)
  }

  override def findByIds(projectId: Int, ids: Seq[Int]): Future[Seq[Product]] = {
    db.run(products.filter(p => p.id.inSet(ids) && p.projectId === projectId && p.status === Status.activated).result)
  }

  override def all(): Future[List[Product]] = {
    db.run(products.result.map(_.toList))
  }

}
