package service.insight

import com.typesafe.scalalogging.LazyLogging
import dao.{ ComputeDao, TagComputeDao, TagDao }
import io.growing.micros.business.dao.ProjectDao
import io.growing.micros.utils.time.DateTimeUtil.UTCNow
import javax.inject.{ Inject, Singleton }
import model.insight._
import org.apache.commons.lang3.StringUtils
import play.api.libs.json.Json

import scala.concurrent.{ ExecutionContext, Future }

/**
 * @author : xhldtc
 * @since : 2019/11/18
 */
@Singleton
class TagService @Inject() (
  tagDao: TagDao,
  computeDao: ComputeDao,
  tagComputeDao: TagComputeDao,
  projectDao: ProjectDao,
  computableService: ComputableService)(implicit ec: ExecutionContext) extends LazyLogging {

  def create(projectId: Int, tag: Tag, userType: String, userId: Int): Future[Tag] = {
    for {
      creatingComputes ← generateComputes(tag, userType, userId)
      dbTag ← tagDao.insert(tag)
      computes ← createComputes(creatingComputes)
      tagId = dbTag.id.get
      tagComputes = computes.map(c => TagCompute(None, projectId, tagId, c._1, Some(c._2), userId, userId, tag.state))
      _ ← tagComputeDao.modify(projectId, tagId, tagComputes)
    } yield {
      dbTag
    }
  }

  def update(projectId: Int, tagId: Int, tag: Tag, userTypeChanged: Boolean, userType: String, userId: Int): Future[Tag] = {
    tagDao.findById(projectId, tagId).flatMap {
      case None =>
        //理论上不会进到这个分支，进了代表stage表有tagId,但是关联tag表却没有记录，打个warn新创建一个tag
        logger.warn("tagId exists in stage but not found in tag table, create a tag for fix")
        create(projectId, tag, userType, userId)
      case Some(dbTag) => if (tag.computes.sortBy(_.expression) == dbTag.computes.sortBy(_.expression) && !userTypeChanged) {
        logger.info(s"tag computes is same, only update tag for projectId $projectId and tagId $tagId, tag: ${Json.toJson(tag)}")
        for {
          //即使computes没更新，也要更新computes表的status字段，和stage，tag的状态同步
          tagComputes <- tagComputeDao.findByTagId(projectId, tagId)
          _ <- Future.traverse(tagComputes) { tagCompute => computeDao.updateStatus(projectId, tagCompute.computeId, tag.state, userId) }
          resultTag <- updateTagOnly(projectId, tagId, tag, dbTag, userId)
        } yield resultTag
      } else {
        logger.info(s"tag computes is different, update tag and compute for project $projectId and tag $tagId, userTypeChanged $userTypeChanged" +
          s"db computes: ${Json.toJson(dbTag.computes)}, update computes: ${Json.toJson(tag.computes)}")
        updateTagAndComputes(projectId, tagId, tag, dbTag, userType, userId)
      }
    }
  }

  def updateTagOnly(projectId: Int, tagId: Int, tag: Tag, dbTag: Tag, userId: Int): Future[Tag] = {
    val newTag = dbTag.copy(name = tag.name, description = tag.description, `type` = tag.`type`,
      computes = tag.computes, updaterId = userId, updatedAt = UTCNow, state = tag.state)
    tagDao.updateById(projectId, tagId, newTag).map(_ => newTag)
  }

  def updateTagAndComputes(projectId: Int, tagId: Int, tag: Tag, dbTag: Tag, userType: String, userId: Int): Future[Tag] = {
    for {
      //找到现有tagId下关联的computes，删掉
      tagComputes <- tagComputeDao.findByTagId(projectId, tagId)
      _ <- Future.sequence(tagComputes.map(tagCompute => computeDao.deleteById(projectId, tagCompute.computeId)))
      //更新tag表
      newTag = dbTag.copy(name = tag.name, description = tag.description, `type` = tag.`type`,
        computes = tag.computes, updaterId = userId, updatedAt = UTCNow, state = tag.state)
      _ <- tagDao.updateById(projectId, tagId, newTag)
      //插入新的computes
      creatingComputes ← generateComputes(tag, userType, userId)
      computes ← createComputes(creatingComputes)
      //关联tagCompute
      tagComputes = computes.map(c => TagCompute(None, projectId, tagId, c._1, Some(c._2), userId, userId, tag.state))
      _ ← tagComputeDao.modify(projectId, tagId, tagComputes)
    } yield newTag
  }

  def generateComputes(tag: Tag, userType: String, userId: Int): Future[Seq[(Compute, String)]] = {
    val typ = tag.`type` match {
      case TagType.AGGREGATED ⇒ ComputableType.NUMERICAL
      case TagType.SEGMENT ⇒ ComputableType.SEGMENT
      case _ ⇒ ComputableType.ATTRIBUTE
    }
    projectDao.findKeyId(tag.projectId).flatMap { ai ⇒
      val futureComputes = tag.computes.map { cd ⇒
        computableService.generate(tag.projectId, userType, ai, typ, SchedulerType.DAILY, cd, userId).map(_.copy(state = tag.state) -> cd.name.getOrElse(StringUtils.EMPTY))
      }
      Future.sequence(futureComputes)
    }
  }

  private[this] def createComputes(computes: Seq[(Compute, String)]): Future[Seq[(Int, String)]] = {
    Future.traverse(computes) {
      case (compute, name) ⇒
        computableService.create(compute).map(c ⇒ c.id.get -> name)
    }
  }
}
