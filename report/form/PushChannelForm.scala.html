package form

import play.api.libs.json.Reads.filter
import utils.RegexUtil

case class PushChannelForm(
                            productId: String,
                            name: String,
                            channelName: String,
                            packageName: String,
                            appId: Option[String],
                            appKey: String,
                            appSecret: String,
                          )

object PushChannelForm {

  import play.api.libs.functional.syntax._
  import play.api.libs.json._

  implicit val reads: Reads[PushChannelForm] = (
    (JsPath \ "productId").read[String] and
      (JsPath \ "name").read[String](filter[String](JsonValidationError("名字不能为空，只支持中英文，数字，英文符号，1到25个字符"))
        (name => RegexUtil.nameRegex.findAllIn(name).isEmpty && name.length < 100)) and
      (JsPath \ "channelName").read[String] and
      (JsPath \ "packageName").read[String] and
      (JsPath \ "appId").readNullable[String] and
      (JsPath \ "appKey").read[String] and
      (JsPath \ "appSecret").read[String]
    ) (PushChannelForm.apply _)
}
