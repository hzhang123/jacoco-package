package utils

/**
 * mappingUtil
 *
 * @author damon lin
 *         2018/12/20
 */
object MappingUtil {

  // a util function to convert usv to cs1, uv to u, or convert in other side.
  def segmentIdMapping(maybeId: Option[String], mapping: Map[String, String], idConverter: String => String): Option[String] = {
    maybeId.map(id =>
      s"${id.split(":").lift(0).map(mapping).getOrElse("")}:${id.split(":").lift(1).map(idConverter).getOrElse("")}")
  }

}
