package dao

import io.growing.micros.play.slickpg.GrowingPostgresDriver
import io.growing.micros.play.slickpg.GrowingPostgresDriver.api._
import javax.inject.Inject
import model.UserLifeJourney
import play.api.db.slick.{ DatabaseConfigProvider, HasDatabaseConfigProvider }
import slick.lifted.{ Rep, Tag }

import scala.concurrent.ExecutionContext.Implicits.global
import scala.concurrent.Future

class UserLifeJourneyTable(tag: Tag) extends BaseTable[UserLifeJourney](tag, "user_life_journey") {

  def ai: Rep[String] = column[String]("ai")

  def name: Rep[String] = column[String]("name")

  def stageNum: Rep[Int] = column[Int]("stage_num")

  def userType: Rep[String] = column[String]("user_type")

  override def * = (id.?, projectId, ai, name, stageNum, userType, creatorId, updaterId, state, createdAt, updatedAt) <>
    ((UserLifeJourney.apply _).tupled, UserLifeJourney.unapply)
}

class UserLifeJourneyDao @Inject() (override val dbConfigProvider: DatabaseConfigProvider)
  extends HasDatabaseConfigProvider[GrowingPostgresDriver] {

  val table = TableQuery[UserLifeJourneyTable]

  def insert(m: UserLifeJourney): Future[UserLifeJourney] = {
    db.run((table returning table.map(_.id) into ((model, id) => model.withId(id))) += m)
  }

  def update(userLifeJourney: UserLifeJourney): Future[Int] = {
    db.run(table.filter(_.id === userLifeJourney.id).update(userLifeJourney))
  }

  def findByProjectId(projectId: Int): Future[List[UserLifeJourney]] = {
    db.run(table.filter(_.projectId === projectId).sortBy(_.updatedAt.desc).result).map(_.toList)
  }

  def findByIdAndProject(id: Int, projectId: Int): Future[Option[UserLifeJourney]] = {
    db.run(table.filter(t => t.id === id && t.projectId === projectId).result.headOption)
  }
}
