package dtos

import com.typesafe.scalalogging.LazyLogging
import play.api.libs.json._

/**
 * UserInfo
 *
 * @author damon lin
 *         2019/5/8
 */
case class UserInfo(
  //visitorId就是u,id就是uid, gio为每一个u生成的的int型的id,方便计算用
  id: String,
  domain: String,
  channel: String,
  token: String,
  visitorId: String)

object UserInfo extends LazyLogging {
  implicit val reads: Reads[Seq[UserInfo]] = new Reads[Seq[UserInfo]] {
    override def reads(json: JsValue): JsResult[Seq[UserInfo]] = {
      val data = (json \ "data").asOpt[Seq[Seq[String]]]
      if (data.isEmpty) {
        JsError("")
      } else {
        JsSuccess(valueReads(data.get))
      }
    }
  }

  implicit val writes: Writes[UserInfo] = Json.writes[UserInfo]

  private[this] def valueReads(values: Seq[Seq[String]]): Seq[UserInfo] = {
    values.map { v =>
      try {
        UserInfo(v.head, v(1), v(2), v(3), v(4))
      } catch {
        case e: Exception =>
          logger.error(e.getMessage)
          logger.error(v.mkString(";"))
          throw e
      }
    }
  }
}
