package model.context

import dtos.{ PushMessageDto, UserInfo }
import form.PushMessageForm
import io.growing.marketing.common.model.{ PushMessage, PushMetrics, PushRule }
import io.growing.micros.business.model.Project

/**
 * AppMessageContext
 *
 * @author damon lin
 *         2019/11/7
 */
final case class AppMessageContext(
  userId: Int,
  project: Project,
  override val form: PushMessageForm,
  override val message: Option[PushMessage] = None,
  override val rules: Option[List[PushRule]] = None,
  override val metrics: Option[PushMetrics] = None,
  override val firstUpdateToActivated: Boolean = false,
  override val from: String = "",
  override val users: List[UserInfo] = List.empty,
  override val apnsProd: Boolean = true) extends MessageContext {

  override def getMessage: PushMessage = this.message.get

  override def parameters: (PushMessageForm, Project, Int) = (this.form, this.project, this.userId)

  override def toPresenter: PushMessageDto = {
    PushMessageDto(getMessage, getRules, Some(getMetrics), form.audienceFilter)
  }

}
