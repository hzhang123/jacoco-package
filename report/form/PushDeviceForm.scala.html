package form

import play.api.libs.functional.syntax._
import play.api.libs.json.Reads._
import play.api.libs.json.{ JsPath, JsonValidationError, Reads }
import utils.RegexUtil

case class PushDeviceForm(
  name: String,
  productId: String,
  platform: String,
  brand: String,
  channelId: Option[String],
  channelName: String,
  sdkVersion: String,
  appVersion: String,
  deviceId: String,
  regId: String)

object PushDeviceForm {

  implicit val reads: Reads[PushDeviceForm] = (
    (JsPath \ "name").read[String](filter[String](JsonValidationError("名字不能为空，只支持中英文，数字，英文符号，1到25个字符"))(name => RegexUtil.nameRegex.findAllIn(name).isEmpty && name.length < 100)) and
    (JsPath \ "productId").read[String] and
    (JsPath \ "platform").read[String] and
    (JsPath \ "brand").read[String] and
    (JsPath \ "channelId").readNullable[String] and
    (JsPath \ "channelName").read[String] and
    (JsPath \ "sdkVersion").read[String] and
    (JsPath \ "appVersion").read[String] and
    (JsPath \ "deviceId").read[String] and
    (JsPath \ "regId").read[String])(PushDeviceForm.apply _)
}
