package actors.job

import actors.Injector
import akka.actor.Props
import business.push.PushJobsHandler
import io.growing.marketing.common.model.PushTask
import javax.inject.Inject
import infrastructure.lock.EmptyLockManagerImpl

import scala.concurrent.ExecutionContext.Implicits.global
import scala.concurrent.Future

/**
 * RePublishJobActor
 *
 * @author damon lin
 *         2019/6/28
 */
class RetryJobActor @Inject() (injector: Injector, pushJobsHandler: PushJobsHandler) extends AbstractTimerActor {

  private val pushTaskDao = injector.pushTaskDao
  override val lockManager = new EmptyLockManagerImpl()

  import RetryJobActor._

  val TimerKey: Any = RetryTickKey
  val delayMillis: Int = 500
  val intervalSeconds: Int = 30

  def initJob(): Unit = {
    logger.info("start to schedule retry task background timer")
  }

  def tickJob(): Future[Unit] = {
    for {
      tasks <- pushTaskDao.findNeedRetryTasks()
    } yield {
      Future.traverse(tasks)(retryTask)
    }
  }

  private[this] def retryTask(pushTask: PushTask): Future[Unit] = {
    //用乐观锁去抢占任务，如果更新成功就处理，更新失败代表可能已经被其他节点抢占，就跳过
    pushTaskDao.updateNeedRetryTask(pushTask.taskId).map {
      case x if x == 1 =>
        logger.info(s"success grasp task ${pushTask.taskId} and begin to retry")
        pushJobsHandler.publishPushTask(pushTask)
      case _ => logger.info(s"taskId ${pushTask.taskId} maybe already consumed by other nodes")
    }
  }
}

object RetryJobActor {

  private case object RetryTickKey

  def props(injector: Injector, pushJobsHandler: PushJobsHandler) = Props[RetryJobActor](
    new RetryJobActor(injector, pushJobsHandler))
}
