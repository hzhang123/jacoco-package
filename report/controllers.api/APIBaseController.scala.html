package controllers.api

import com.google.common.util.concurrent.RateLimiter
import dtos.ApiErrorResponse
import io.growing.micros.business.controller.AbstractController
import io.growing.micros.business.dao.ProjectDao
import io.growing.micros.play.exception.GrowingPlayException
import io.growing.micros.utils.conversion.GrowingConversions._
import javax.inject.Inject
import model.auth.Authentication
import model.auth.WrappedRequests.AuthenticationRequest
import play.api.mvc._
import service.auth.AuthenticationService

import scala.collection.mutable
import scala.concurrent.ExecutionContext.Implicits.global
import scala.concurrent.Future.successful
import scala.concurrent.{ ExecutionContext, Future }

/**
 * ApiBaseController
 *
 * @author damon lin
 *         2018/5/17
 */
abstract class APIBaseController @Inject() (
  parse: PlayBodyParsers,
  projectDao: ProjectDao,
  authService: AuthenticationService) extends AbstractController
  with ProjectAuthenticationComponent {

  def Actions[B](projectUid: String, parse: BodyParser[B]) = WithTimeoutAction(parse) andThen new AuthenticationAction(parse) andThen ProjectAuthenticationAction(projectUid.toId)

  private val rateLimiters: mutable.Map[String, RateLimiter] = new mutable.HashMap[String, RateLimiter]()
  private val QpsLimit = 20 // 1200 次/分钟

  class AuthenticationAction[B](parse: BodyParser[B]) extends ActionBuilder[AuthenticationRequest, B] {

    override protected def executionContext: ExecutionContext = global

    override def parser: BodyParser[B] = parse

    def requestAuthInSession[A](request: Request[A]): Future[Option[Authentication]] = {
      val auth = request.headers.get("Authorization").getOrElse("").replaceFirst("Token ", "")
      authService.findByAuth(auth)
    }

    def invokeBlock[A](request: Request[A], block: AuthenticationRequest[A] => Future[Result]) = {
      val clientId = request.headers.get("X-Client-Id").getOrElse("")
      val maybeRateLimiter = rateLimiters.get(clientId)
      if (maybeRateLimiter.isEmpty) {
        rateLimiters.put(clientId, RateLimiter.create(QpsLimit))
      }
      if (rateLimiters(clientId).tryAcquire()) {
        requestAuthInSession(request).flatMap {
          case Some(authentication) =>
            projectDao.findByKey(authentication.provider) flatMap {
              case Some(project) if project.disabled => successful(ApiErrorResponse.PROJECT_DISABLED.notFound)
              case Some(project) => block(new AuthenticationRequest(authentication, project, request))
              case None => successful(ApiErrorResponse.PROJECT_NOT_FOUND.notFound)
            }
          case None => successful(ApiErrorResponse.INVALID_AUTH_TOKEN.forbidden)
        }.recover { //如果业务逻辑里抛Future.failed(exception)出来了，在这统一转api error处理
          case e: GrowingPlayException =>
            logger.error(s"gio exception for request: ${request.body.toString}", e)
            ApiErrorResponse.customBadRequest(e.errorMsg.message)
          //其他未知异常
          case e: Throwable =>
            logger.error(s"unexpected exception for request: ${request.body.toString}", e)
            ApiErrorResponse.INTERNAL_SERVER_ERROR.internalServerError
        }
      } else {
        logger.error(s"Found api request rate limit : $clientId !!!")
        Future.successful(ApiErrorResponse.RATE_LIMIT_DENY.forbidden)
      }
    }
  }

}
