package model.context

import dtos.{ PushMessageDto, UserInfo }
import form.{ BaseMessageForm, PushFilterForm }
import io.growing.marketing.common.model.{ BaseMessage, PushMetrics, PushRule }
import io.growing.micros.business.model.Project

/**
 * MessageContext
 *
 * @author damon lin
 *         2019/11/6
 */
trait MessageContext {

  val userId: Int
  val project: Project
  val form: BaseMessageForm
  val message: Option[BaseMessage] = None
  val rules: Option[List[PushRule]] = None
  val metrics: Option[PushMetrics] = None
  val firstUpdateToActivated: Boolean
  val from: String
  val users: List[UserInfo] = List.empty
  val apnsProd: Boolean = true

  def getMessage: BaseMessage = this.message.get
  def audienceFilter: Option[PushFilterForm] = this.form.audienceFilter
  def getRules: List[PushRule] = this.rules.get
  def getMetrics: PushMetrics = this.metrics.get

  def parameters: (BaseMessageForm, Project, Int) = (this.form, this.project, this.userId)

  def toPresenter: PushMessageDto

}
