package model.insight

import play.api.libs.json.{ Json, Reads }

/**
 * @author : xhldtc
 * @since : 2019/11/18
 */
final case class Directive(
  alias: String,
  expr: String,
  `type`: String,
  op: Option[String],
  values: Option[Seq[String]],
  attribute: Option[String],
  aggregator: Option[String],
  timeRange: String,
  filter: Option[Filter],
  excluded: Boolean)

object Directive {

  implicit val reads: Reads[Directive] = Json.reads[Directive]

  implicit val writes = Json.writes[Directive]
}
