package form

import io.growing.marketing.common.model.{ SmsConfig, SmsProvider }
import io.growing.marketing.common.model.SmsProvider.SmsProvider
import io.growing.micros.business.model.Status
import play.api.libs.functional.syntax._
import play.api.libs.json.{ JsPath, Reads }

final case class SmsConfigForm(
  smsProvider: SmsProvider,
  providerKey: String,
  providerValue: Option[String],
  signNames: List[String],
  numAttribute: Option[String],
  sendPath: Option[String]) {

  def buildSmsConfig(projectId: Int, creatorId: Int, updaterId: Int): SmsConfig = {
    SmsConfig(
      None,
      projectId,
      creatorId,
      updaterId,
      state,
      smsProvider,
      providerKey,
      providerValue,
      signNames,
      numAttribute,
      sendPath)
  }

  def state: Status.Value = {
    smsProvider match {
      case SmsProvider.aliyun =>
        if (providerValue.isDefined) Status.activated else Status.draft
      case SmsProvider.yunpian =>
        Status.activated
      case SmsProvider.chuanglan =>
        if (providerValue.isDefined && sendPath.isDefined) Status.activated else Status.draft
    }
  }
}

object SmsConfigForm {
  implicit val reads: Reads[SmsConfigForm] = (
    (JsPath \ "smsProvider").read[SmsProvider] and
    (JsPath \ "providerKey").read[String] and
    (JsPath \ "providerValue").readNullable[String] and
    (JsPath \ "signNames").read[List[String]] and
    (JsPath \ "numAttribute").readNullable[String] and
    (JsPath \ "sendPath").readNullable[String])(SmsConfigForm.apply _)
}
