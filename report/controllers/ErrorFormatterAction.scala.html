import scala.concurrent.{ ExecutionContext, Future }

class ErrorFormatterAction @Inject() (val parser: BodyParsers.Default)(implicit val executionContext: ExecutionContext)
  extends ActionBuilder[BehindProxyRequest, AnyContent] with Controller with CaseClassJsonSupport {

  def invokeBlock[A](request: Request[A], block: BehindProxyRequest[A] â‡’ Future[Result]): Future[Result] = {

        }
    }
  }
