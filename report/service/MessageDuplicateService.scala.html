package service

import akka.actor.ActorSystem
import cache.RedisCaches
import com.typesafe.scalalogging.LazyLogging
import dtos.UserInfo
import javax.inject.Singleton
import redis.{ RedisClientPool, RedisServer }

import scala.concurrent.ExecutionContext.Implicits.global
import scala.concurrent.Future

/**
 * @author : zhujingdi
 * @since : 2019-12-02
 */
@Singleton
class MessageDuplicateService extends LazyLogging {
  implicit val akkaSystem: ActorSystem = akka.actor.ActorSystem()
  private lazy val host: String = RedisCaches.redisConfigs.getDefaultRedisConfig.host
  private lazy val port: Int = RedisCaches.redisConfigs.getDefaultRedisConfig.port
  private lazy val redisClient = RedisClientPool(Seq(RedisServer(host, port)))

  /**
   * 以 messageUid 为粒度查询设备 uid 是否没有被推送
   *
   * @param token
   * @param flag
   * @return
   */
  def isUnPush(token: String, flag: String): Future[Boolean] = {
    redisClient.sadd("uidCache:" + flag, token).map {
      x =>
        if (x == 1) true else {
          logger.warn(s"flag: $flag token: $token has been pushed, ignore this device")
          false
        }

    }
  }

  /**
   * 对QS查询出来的数据进行去重，去掉 uid 相同的 user
   *
   * @param validUsers qs 查询出来的原始数据
   * @param flag       粒度
   * @return
   */
  def filterUserInfo(validUsers: Seq[UserInfo], flag: String): Future[List[UserInfo]] = {
    Future.sequence(validUsers.map(x => isUnPush(x.token, flag))).map {
      bs => (validUsers zip bs).filter(_._2).map(_._1)
    }.map(result => {
      redisClient.expire("uidCache:" + flag, 3600)
      result.toList
    })
  }

}
