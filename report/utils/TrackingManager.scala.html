package utils

import com.typesafe.scalalogging.LazyLogging
import dao.{ ProductDao, SegmentationDao }
import infrastructure.measurement.metrics.MetricRpcClient
import io.growing.marketing.common.model.{ PushMessage, PushRule }
import javax.inject.Inject

import scala.concurrent.Future
import scala.concurrent.ExecutionContext.Implicits.global

/**
 * TrackingManager
 *
 * @author damon lin
 *         2019/7/30
 */
class TrackingManager @Inject() (
  segmentationDao: SegmentationDao,
  productDao: ProductDao,
  rpcServiceClient: MetricRpcClient) extends LazyLogging {

  def trackingPopupActive(message: PushMessage, maybeRule: Option[PushRule], from: String): Future[_] = {
    Future.traverse(maybeRule.toSeq) {
      rule =>
        val futureMaybeSegName = {
          if (message.audienceId.isEmpty) {
            Future.successful(message.audienceType.map(_.toString))
          } else {
            segmentationDao.findById(message.projectId, message.segmentId.get).map(_.map(_.name))
          }
        }

        val futureMaybeProductName = {
          if (rule.productId.isEmpty) {
            Future.successful(Some("noProduct"))
          } else {
            productDao.findById(rule.productId.get).map(_.map(p => p.platform + "_" + p.name))
          }
        }

        val futureMaybeEventName = {
          if (rule.actionType.isDefined && rule.actionType.get == "system") {
            Future.successful(rule.action)
          } else {
            try {
              rpcServiceClient.queryCustomEvents(rule.action.toSeq, message.projectId).map(_.headOption.map(_.getName))
            } catch {
              case t: Throwable =>
                // ignore
                logger.error("rpcServiceClient catch exception: " + t.getMessage)
                Future.successful(Some("no event"))
            }
          }
        }

        for {
          segName <- futureMaybeSegName
          productName <- futureMaybeProductName
          eventName <- futureMaybeEventName
        } yield {
          val properties = Map(
            "name" -> message.name,
            "platform" -> message.platform.toString,
            "segmentName" -> segName.getOrElse("no seg"),
            "product" -> productName.getOrElse("no product"),
            "triggerTime" -> rule.limit.map(_.toString).getOrElse("no limit"),
            "triggerEvent" -> eventName.getOrElse("no event"),
            "assetURL" -> message.contentUrl.getOrElse("no url"),
            "type" -> from)
          TrackingUtil.traceForGIO("popupActive", message.updaterId.toString, properties)
        }
    }.recover {
      case t: Throwable =>
        // ignore
        logger.error(t.getMessage)
    }
  }

}
