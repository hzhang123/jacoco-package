package dtos

import io.growing.marketing.common.model._
import io.growing.micros.utils.conversion.GrowingConversions._
import org.joda.time.DateTime
import play.api.libs.json.{ JsNumber, JsValue, Json, Writes }

case class AbTestDto(at: AbTest, abMsgDto: Seq[AbTestMessageDto])

object AbTestDto {
  implicit val writes: Writes[AbTestDto] = (dto: AbTestDto) => {
    val abTest = dto.at
    implicit object JodaDateTimeNumberWrites extends Writes[DateTime] {
      def writes(d: DateTime): JsValue = JsNumber(d.getMillis)
    }
    Json.obj(
      "id" -> abTest.id.getOrElse(-1).toUid,
      "name" -> abTest.name,
      "state" -> abTest.state,
      "creatorId" -> abTest.creatorId.toUid,
      "updaterId" -> abTest.updaterId.toUid,
      "createdAt" -> abTest.createdAt,
      "updatedAt" -> abTest.updatedAt,
      "dimension" -> abTest.abDimension,
      "platform" -> MessagePlatform.App,
      "type" -> MessageType.abTest,
      "compareType" -> abTest.compareType,
      "messages" -> dto.abMsgDto.map(Json.toJson(_)))
  }
}
