package service.insight

import java.util.UUID

import dao.ComputeDao
import infrastructure.measurement.metrics.MetricRpcClient
import javax.inject.{ Inject, Singleton }
import model.insight.ComputableType.ComputableType
import model.insight.SchedulerType.SchedulerType
import model.insight.{ Compute, ComputeDefinition, Definition, Directive }
import org.apache.commons.lang3.StringUtils

import scala.concurrent.{ ExecutionContext, Future }

/**
 *
 * @author AI
 *         2019/10/8
 */
@Singleton
class ComputableService @Inject() (
  computeDao: ComputeDao,
  measurementParser: MeasurementParser,
  metricRpcClient: MetricRpcClient)(implicit ec: ExecutionContext) {

  def create(compute: Compute): Future[Compute] = {
    computeDao.insert(compute)
  }

  def generate(projectId: Int, targetUser: String, ai: String, typ: ComputableType, scheduler: SchedulerType, definition: ComputeDefinition, userId: Int): Future[Compute] = {
    val key = UUID.randomUUID().toString.replace("-", StringUtils.EMPTY)
    generateDefinition(projectId, targetUser, typ, definition).map { dft ⇒
      Compute(None, projectId, typ, ai, key, dft, scheduler, 1, userId, userId)
    }
  }

  def generateDefinition(projectId: Int, targetUser: String, typ: ComputableType, definition: ComputeDefinition): Future[Definition] = {
    val futures = definition.directives.map { directive ⇒
      val futureAttributeOpt = directive.attribute match {
        case None ⇒ Future.successful(None)
        case Some(attribute) ⇒ measurementParser.parseAttribute(projectId, attribute).map(Option(_))
      }
      val futureExpression = measurementParser.parseMeasurement(projectId, directive.measurement)
      for {
        expression ← futureExpression
        attributeOpt ← futureAttributeOpt
      } yield {
        Directive(
          alias = directive.alias,
          expr = expression.expression,
          `type` = expression.typ,
          op = directive.op,
          values = directive.values,
          attribute = attributeOpt,
          aggregator = directive.aggregator,
          timeRange = directive.timeRange,
          filter = directive.filter,
          excluded = directive.excluded)
      }
    }
    Future.sequence(futures).map { directives ⇒
      Definition(typ.toString, Some(targetUser), definition.expression, directives)
    }
  }
}
