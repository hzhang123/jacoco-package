package dao

import io.growing.marketing.common.model.AudienceType.AudienceType
import io.growing.marketing.common.model.PushJob
import io.growing.micros.business.model.Status
import io.growing.micros.play.slickpg.GrowingPostgresDriver
import io.growing.micros.play.slickpg.GrowingPostgresDriver.api._
import io.growing.micros.utils.time.DateTimeUtil
import javax.inject.Inject
import org.joda.time.DateTime
import play.api.db.slick.{ DatabaseConfigProvider, HasDatabaseConfigProvider }
import slick.lifted.{ ProvenShape, Rep, TableQuery, Tag }

import scala.concurrent.Future
import scala.concurrent.duration._

/**
 * PushJobDao
 *
 * @author damon lin
 *         2019/5/9
 */
class PushJobsTable(tag: Tag) extends BaseTable[PushJob](tag, "push_jobs") {

  def messageId: Rep[Int] = column[Int]("message_id")

  def ai: Rep[String] = column[String]("ai")

  def audienceId: Rep[Option[Int]] = column[Option[Int]]("audience_id")

  def audienceType: Rep[Option[AudienceType]] = column[Option[AudienceType]]("audience_type")

  def segmentField: Rep[String] = column[String]("segment_field")

  def reason: Rep[String] = column[String]("reason")

  override def * : ProvenShape[PushJob] = (id.?, projectId, ai, messageId, audienceId, audienceType, segmentField, creatorId, updaterId, state, reason, createdAt, updatedAt) <> ((PushJob.apply _).tupled, PushJob.unapply)
}

class PushJobDao @Inject() (override protected val dbConfigProvider: DatabaseConfigProvider)
  extends BaseDao[PushJobsTable, PushJob] with HasDatabaseConfigProvider[GrowingPostgresDriver] {

  override val table = TableQuery[PushJobsTable]

  def findInitialed(): Future[Seq[PushJob]] = {
    db.run(table.filter(job => job.state === Status.initialed).result)
  }

  def findByMessageId(projectId: Int, messageId: Int): Future[Option[PushJob]] = {
    db.run(table.filter(job => job.messageId === messageId && job.state === Status.planning).result.headOption)
  }

  def findDailyJob(msgId: Int, startOfDay: DateTime, endOfDay: DateTime): Future[Option[PushJob]] = {
    db.run(table.filter(job => job.updatedAt >= startOfDay && job.updatedAt <= endOfDay && job.messageId === msgId).result.headOption)
  }

  def findPlanningJobMessageIdsPeriodAgo(duration: Duration): Future[Seq[Int]] = {
    val periodAgo = DateTimeUtil.UTCNow.minus(duration.toMillis)
    db.run(table.filter(job => job.updatedAt <= periodAgo && job.state === Status.planning).map(_.messageId).distinct.result)
  }

  override def update(projectId: Int, id: Int, model: PushJob): Future[Int] = {
    db.run(table.filter(job => job.projectId === projectId && job.id === id).update(model))
  }

  def updateToFinishedByMessageIds(messageIds: Seq[Int]): Future[Int] = {
    db.run(table.filter(job => job.messageId.inSet(messageIds)).map(_.state).update(Status.finished))
  }
}
