package form

import play.api.libs.json.{ Json, Writes }

/**
 * @author wusi
 * @since 2019-05-20 17:52
 */
case class EventForm(
  key: String,
  name: String,
  description: Option[String],
  valueType: String,
  businessType: String,
  attributes: List[Attribute]) {
}

object EventForm {

  implicit val eventFormWrites: Writes[EventForm] = (eventForm: EventForm) => Json.obj(
    "key" -> eventForm.key,
    "name" -> eventForm.name,
    "description" -> eventForm.description,
    "valueType" -> eventForm.valueType,
    //    "businessType" -> eventForm.businessType,
    "attributes" -> eventForm.attributes)
}
