package dao

import com.typesafe.scalalogging.LazyLogging
import io.growing.marketing.common.model.AbTest
import io.growing.micros.business.model.Status
import io.growing.micros.play.slickpg.GrowingPostgresDriver
import io.growing.micros.play.slickpg.GrowingPostgresDriver.api._
import javax.inject.Inject
import play.api.db.slick.{ DatabaseConfigProvider, HasDatabaseConfigProvider }
import slick.lifted.{ Rep, Tag }

import scala.concurrent.Future
import scala.concurrent.ExecutionContext.Implicits.global

class AbTestTable(tag: Tag) extends BaseTable[AbTest](tag, "ab_test") {
  def ai: Rep[String] = column("ai")

  def name: Rep[String] = column("name")

  def messages: Rep[List[Int]] = column("messages")

  def latestMessages: Rep[List[Int]] = column("latest_messages")

  def abDimension: Rep[Option[String]] = column("ab_dimension")

  def compareType: Rep[String] = column("compare_type")

  @Override
  def * = (id.?, ai, projectId, creatorId, updaterId, state, name, messages,
    latestMessages, abDimension, compareType, createdAt, updatedAt) <> ((AbTest.apply _).tupled, AbTest.unapply)
}

class AbTestDao @Inject() (override val dbConfigProvider: DatabaseConfigProvider)
  extends HasDatabaseConfigProvider[GrowingPostgresDriver] with LazyLogging {

  val table = TableQuery[AbTestTable]

  val abTestMessage = TableQuery[AbTestMessageTable]

  def insert(m: AbTest): Future[AbTest] = {
    db.run((table returning table.map(_.id) into ((model, id) => model.withId(id))) += m)
  }

  def update(m: AbTest, id: Int): Future[Int] = {
    db.run(table.filter(t => t.id === id && t.state =!= Status.archived).update(m))
  }

  def findByName(projectId: Int, name: String): Future[Option[AbTest]] = {
    db.run(table.filter(t => t.projectId === projectId && t.name === name && t.state =!= Status.archived).result.headOption)
  }

  def findByNameInOthers(projectId: Int, id: Int, name: String): Future[Option[AbTest]] = {
    db.run(table.filter(t => t.projectId === projectId && t.id =!= id && t.name === name && t.state =!= Status.archived).result.headOption)
  }

  def findById(projectId: Int, id: Int): Future[Option[AbTest]] = {
    db.run(table.filter(t => t.id === id && t.projectId === projectId && t.state =!= Status.archived).result.headOption)
  }

  def findByProjectId(projectId: Int): Future[Seq[AbTest]] = {
    db.run(table.filter(t => t.projectId === projectId && t.state =!= Status.archived).sortBy(_.updatedAt.desc).result)
  }

  def updatePlanningToActivated(ids: Seq[Int]): Future[Int] = {
    db.run(abTestMessage.filter(a => a.id inSet ids).map(_.dimension.getOrElse("")).result).flatMap { l =>
      db.run(table.filter(t => t.abDimension.inSet(l) && t.state === Status.planning).map(_.state).update(Status.activated))
    }
  }

  def updateExpiredAbTest(ids: Seq[Int]): Future[List[Int]] = {
    val abTestMessageGroup = db.run(abTestMessage.filter(a => a.id inSet ids).result).map { m =>
      m.groupBy(_.dimension).values
    }.map(_.toList)

    val abTestGroup = abTestMessageGroup.flatMap { m =>
      Future.sequence(m.map { n =>
        db.run(table.filter(_.abDimension inSet n.map(_.dimension.getOrElse(""))).result.head)
      })
    }.map(_.toList)

    val aux = for {
      a <- abTestGroup
      b <- abTestMessageGroup
    } yield a.zip(b)

    aux.flatMap { a =>
      Future.sequence(a.map { pair =>
        val latestMessages = pair._2.map(_.dbid).toList
        val messages = pair._1.messages ++ latestMessages
        db.run(table.filter(t => t.id === pair._1.id.get).map(t => (t.state, t.latestMessages, t.messages)).update((Status.stop, latestMessages, messages)))
      })
    }
  }

  def deleteById(projectId: Int, id: Int): Future[Int] = {
    db.run(table.filter(t => t.id === id && t.projectId === projectId && t.state =!= Status.archived).map(_.state).update(Status.archived))
  }
}
