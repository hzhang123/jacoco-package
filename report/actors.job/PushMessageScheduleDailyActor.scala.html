package actors.job

import actors.Injector
import actors.job.PushMessageScheduleDailyActor.OnTime
import akka.actor.Props
import business.push.PushJobsHandler
import com.typesafe.scalalogging.LazyLogging
import dao.{ PushJobDao, PushMessageDao, PushRuleDao }
import javax.inject.Inject
import org.joda.time.DateTime

import scala.concurrent.ExecutionContext.Implicits.global
import scala.concurrent.Future

/**
 * 循环定时推送
 * 有 rule 对应 startAt < 当前时间 < endAt, scheduleAt 字段不为空
 *
 * @param pushMessageDao
 * @param pushRuleDao
 * @param pushJobsHandler
 * @param pushJobDao
 */
class PushMessageScheduleDailyActor @Inject() (
  pushMessageDao: PushMessageDao,
  pushRuleDao: PushRuleDao,
  pushJobsHandler: PushJobsHandler,
  injector: Injector,
  pushJobDao: PushJobDao) extends AbstractTimerActor with LazyLogging {

  val lockManager = injector.lockManager
  val TimerKey: Any = OnTime
  val delayMillis: Int = 600
  val intervalSeconds: Int = 30

  override def initJob(): Unit = {

  }

  override def tickJob(): Future[Any] = {
    val expiredMsgIds = pushMessageDao.findExpiredMessages()
    expiredMsgIds.flatMap(ids => pushMessageDao.updateActivatedToStop(ids))
    val startOfDay = new DateTime().toLocalDate.toDateTimeAtStartOfDay(new DateTime().getZone)
    val endOfDay = new DateTime().toLocalDate.plusDays(1).toDateTimeAtStartOfDay(new DateTime().getZone)
    pushMessageDao.findDailyPushMessage().map { dailyJobMessages =>
      if (dailyJobMessages.nonEmpty) {
        val ids = dailyJobMessages.map(_.dbid)
        logger.warn(s" ${this.getClass.getSimpleName} find daily push message: ${dailyJobMessages.map(_.uid).mkString(",")} !")
        // 如果有状态为 planning 的 message, 则将状态置为 activated 即完成第一次上线
        // daily message 查询条件不包含 message 的状态
        pushMessageDao.updatePlanningToActivated(ids)
        pushRuleDao.updateActivatedAt(ids)
        ids.map { msgId =>
          // 如果当天没有生成过 job 则判断为当天第一次执行
          pushJobDao.findDailyJob(msgId, startOfDay, endOfDay).flatMap {
            case None => pushMessageDao.findByMsgId(msgId).map(m => pushJobsHandler.publishPushJobBySegment(m))
          }
        }
      }
    }
  }
}

object PushMessageScheduleDailyActor {

  private case object OnTime

  def props(
    pushMessageDao: PushMessageDao,
    pushRuleDao: PushRuleDao,
    pushJobsHandler: PushJobsHandler,
    injector: Injector,
    pushJobDao: PushJobDao): Props =
    Props[PushMessageScheduleDailyActor](new PushMessageScheduleDailyActor(pushMessageDao, pushRuleDao, pushJobsHandler, injector, pushJobDao))
}
