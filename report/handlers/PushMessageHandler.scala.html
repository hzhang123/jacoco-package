package handlers

import com.google.inject.ImplementedBy
import form.{ PushFilterForm, PushMessageForm, PushRuleForm, PushRuleTarget }
import handlers.impl.PushMessageHandlerImpl
import io.growing.marketing.common.model.MessagePlatform.MessagePlatform
import io.growing.marketing.common.model.{ BaseMessage, PushFilter, PushMessage, PushMetrics, PushRule }
import io.growing.micros.business.model.{ Product, Status }
import model.context.MessageContext

import scala.concurrent.Future

/**
 * pushMessageHandler
 *
 * @author damon lin
 *         2019/6/12
 */
@ImplementedBy(classOf[PushMessageHandlerImpl])
trait PushMessageHandler {

  type RULES = Seq[PushRule]

  def buildRulesFromForm(form: PushMessageForm, products: Seq[Product], message: PushMessage): Seq[PushRule]

  def getProductsFromRuleForm(projectId: Int, ruleForm: PushRuleForm, platform: MessagePlatform): Future[Seq[Product]]

  def buildRuleTargetFromContentTarget(links: Seq[String]): Option[PushRuleTarget]

  def validateMessageRunnable(pushMessageForm: PushMessageForm): Either[String, Boolean]

  def diffRules(message: BaseMessage, newRules: Seq[PushRule], oldRules: Seq[PushRule]): (RULES, RULES, RULES)

  def calState(pushMessage: PushMessage, rulesToCreate: Seq[PushRule], rulesToUpdate: Seq[PushRule]): PushMessage

  def copyNewMessage(userId: Int, message: PushMessage, f: PushMessageForm, maybeURL: Option[String], maybePushMetrics: Option[PushMetrics], filter: PushFilter): PushMessage

  def adjustMsgForm(messageForm: PushMessageForm): PushMessageForm

  def determineTimeLegality(messageForm: PushMessageForm): Future[Unit]

  def convertFormToFilter(form: Option[PushFilterForm]): PushFilter

  def convertFilterToForm(filter: PushFilter): Option[PushFilterForm]

  def findDuplicatedName(name: String, projectId: Int, id: Option[Int] = None): Future[Unit]

  def validatePushState(context: MessageContext, messageId: Int, expectState: Status.Value): Future[Unit]
}
