package controllers.api

import io.growing.micros.business.dao.ProjectDao
import model.auth.Authentication
import model.auth.WrappedRequests.AuthenticationRequest
import play.api.libs.json.{ JsArray, Json }
import play.api.mvc._
import service.auth.AuthenticationService

import scala.concurrent.{ ExecutionContext, Future }
import scala.concurrent.Future.{ successful => resolve }

trait AuthenticationActionComponent {

  import play.api.libs.concurrent.Execution.Implicits.defaultContext

  val authenticationService: AuthenticationService = ???
  val projectService: ProjectDao = ???

  private lazy val projectDisabledResult = Results.NotFound(Json.obj("message" -> "项目已被禁用", "errros" -> JsArray()))
  private lazy val projectNotFoundResult = Results.NotFound(Json.obj("message" -> "项目未找到", "errors" -> JsArray()))
  private lazy val invalidAuthResult = Results.Forbidden(Json.obj("message" -> "无效的认证码", "errors" -> JsArray()))

  class AuthenticationAction[B](parse: BodyParser[B]) extends ActionBuilder[AuthenticationRequest, B] {

    override protected def executionContext: ExecutionContext = defaultContext
    override def parser: BodyParser[B] = parse

    def requestAuthInSession[A](request: Request[A]): Future[Option[Authentication]] = {
      val auth = request.headers.get("Authorization").getOrElse("").replaceFirst("Token ", "")
      authenticationService.findByAuth(auth)
    }

    def invokeBlock[A](request: Request[A], block: AuthenticationRequest[A] => Future[Result]) = {
      requestAuthInSession(request) flatMap {
        case Some(authentication) =>
          projectService.findByKey(authentication.provider) flatMap {
            case Some(project) if project.disabled => resolve(projectDisabledResult)
            case Some(project) => block(new AuthenticationRequest(authentication, project, request))
            case None => resolve(projectNotFoundResult)
          }
        case None => resolve(invalidAuthResult)
      }
    }
  }
}
