package exception

import play.api.libs.ws.WSResponse

/**
 * @author :zhujingdi
 * @since : 2019/11/11
 */
case class WechatInvalidTokenException(code: Int, message: String) extends IllegalAccessException

object WechatInvalidTokenException {
  def apply(wsResponse: WSResponse): WechatInvalidTokenException = {
    val errcode = (wsResponse.json \ "errcode").as[Int]
    val errmsg = "凭据无效，access_token无效或不是最新的。"
    WechatInvalidTokenException(errcode, errmsg)
  }
}
