package presenters

import dtos.{ MessageCampaignDto, PushMessageDto }
import io.growing.marketing.common.model.{ MessageCampaign, Sequencing }
import play.api.libs.json._

/**
 * MessageCampaignPresenter
 *
 * @author damon lin
 *         2019/8/21
 */
final case class MessageCampaignPresenter(
  messageCampaign: MessageCampaign,
  messages: List[PushMessageDto])

object MessageCampaignPresenter {

  implicit val fmt = new Format[MessageCampaignPresenter] {
    override def reads(json: JsValue): JsResult[MessageCampaignPresenter] = {
      JsError("Not implemented yet!")
    }

    override def writes(o: MessageCampaignPresenter): JsValue = {
      val mps = o.messages.map(_.toPresenter)
      val c = o.messageCampaign
      val cp = Json.toJson(MessageCampaignDto(c))
      cp.as[JsObject] ++ Json.obj("messages" -> Json.toJson(mps))
    }
  }
}
