package dao

import io.growing.micros.play.slickpg.GrowingPostgresDriver
import io.growing.micros.play.slickpg.GrowingPostgresDriver.api._
import javax.inject.{ Inject, Singleton }
import model.UserLifeJourneyStage
import play.api.db.slick.{ DatabaseConfigProvider, HasDatabaseConfigProvider }
import slick.lifted.{ Rep, TableQuery, Tag }

import scala.concurrent.ExecutionContext.Implicits.global
import scala.concurrent.Future

class UserLifeJourneyStageTable(tag: Tag) extends BaseTable[UserLifeJourneyStage](tag, "user_life_journey_stage") {

  def journeyId: Rep[Int] = column[Int]("journey_id")

  def description: Rep[String] = column[String]("description")

  def name: Rep[String] = column[String]("name")

  def tagId: Rep[Option[Int]] = column[Option[Int]]("tag_id")

  def rank: Rep[Int] = column[Int]("rank")

  override def * = (id.?, projectId, journeyId, description, name, tagId, rank, creatorId, updaterId, state, createdAt, updatedAt) <>
    ((UserLifeJourneyStage.apply _).tupled, UserLifeJourneyStage.unapply)
}

@Singleton
case class UserLifeJourneyStageDao @Inject() (override val dbConfigProvider: DatabaseConfigProvider)
  extends HasDatabaseConfigProvider[GrowingPostgresDriver] {

  val table = TableQuery[UserLifeJourneyStageTable]

  def insert(m: UserLifeJourneyStage): Future[UserLifeJourneyStage] = {
    db.run((table returning table.map(_.id) into ((model, id) => model.withId(id))) += m)
  }

  def update(userLifeJourneyStage: UserLifeJourneyStage): Future[Int] = {
    db.run(table.filter(_.id === userLifeJourneyStage.id).update(userLifeJourneyStage))
  }

  def findByJourneyId(journeyId: Int): Future[List[UserLifeJourneyStage]] = {
    db.run(table.filter(_.journeyId === journeyId).sortBy(_.updatedAt.desc).result).map(_.toList)
  }

  def findById(stageId: Int): Future[UserLifeJourneyStage] = {
    db.run(table.filter(_.id === stageId).result.head)
  }

}
