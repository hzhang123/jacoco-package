package service.impl

import java.util.UUID

import cache.RedisCaches
import io.growing.micros.middleware.cache.CacheConfig
import service.TokenService

import scala.concurrent.Future
import scala.concurrent.duration._

/**
 * @author wusi
 * @since 2019-06-05 17:10
 */
class TokenServiceImpl extends TokenService {

  private lazy val tokenCache = RedisCaches.newRedisCache[String](CacheConfig("tokenCache", Some(1.hour.toSeconds)))

  override def applyToken(userId: Option[String], projectUid: String): Future[Option[String]] = {
    userId match {
      case Some(id) ⇒
        val randomToken = UUID.randomUUID().toString
        tokenCache.put(enrichCacheKey(id, projectUid, randomToken), randomToken)
        Future.successful(Some(randomToken))
      case None ⇒ Future.successful(None)
    }
  }

  private def enrichCacheKey(key: String, projectUid: String, token: String): String = {
    s"applyToken$key$projectUid$token"
  }
}
