package controllers
import com.typesafe.scalalogging.LazyLogging
import io.growing.marketing.common.model.EditionStatus.EditionStatus
import io.growing.marketing.common.model.EditionType.EditionType
import io.growing.marketing.common.model.{MarketingEdition, MarketingLead}
import io.growing.micros.business.auth._
import io.growing.micros.business.controller.{AbstractController, ProjectActionComponent}
import io.growing.micros.play.exception.BadRequestException
import io.growing.micros.play.mvc.{FutureUtils, TimeoutAction}
import io.growing.micros.utils.conversion.GrowingConversions._
import javax.inject.{Inject, Singleton}
import org.joda.time.DateTime
import play.api.mvc.Results.Ok
import play.api.mvc.{BodyParser, PlayBodyParsers}
import service.MarketingEditionService
import play.api.libs.json._
import play.api.libs.json.Reads._
import play.api.libs.functional.syntax._

import scala.concurrent.ExecutionContext.Implicits.global
import scala.concurrent.Future

@Singleton
class MarketingEditionController @Inject()(
	marketingEditionService: MarketingEditionService,
	ac: TimeoutAction,
	errorFormatterAction: ErrorFormatterAction,
	parse: PlayBodyParsers,
) extends AbstractController with ProjectActionComponent with ACLControlComponent with LazyLogging with FutureUtils {

	val readPermission: String = AuthConstants.Marketing.Read
	val managePermission: String = AuthConstants.Marketing.Manage

	def Actions[B](projectUid: String, parse: BodyParser[B]) = {
		WithTimeoutAction(parse) andThen errorFormatterAction andThen new AuthAction(parse) andThen
			ProjectAction(projectUid.toId)
	}

	def find(projectUid: String, organizationUid: String) = Actions(projectUid, parse.empty).async { implicit request =>
		val projectId = projectUid.toId
		val organizationId = organizationUid.toId
		logger.info(s"MarketingEditionController get find request: $projectId $organizationId .")
		marketingEditionService.findByOrganizationId(organizationId).flatMap(me ⇒ {
			if (me.isEmpty) {
				marketingEditionService.insert(MarketingEdition.default(organizationId, request.userId))
			} else {
				Future.successful(me.get)
			}
		}).map(edition ⇒ {
			Ok(Json.toJson(MarketingEditionPresenter(edition.id.get.toUid, edition.organizationId.toUid, edition.status, edition.applyAt, edition.expiredAt, edition.`type`)))
		})
	}

	case class MarketingEditionPresenter(
		id: String,
		organizationId: String,
		status: Option[EditionStatus],
		applyAt: Option[DateTime],
		expiredAt: Option[DateTime],
		`type`: Option[EditionType]
	)
	object MarketingEditionPresenter {
		implicit val writes: Writes[MarketingEditionPresenter] = (o: MarketingEditionPresenter) => {
			Json.obj(
				"id" -> o.id,
				"organizationId" -> o.organizationId,
				"status" -> o.status,
				"applyAt" -> o.applyAt.map(_.getMillis),
				"expiredAt" -> o.expiredAt.map(_.getMillis),
				"type" -> o.`type`
			)
		}
	}

	/**
	 * @todo 存储leads
	 * @param organizationUid
	 * @param projectUid
	 * @return
	 */
	def applyTrial(projectUid: String, organizationUid: String) = Actions(projectUid, parse.empty).async {implicit request =>
		val organizationId = organizationUid.toId
		val projectId = projectUid.toId
		for {
			edition ← marketingEditionService.findOrCreate(organizationId, request.userId)
			_ ← failCondition(edition.`type`.isDefined, BadRequestException("请勿重复申请"))
			_ ← marketingEditionService.createLead(MarketingLead.trailLead(organizationId, projectId, request.userId, None, None, request.headers.get("referer").getOrElse("https://www.growingio.com")))
			_ <- marketingEditionService.applyTrial(edition, request.userId)
		} yield Ok("")
	}

	case class MarketingEditionForm(
		userPhone: Option[String],
		comment: Option[String],
		userName: Option[String]
	)

	object MarketingEditionForm {
		implicit val reads: Reads[MarketingEditionForm] = (
			(JsPath \ "userPhone").readNullable[String] and
				(JsPath \ "comment").readNullable[String] and
				(JsPath \ "userName").readNullable[String])(MarketingEditionForm.apply _)
	}

	/**
	 * @todo 购买过程，我们不修改状态，只上报 leads
	 * @param organizationUid
	 * @param projectUid
	 * @return
	 */
	def applyPurchase(projectUid: String, organizationUid: String) = Actions(projectUid, parse.tolerantJson).async(parse.tolerantJson) { implicit request ⇒
		request.body.validate[MarketingEditionForm].fold(
			errors => jsonBadRequest(errors),
			form ⇒ {
				val organizationId = organizationUid.toId
				val projectId = projectUid.toId
				val lead = MarketingLead.purchaseLead(organizationId, projectId, request.userId, form.userName, form.userPhone, request.headers.get("referer").getOrElse("https://www.growingio.com"), form.comment)
				marketingEditionService.delayTrail(organizationId)
				marketingEditionService.createLead(lead).map(_ ⇒ Ok(""))
			}
		)
	}

}
