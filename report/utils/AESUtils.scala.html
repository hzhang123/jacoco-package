package utils

import java.nio.charset.StandardCharsets
import java.util.Base64
import com.typesafe.scalalogging.LazyLogging
import javax.crypto.Cipher
import javax.crypto.spec.SecretKeySpec
import org.bouncycastle.jce.provider.BouncyCastleProvider

/**
 * @author : xhldtc
 * @since : 2019/10/10
 */
object AESUtils extends LazyLogging {

  /**
   * do not change the seed after you apply the encrypt and decrypt function to real data, otherwise the decryption
   * will be broken
   */
  private val SEED = "0d00635a0ce64a69"
  private val key = initKey(SEED)

  private val localCipher: ThreadLocal[Cipher] = ThreadLocal.withInitial(() => Cipher.getInstance("AES/ECB/PKCS5Padding", new BouncyCastleProvider()))

  def initKey(seed: String): SecretKeySpec = {
    val bytes = SEED.getBytes
    new SecretKeySpec(bytes, "AES")
  }

  /**
   * get the string bytes by utf-8, then base64 encode the encrypt result bytes
   *
   * @param plainText
   * @return
   */
  def encrypt(plainText: String): String = try {
    val bytes = plainText.getBytes(StandardCharsets.UTF_8)
    val cipher = localCipher.get()
    cipher.init(Cipher.ENCRYPT_MODE, key)
    val encrypted = cipher.doFinal(bytes)
    Base64.getEncoder.encodeToString(encrypted)
  } catch {
    case e: Throwable =>
      logger.error("encrypt failed", e)
      plainText
  }

  /**
   * get the decrypted bytes and return utf-8 string
   *
   * @param secretText : base64 format of encrypted bytes
   * @return
   */
  def decrypt(secretText: String): String = try {
    // 对于对未加密过的内容进行解密的时候
    // 会产生异常，这里采用捕获的方式进行处理
    val decoded = Base64.getDecoder.decode(secretText)
    val cipher = localCipher.get()
    cipher.init(Cipher.DECRYPT_MODE, key)
    val decrypted = cipher.doFinal(decoded)
    new String(decrypted, StandardCharsets.UTF_8)
  } catch {
    case e: Throwable =>
      logger.warn("unencrypted field, cannot be decrypted, return original appSecret:" + secretText)
      secretText
  }
}
