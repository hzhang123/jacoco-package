package dtos

import io.growing.marketing.common.model.ShortMessage
import io.growing.micros.utils.conversion.GrowingConversions._
import org.joda.time.DateTime
import play.api.libs.json.{ JsNumber, JsValue, Json, Writes }

case class ShortMessageDto(shortMessage: ShortMessage)

object ShortMessageDto {

  implicit object JodaDateTimeNumberWrites extends Writes[DateTime] {
    def writes(d: DateTime): JsValue = JsNumber(d.getMillis)
  }

  implicit val writes: Writes[ShortMessageDto] = (dto: ShortMessageDto) => {
    val m = dto.shortMessage
    var base = Json.obj(
      "id" -> m.id.map(_.toUid),
      "configId" -> m.configId.toUid,
      "creatorId" -> m.creatorId.toUid,
      "updaterId" -> m.updaterId.toUid,
      "createdAt" -> m.createdAt,
      "updatedAt" -> m.updatedAt,
      "audienceType" -> m.audienceType,
      "state" -> m.state,
      "name" -> m.name,
      "templateParams" -> m.templateParams,
      "segAttrs" -> m.segAttrs,
      "testValues" -> m.testValues,
      "startAt" -> m.startAt,
      "smsProvider" -> m.smsProvider)

    m.audienceId.foreach(i => base ++= Json.obj("audienceId" -> i))
    m.signName.foreach(n => base ++= Json.obj("signName" -> n))
    m.templateCode.foreach(c => base ++= Json.obj("templateCode" -> c))
    m.templateContent.foreach(c => base ++= Json.obj("templateContent" -> c))
    m.testPhoneNum.foreach(n => base ++= Json.obj("testPhoneNum" -> n))
    m.sendResult.foreach(r => base ++= Json.obj("sendResult" -> r))
    base
  }
}
