package dao

import io.growing.micros.business.model.Status
import io.growing.micros.business.model.Status.Status
import javax.inject.Inject
import model.Segmentation
import org.joda.time.DateTime

import scala.concurrent.Future
import slick.lifted.{ TableQuery, Tag }

import scala.concurrent.ExecutionContext.Implicits.global
import io.growing.micros.play.slickpg.GrowingPostgresDriver
import io.growing.micros.play.slickpg.GrowingPostgresDriver.api._
import play.api.db.slick.{ DatabaseConfigProvider, HasDatabaseConfigProvider }

/**
 * SegmentationDao
 *
 * @author damon lin
 *         2019/5/9
 */

class SegmentationTable(tag: Tag) extends BaseTable[Segmentation](tag, "user_segmentations") {

  def ai = column[String]("ai")

  def name = column[String]("name")

  /**
   * 用户类型    注意和userRange的区别
   * cs1 -> 登录用户
   * u -> 访问用户
   */
  def field = column[String]("field")

  /**
   * 用户类型 uv, usv, nuv, nusv
   */
  def userScope = column[String]("user_scope")

  def calStatus = column[String]("cal_status")

  def calAt = column[Option[DateTime]]("cal_at")

  def userNum = column[Option[Int]]("user_num")

  def status = column[Status]("status")

  override def * = (id.?, projectId, ai, name, userScope, creatorId, updaterId, createdAt, updatedAt, status, calAt, userNum) <> ((Segmentation.apply _).tupled, Segmentation.unapply)

}

class SegmentationDao @Inject() (override protected val dbConfigProvider: DatabaseConfigProvider)
  extends BaseDao[SegmentationTable, Segmentation] with HasDatabaseConfigProvider[GrowingPostgresDriver] {

  override val table = TableQuery[SegmentationTable]

  override def findById(projectId: Int, id: Int): Future[Option[Segmentation]] = {
    db.run(table.filter(t => t.projectId === projectId && t.id === id && t.status === Status.activated).result).map(_.toList.headOption)
  }

}
