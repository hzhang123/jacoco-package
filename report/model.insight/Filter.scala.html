package model.insight

import play.api.libs.json.{ Json, Reads }

/**
 *
 * @author AI
 *         2019/11/7
 */
final case class Filter(
  key: Option[String],
  name: Option[String],
  op: Option[String],
  values: Option[Seq[String]],
  exprs: Option[Seq[Filter]]) {

  override def equals(that: Any): Boolean = that match {
    case that: Filter => {
      (key == that.key && name == that.name && op == that.op && values.map(_.sorted) == that.values.map(_.sorted)
        && exprs.map(_.sortBy(_.key)) == that.exprs.map(_.sortBy(_.key)))
    }
    case _ => false
  }
}

object Filter {
  implicit val reads: Reads[Filter] = Json.reads[Filter]

  implicit val writes = Json.writes[Filter]
}
