case class SequencingForm(
  messageId: String,
  index: Int,
  priority: Int) {

  def toSequencing: Sequencing = {
    Sequencing(
      messageId.toId,
      index,
      priority)
  }
}

object SequencingForm {

  implicit val fmt = Json.format[SequencingForm]

  def fromSeq(seq: Sequencing): SequencingForm = {
    SequencingForm(
      seq.messageId.toUid,
      seq.index,
      seq.priority)
  }

}
final case class MessageCampaignForm(
  name: String,
  `type`: CampaignType,
  productIds: List[String],
  platform: MessagePlatform,
  sequencing: Option[List[SequencingForm]],
  state: Status,
  key: Option[String]) {

  def toMessCampaign(project: Project, creatorId: Int, updaterId: Int): MessageCampaign = {
    val s = sequencing.map(seq => seq.map(s => s.toSequencing)).getOrElse(List.empty)
    MessageCampaign(
      None,
      project.dbid,
      project.keyid,
      name,
      platform,
      `type`,
      productIds.map(_.toId),
      key.getOrElse(UUID.randomUUID().getLeastSignificantBits.toHexString),
      s,
      creatorId,
      updaterId,
      state)
  }

  def isActivated: Boolean = state == Status.activated

}

object MessageCampaignForm {

  implicit val reads: Reads[MessageCampaignForm] = (
    (JsPath \ "name").read[String](filter[String](JsonValidationError("名字不能为空，只支持中英文，数字，英文符号，1到25个字符"))(name => RegexUtil.nameRegex.findAllIn(name).isEmpty && name.length < 100 && name.nonEmpty)) and
    (JsPath \ "type").read[CampaignType] and

}
