package actors.job

import actors.Injector
import akka.actor.Actor
import akka.kafka.scaladsl.Consumer
import akka.kafka.scaladsl.Consumer.DrainingControl
import akka.kafka.{ ConsumerSettings, Subscriptions }
import akka.stream.Materializer
import akka.stream.scaladsl.{ Keep, Sink }
import business.push.PushJobsHandler
import com.typesafe.config.Config
import com.typesafe.scalalogging.LazyLogging
import dtos.Messages.PushTaskRequest
import infrastructure.kafka.deserializer.PushTaskRequestDeserializer
import javax.inject.Inject
import model.Statics
import org.apache.kafka.clients.consumer.{ ConsumerConfig, ConsumerRecord }
import org.apache.kafka.common.serialization.StringDeserializer
import utils.MessageChecker

import scala.collection.JavaConverters._
import scala.concurrent.Future
import scala.concurrent.duration._

/**
 * @author : xhldtc
 * @since : 2019-09-12
 *        用了akka kafka的库启动消费者，文档参考自https://doc.akka.io/docs/alpakka-kafka/current/consumer.html
 */
class PushTaskRequestConsumer @Inject() (implicit
  mat: Materializer,
  pushJobsHandler: PushJobsHandler,
  config: Config,
  injector: Injector,
  messageChecker: MessageChecker) extends Actor with LazyLogging {

  private val consumerSettings = ConsumerSettings(context.system, new StringDeserializer, new PushTaskRequestDeserializer)
    .withBootstrapServers(config.getStringList("channel.kafka.brokers").asScala.mkString(","))
    .withGroupId(injector.configParams.pushRequestConsumerGroupId)
    .withProperty(ConsumerConfig.AUTO_OFFSET_RESET_CONFIG, "earliest")
    .withProperty(ConsumerConfig.ENABLE_AUTO_COMMIT_CONFIG, "true")
    .withProperty(ConsumerConfig.AUTO_COMMIT_INTERVAL_MS_CONFIG, "5000")

  self ! "start"

  override def preStart: Unit = {
    logger.info("starting push task request consumer...")
  }

  override def receive: Receive = {
    case "start" => run()
  }

  private def run() = {
    logger.info("running push task request consumer...")
    val flow = Consumer.plainSource(consumerSettings, Subscriptions.topics(Statics.TOPIC_PUSH_REQUEST))
    val maybeElements = injector.configParams.pushRequestConsumerThrottleElements
    val maybeDuration = injector.configParams.pushRequestConsumerThrottleDurationSeconds
    if (maybeElements.isDefined && maybeDuration.isDefined) {
      flow.throttle(maybeElements.get, maybeDuration.get.seconds)
    }
    flow.mapAsync(1)(consume)
      .toMat(Sink.seq)(Keep.both)
      .mapMaterializedValue(DrainingControl.apply)
      .run()
  }

  private def consume(record: ConsumerRecord[String, PushTaskRequest]): Future[Unit] = {
    try {
      if (!messageChecker.consumed(record.key(), Statics.TOPIC_PUSH_REQUEST)) {
        logger.info(s"consuming key: ${record.key()}, record: ${record.value.toString}")
        pushJobsHandler.publishPushRequests(record.value)
      }
    } catch {
      case e: Throwable => logger.error("consuming exception, ignore and continue", e)
    }
    Future.successful(Unit)
  }
}
