package controllers.api

import model.auth.WrappedRequests.AuthenticationRequest
import play.api.libs.json.{ JsArray, Json }
import play.api.mvc.{ ActionFilter, Results }

import scala.concurrent.ExecutionContext
import scala.concurrent.Future.successful
import scala.concurrent.ExecutionContext.Implicits.global

trait ProjectAuthenticationComponent {
  private lazy val projectUnauthorizedResult = Results.Forbidden(Json.obj("message" -> "Unauthorized", "errors" -> JsArray()))

  def ProjectAuthenticationAction(key: String) = new ActionFilter[AuthenticationRequest] {
    override protected def executionContext: ExecutionContext = global
    def filter[A](request: AuthenticationRequest[A]) = {
      if (request.project.keyid == key) {
        successful(None)
      } else {
        successful(Some(projectUnauthorizedResult))
      }
    }
  }

  def ProjectAuthenticationAction(projectId: Int) = new ActionFilter[AuthenticationRequest] {
    override protected def executionContext: ExecutionContext = global
    def filter[A](request: AuthenticationRequest[A]) = {
      if (request.projectId == projectId) {
        successful(None)
      } else {
        successful(Some(projectUnauthorizedResult))
      }
    }
  }
}
