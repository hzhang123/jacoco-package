package infrastructure.kafka.deserializer

import java.util

import dtos.Messages.PushResult
import org.apache.kafka.common.serialization.Deserializer

/**
 * PushResultDeserializer
 *
 * @author damon lin
 *         2019/5/23
 */
class PushResultDeserializer extends Deserializer[PushResult] {
  override def configure(configs: util.Map[String, _], isKey: Boolean): Unit = ()

  override def close(): Unit = ()

  override def deserialize(topic: String, data: Array[Byte]): PushResult = PushResult.parseFrom(data)

}
