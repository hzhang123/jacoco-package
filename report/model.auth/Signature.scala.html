package model.auth

import java.util.concurrent.TimeUnit

import javax.inject.{ Inject, Singleton }
import com.google.common.cache.{ CacheBuilder, CacheLoader }
import io.growing.micros.business.dao.ProjectDao

import scala.concurrent.Await
import scala.concurrent.duration._
import scala.util.Random

/**
 * Created by yedingding on 15/10/28.
 */

case class AuthenticateError(reason: String) extends Exception

@Singleton
class Signature @Inject() (projectDao: ProjectDao) {

  private lazy val tokenMap = CacheBuilder.newBuilder()
    .maximumSize(1000)
    .expireAfterAccess(1, TimeUnit.HOURS)
    .build(new CacheLoader[String, Token] {
      override def load(service: String): Token = {
        val optProject = Await.result(projectDao.findByKey(service), 1 seconds)
        optProject.map { project =>
          Token(project.clientId.getOrElse(""), project.clientSecret.getOrElse(""))
        } getOrElse (Token.empty)
      }
    })

  def authenticate(key: String, service: String, method: String, path: String, params: Map[String, String]): Either[AuthenticateError, String] = {
    val token = authenticationToken(service)
    if (!token.key.equals(key)) {
      return Left(new AuthenticateError(s"Invalid key: you should have sent your_client_id, but you sent $key"))
    }

    val req = SignedRequest(method, path, params)
    req.sign(token)
    if (req.validate(token)) {
      Right(Random.alphanumeric.take(64).mkString(""))
    } else {
      Left(new AuthenticateError(s"Invalid signature: you should have sent HmacSHA256Hex(${req.stringToSign}, your_client_secret)" +
        s", but you sent ${params.get("auth").getOrElse("")}"))
    }
  }

  def authenticationToken(service: String): Token = {
    tokenMap.get(service)
  }
}
