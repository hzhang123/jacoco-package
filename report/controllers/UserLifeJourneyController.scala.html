package controllers

import form.{ TagForm, UserLifeJourneyForm }
import io.growing.micros.business.auth.{ AuthAction, AuthConstants, AuthRequest, PermissionCheckComponent }
import io.growing.micros.business.controller.{ AbstractController, ProjectActionComponent }
import io.growing.micros.play.mvc.TimeoutAction
import io.growing.micros.utils.conversion.GrowingConversions.UidWrapper
import javax.inject.{ Inject, Singleton }
import play.api.libs.json.{ JsValue, Json }
import play.api.mvc.Results.Ok
import play.api.mvc.{ Action, ActionBuilder, BodyParser, PlayBodyParsers }
import service.UserLifeJourneyService

import scala.concurrent.ExecutionContext.Implicits.global

@Singleton
class UserLifeJourneyController @Inject() (
  ac: TimeoutAction,
  userLifeJourneyService: UserLifeJourneyService,
  parse: PlayBodyParsers)
  extends AbstractController
  with ProjectActionComponent
  with PermissionCheckComponent {

  val readPermission: String = AuthConstants.Marketing.Read
  val managePermission: String = AuthConstants.Marketing.Manage

  def Actions[B](auth: String, projectUid: String, parse: BodyParser[B]): ActionBuilder[AuthRequest, B] =
    WithTimeoutAction(parse) andThen
      new AuthAction(parse) andThen
      ProjectAction(projectUid.toId) andThen
      PermissionCheckAction(auth)

  def create(projectUid: String): Action[JsValue] = Actions(managePermission, projectUid, parse.json).async {
    implicit request =>
      request.body.validate[UserLifeJourneyForm].fold(
        errors => jsonBadRequest(errors),
        form => {
          userLifeJourneyService.create(form, request.project, request.userId).map {
            dto => Ok(Json.toJson(dto))
          }
        })
  }

  def update(projectUid: String, journeyUid: String): Action[JsValue] = Actions(managePermission, projectUid, parse.json).async {
    implicit request =>
      request.body.validate[UserLifeJourneyForm].fold(
        errors => jsonBadRequest(errors),
        form => {
          userLifeJourneyService.update(form, projectUid.toId, journeyUid.toId, request.userId).map {
            dto => Ok(Json.toJson(dto))
          }
        })
  }

  def index(projectUid: String) = Actions(readPermission, projectUid, parse.empty).async {
    implicit request =>
      userLifeJourneyService.findById(request.projectId).map {
        dto => Ok(Json.toJson(dto))
      }
  }

  def queryStages(projectUid: String) = Actions(readPermission, projectUid, parse.empty).async {
    implicit request =>
      userLifeJourneyService.queryStages(request.projectId).map {
        dto => Ok(Json.toJson(dto))
      }
  }

  def tagUserNum(projectUid: String, userType: String) = Actions(readPermission, projectUid, parse.json).async {
    implicit request =>
      request.body.validate[TagForm].fold(
        errors => jsonBadRequest(errors),
        form => {
          userLifeJourneyService.queryTagUserNum(form, userType, projectUid.toId).map {
            userNum => Ok(Json.obj("userNum" -> userNum))
          }
        })
  }
}
