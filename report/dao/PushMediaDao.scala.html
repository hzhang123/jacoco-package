package dao

import dao.component.PushMediaComponent
import io.growing.marketing.common.model.PushMedia
import io.growing.micros.play.slickpg.GrowingPostgresDriver
import io.growing.micros.play.slickpg.GrowingPostgresDriver.api._
import javax.inject.Inject
import play.api.db.slick.{ DatabaseConfigProvider, HasDatabaseConfigProvider }

import scala.concurrent.Future

class PushMediaDao @Inject() (override protected val dbConfigProvider: DatabaseConfigProvider)
  extends PushMediaComponent with HasDatabaseConfigProvider[GrowingPostgresDriver] {

  lazy val insertQuery = PushMedias returning PushMedias.map(_.id) into ((Media, id) => Media.copy(id = Option(id)))

  def insert(media: PushMedia): Future[PushMedia] = {
    db.run(insertQuery += media)
  }

  def findMedia(projectId: Int, fileName: String): Future[Option[PushMedia]] = {
    db.run(PushMedias.filter(m => m.projectId === projectId && m.fileName === fileName).result.headOption)
  }
}
