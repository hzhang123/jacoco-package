package form

import play.api.libs.json.Reads._
import play.api.libs.functional.syntax._
import play.api.libs.json.{ JsPath, Reads, Writes }

case class ProductForm(
  productId: Option[String],
  pages: Option[List[String]] = None)

object ProductForm {
  implicit val reads: Reads[ProductForm] = (
    (JsPath \ "productId").readNullable[String] and
    (JsPath \ "pages").readNullable[List[String]])(ProductForm.apply _)
  implicit val writes: Writes[ProductForm] = (
    (JsPath \ "productId").writeNullable[String] and
    (JsPath \ "pages").writeNullable[List[String]])(unlift(ProductForm.unapply))
}
