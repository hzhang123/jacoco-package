package controllers

import java.util.Locale

import play.api.mvc._
import javax.inject.{ Inject, Singleton }
import io.growing.micros.business.auth._
import play.api.libs.json.{ JsValue, Json }
import com.typesafe.scalalogging.LazyLogging
import form.{ PushDeviceForm, PushMessageForm }
import service.{ PushDeviceService, TokenService }

import scala.concurrent.ExecutionContext.Implicits.global
import io.growing.micros.utils.conversion.GrowingConversions._
import io.growing.micros.play.mvc.{ FutureUtils, TimeoutAction }
import play.api.mvc.Results.{ BadRequest, NoContent, Ok, UnprocessableEntity }
import io.growing.micros.business.controller.{ AbstractController, ProjectActionComponent }

@Singleton
class PushDeviceController @Inject() (
  ac: TimeoutAction,
  errorFormatterAction: ErrorFormatterAction,
  pushDeviceService: PushDeviceService,
  parse: PlayBodyParsers,
  tokenService: TokenService)
  extends AbstractController with ProjectActionComponent with PermissionCheckComponent
  with ACLControlComponent with LazyLogging with FutureUtils {

  val readPermission: String = AuthConstants.Marketing.Read
  val managePermission: String = AuthConstants.Marketing.Manage

  def Actions[B](auth: String, projectUid: String, parse: BodyParser[B]): ActionBuilder[AuthRequest, B] = {
    WithTimeoutAction(parse) andThen errorFormatterAction andThen new AuthAction(parse) andThen
      ProjectAction(projectUid.toId) andThen PermissionCheckAction(auth)
  }

  def index(projectUid: String): Action[Unit] = Actions(readPermission, projectUid, parse.empty).async { implicit request =>
    pushDeviceService.findByProject(request.project.dbid).map(devices => Ok(Json.toJson(devices)))
  }

  def create(projectUid: String): Action[JsValue] = Actions(managePermission, projectUid, parse.json).async { implicit request =>
    request.body.validate[PushDeviceForm].fold(
      errors => jsonBadRequest(errors),
      form =>
        pushDeviceService.create(form, request.project.dbid, request.userId).map {
          case Left(error) => UnprocessableEntity(jsonErrMsgGen(Seq(error)))
          case Right(dto) => Ok(Json.toJson(dto))
        })
  }

  def update(projectUid: String, id: String): Action[JsValue] = Actions(managePermission, projectUid, parse.json).async { implicit request =>
    request.body.validate[PushDeviceForm].fold(
      errors => jsonBadRequest(errors),
      form => {
        implicit val local: Locale = request.locale
        pushDeviceService.update(form, request.project.dbid, id.toId, request.userId).map {
          case Left(error) => UnprocessableEntity(jsonErrMsgGen(Seq(error)))
          case Right(dto) => Ok(Json.toJson(dto))
        }
      })
  }

  def archive(projectUid: String, id: String): Action[Unit] = Actions(managePermission, projectUid, parse.empty).async { implicit request =>
    pushDeviceService.archive(request.project.dbid, id.toId).map(_ => NoContent)
  }

  def preview(projectUid: String, id: String): Action[JsValue] = Actions(managePermission, projectUid, parse.tolerantJson).async(parse.tolerantJson) { implicit request ⇒
    request.body.validate[PushMessageForm].fold(
      errors => jsonBadRequest(errors),
      form ⇒ {
        pushDeviceService.preview(form, id.toId, request.project, request.userId)
          .map(_ ⇒ Ok)
      })
  }

  def applyToken(projectUid: String): Action[AnyContent] = {
    ac.async {
      implicit request ⇒
        tokenService.applyToken(request.headers.get("X-Inner-User-Id"), projectUid).map {
          case Some(token) ⇒ Ok(token)
          case None ⇒ BadRequest("apply token require userId")
        }
    }
  }

  def getDeviceByToken(projectUid: String, token: String): Action[AnyContent] = {
    ac.async {
      pushDeviceService.getPushDeviceByToken(projectUid, token).map {
        case Some(pushDeviceDto) ⇒ Ok(pushDeviceDto)
        case None ⇒ Ok
      }
    }
  }

}
