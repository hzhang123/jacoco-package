package business.push

import business.BaseMessageHandler
import business.inapp.InAppMessageHandler
import com.google.inject.Inject
import dao.PushChannelDao
import dtos.FlushCacheMessage
import handlers.PushMessageHandler
import infrastructure.lock.MessageCacheManager
import infrastructure.measurement.MessageMeasurementHandler
import io.growing.micros.business.model.Status
import javax.inject.Singleton
import model.Statics.Queue.FLUSH_MESSAGE_CACHE
import model.context.{ AppMessageContext, MessageContext }
import play.api.libs.json.Json

import scala.concurrent.ExecutionContext.Implicits.global
import scala.concurrent.Future

/**
 * AppPushMessageHandler
 *
 * @author damon lin
 *         2019/11/7
 */
@Singleton
class AppPushMessageHandler @Inject() (
  inAppMessageHandler: InAppMessageHandler,
  messageCacheManager: MessageCacheManager,
  metricsHandler: MessageMeasurementHandler,
  messageHandler: PushMessageHandler,
  pushJobHandler: PushJobsHandlerImpl,
  pushChannelDao: PushChannelDao) extends BaseMessageHandler[AppMessageContext] {

  override def preCreateHandle(context: MessageContext): Future[MessageContext] = {
    inAppMessageHandler.preCreateHandle(context)
  }

  override def handleMetrics(context: MessageContext): Future[MessageContext] = {
    inAppMessageHandler.handleMetrics(context)
  }

  override def insertMessage(context: MessageContext): Future[MessageContext] = {
    inAppMessageHandler.insertMessage(context)
  }

  override def createRule(c: MessageContext): Future[MessageContext] = {
    val context = shape(c)
    val (form, project, _) = context.parameters
    val message = context.getMessage
    val pushMetrics = context.getMetrics
    for {
      products ← messageHandler.getProductsFromRuleForm(project.dbid, form.rule.get, form.platform)
      rulesToCreate = messageHandler.buildRulesFromForm(form, products, message)
      // 推送不需要上传图片生成网页
      newMessage = message.copy(contentUrl = None, metricsRef = pushMetrics.dbid)
      _ ← inAppMessageHandler.updateMessageAndRules(newMessage, rulesToCreate)
      newContext = context.copy(message = Some(newMessage), rules = Some(rulesToCreate.toList))
      // 推送需要发布推送任务
      _ <- if (newMessage.isActive && newMessage.isPush) pushJobHandler.publishPushJob(newContext) else Future.successful(Unit)
    } yield newContext
  }

  override def afterCreateHandle(c: MessageContext): Future[AppMessageContext] = {
    inAppMessageHandler.afterCreateHandle(c)
  }

  // ================================= update ==========================

  override def preUpdateHandle(context: MessageContext, id: Int): Future[MessageContext] = {
    // 推送的状态如果为 stop ，就返回badRequest
    messageHandler.validatePushState(context, id, Status.stop).flatMap(
      _ => inAppMessageHandler.preUpdateHandle(context, id))
  }

  override def updateMessage(context: MessageContext): Future[MessageContext] = {
    inAppMessageHandler.updateMessage(context)
  }

  override def updateRule(context: MessageContext): Future[MessageContext] = {
    inAppMessageHandler.updateRule(context)
  }

  override def afterUpdateHandle(c: MessageContext): Future[AppMessageContext] = {
    val context = shape(c)
    val form = context.form
    val message = context.getMessage
    val rules = context.getRules
    val firstUpdateToActivated = context.firstUpdateToActivated
    for {
      _ ← messageCacheManager.sendMessage(FLUSH_MESSAGE_CACHE, Json.toJson(FlushCacheMessage(message.ai, rules.map(_.urlScheme))).toString())
      scheduleAt = form.rule.get.scheduleAt
      // 推送需要发布推送任务
      _ ← if (firstUpdateToActivated && message.isPush && scheduleAt.isEmpty) pushJobHandler.publishPushJobBySegment(message) else Future.successful(Unit)
    } yield context.asInstanceOf[AppMessageContext]
  }

  override def shape(context: MessageContext): AppMessageContext = {
    context.asInstanceOf[AppMessageContext]
  }

}
