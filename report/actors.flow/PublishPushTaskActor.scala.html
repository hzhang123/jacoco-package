package actors.flow

import actors.Injector
import actors.protocol.JobProtocols.PublishTaskSuccess
import akka.actor.{ ActorRef, Props }
import dtos.Messages.KafkaPushMessage
import dtos.{ JobContext, PushTaskDto }
import io.growing.marketing.common.model.{ PushChannel, PushMessage, PushRule, PushTask }
import javax.inject.Inject
import model.Statics
import model.Statics.ChannelName
import infrastructure.kafka.KafkaCommonClient

import scala.collection.JavaConverters._
import scala.concurrent.ExecutionContext.Implicits.global
import scala.concurrent.Future
import scala.concurrent.duration._

/**
 * PublishPushTaskActor
 *
 * @author damon lin
 *         2019/5/11
 */
class PublishPushTaskActor @Inject() (injector: Injector)
  extends AbstractActor {

  private val configParams = injector.configParams
  private val kafkaChannelClient = KafkaCommonClient[KafkaPushMessage](configParams.pushChannelTopic, injector.config)
  private val pushTaskDao = injector.pushTaskDao

  override def process(jobContext: JobContext, sender: ActorRef): Future[Any] = {
    publishTasks(jobContext.tasks, jobContext.message, jobContext.rule).map { ack =>
      sender ! PublishTaskSuccess(jobContext)
      PublishTaskSuccess(jobContext)
    }
  }

  def publishTasks(tasks: Seq[PushTaskDto], pushMessage: PushMessage, rule: PushRule): Future[_] = {
    val kafkaPushMessages = tasks.map(t => buildKafkaPushMessage(t.task, pushMessage, rule, t.channel))
    Future.traverse(kafkaPushMessages) { message =>
      logger.info(s" ${this.getClass.getSimpleName} kafka client try to send message: ${message.getId} .")
      kafkaChannelClient.send(message).map {
        case Left(t) =>
          logger.error(s"${this.getClass.getSimpleName} send kafka message failed taskId: ${message.getTaskId}  message: ${message.getId} .")
          pushTaskDao.updateFailedTasks(Seq(message.getTaskId), t.getMessage)
        case Right(recordMetadata) => recordMetadata
      }
    }
  }

  private[this] def buildKafkaPushMessage(task: PushTask, message: PushMessage, rule: PushRule, channel: PushChannel): KafkaPushMessage = {
    val messageExtra = rule.actionParameters ++ Map("gio_internal_actionType" -> rule.realActionType, "gio_internal_actionTarget" -> rule.realActionTarget)
    val base = KafkaPushMessage.newBuilder()
      .setTitle(message.pushTitle)
      .setRestrictedPackageName(channel.packageName)
      .setId(message.uid) // 用于消息去重
      .putAllExtra(messageExtra.asJava)
      .setRetries(configParams.pushRetries)
      .setExpires(configParams.pushChannelExpires.days.toMillis.toInt)
      .setCreated(System.currentTimeMillis())
      .setDescription(message.pushPayload) // 小米推送的正文是在 description 里
      .setPayload(message.pushPayload)
      .setSecrectKey(channel.appSecret)
      .setAppKey(channel.appKey)
      .setAppId(channel.appId.getOrElse(""))
      .setMessageName(message.name)
      .addAllAudience(task.audience.asJava)
      .setTaskId(task.taskId)
      .setChannel(channel.channelName)
    if (channel.isiOS) {
      base.putExtra("server", rule.actionParameters.getOrElse("server", rule.APNsServer))
        .putExtra("port", rule.actionParameters.getOrElse("port", "443"))
        .putExtra("gio_internal_ai", message.ai)
        .build()
    } else {
      base.putAllExtra(appendCallbackParameters(channel.channelName, message.callbackParameters).asJava)
      base.build()
    }
  }

  private def appendCallbackParameters(channel: String, parameters: String): Map[String, String] = {
    channel match {
      case ChannelName.XiaoMi =>
        Map("callback" -> configParams.xiaoMiCallback, "callback.param" -> parameters)
      case ChannelName.Meizu =>
        // meizu 需要在平台上配置后才能发这个信息
        Map("callback" -> configParams.meiZuCallback, "callback.param" -> parameters)
      case ChannelName.Huawei =>
        Map("biTag" -> parameters)
      case ChannelName.Oppo =>
        Map("callback" -> configParams.oppoCallback, "callback.param" -> parameters)
      case ChannelName.Vivo =>
        Map("callback" -> configParams.vivoCallback, "callback.param" -> parameters)
      case _ =>
        Map.empty
    }
  }

}

object PublishPushTaskActor {

  def props(injector: Injector): Props = Props[PublishPushTaskActor](new PublishPushTaskActor(injector))

}
