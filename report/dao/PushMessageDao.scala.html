package dao

import dao.component.{ BaseMessageComponent, PushMessageComponent, PushMetricsComponent, PushRuleComponent }
import io.growing.marketing.common.model.MessageType.MessageType
import io.growing.marketing.common.model._
import io.growing.micros.business.model.Status
import io.growing.micros.play.slickpg.GrowingPostgresDriver
import io.growing.micros.play.slickpg.GrowingPostgresDriver.api._
import io.growing.micros.utils.time.DateTimeUtil.{ DefaultTimeZone, UTCNow }
import javax.inject.{ Inject, Singleton }
import org.joda.time.{ DateTime, DateTimeZone }
import play.api.db.slick.{ DatabaseConfigProvider, HasDatabaseConfigProvider }

import scala.concurrent.ExecutionContext.Implicits.global
import scala.concurrent.Future

@Singleton
class PushMessageDao @Inject() (
  protected val dbConfigProvider: DatabaseConfigProvider,
  pushRuleDao: PushRuleDao) extends PushMessageComponent with BaseMessageComponent with PushRuleComponent with PushMetricsComponent with HasDatabaseConfigProvider[GrowingPostgresDriver] {

  override val messageTableName = "push_messages"
  override val ruleTableName = "push_rules"

  lazy val insertQuery = PushMessages returning PushMessages.map(_.id) into ((Message, id) => Message.copy(id = Option(id)))
  lazy val find = PushMessages.findBy(_.id)

  def insert(m: PushMessage): Future[PushMessage] = {
    db.run(insertQuery += m)
  }

  def batchInsert(ms: Seq[PushMessage]): Future[Seq[PushMessage]] = {
    db.run(insertQuery ++= ms)
  }

  def batchForceInsert(ms: Seq[PushMessage]): Future[Seq[PushMessage]] = {
    db.run(insertQuery.forceInsertAll(ms))
  }

  def archived(id: Int, projectId: Int, userId: Int): Future[Int] = {
    db.run(PushMessages.filter(m => m.id === id && m.projectId === projectId)
      .map(m => (m.state, m.updatedAt, m.updaterId))
      .update((Status.archived, UTCNow, userId)))
  }

  def deleteByCampaign(projectId: Int, campaign: Int): Future[Int] = {
    db.run(PushMessages.filter(m => m.projectId === projectId && m.campaign === campaign).delete)
  }

  def batchArchived(projectId: Int, ids: Seq[Int], userId: Int): Future[Int] = {
    db.run(PushMessages.filter(m => m.id.inSet(ids) && m.projectId === projectId)
      .map(m => (m.state, m.updatedAt, m.updaterId))
      .update((Status.archived, UTCNow, userId)))
  }

  def update(pm: PushMessage): Future[Int] = {
    db.run(PushMessages.filter(m => m.id === pm.dbid && m.projectId === pm.projectId).update(pm))
  }

  def updateActivatedToStop(ids: Seq[Int]): Future[Int] = {
    db.run(PushMessages.filter(m => (m.id inSet ids) && m.state === Status.activated).map(m => (m.updatedAt, m.state)).update((UTCNow, Status.stop)))
  }

  def updatePlanningToActivated(ids: Seq[Int]): Future[Int] = {
    db.run(PushMessages.filter(m => (m.id inSet ids) && m.state === Status.planning).map(m => (m.updatedAt, m.state)).update(UTCNow, Status.activated))
  }

  def findById(id: Int, projectId: Int): Future[Option[PushMessage]] = {
    db.run(PushMessages.filter(m => m.id === id && m.projectId === projectId).result.headOption)
  }

  // 仅内部调用
  override def innerFindByIds(ids: Seq[Int]): Future[Seq[PushMessage]] = {
    db.run(PushMessages.filter(m => m.id.inSet(ids)).result)
  }

  override def findStopMessagesOneWeekAgo(): Future[Seq[BaseMessage]] = {
    val sevenDayBefore = DateTime.now(DateTimeZone.UTC).minusHours(7 * 24) // 7 天前
    val eightDayBefore = DateTime.now(DateTimeZone.UTC).minusHours(7 * 24 + 1) // 再往前推移一个小时
    db.run(PushMessages.filter(m => m.state === Status.stop && m.updatedAt > eightDayBefore && m.updatedAt < sevenDayBefore).result)
  }

  def findByCampaign(projectId: Int, campaign: Int): Future[Seq[PushMessage]] = {
    db.run(PushMessages.filter(m => m.projectId === projectId && m.campaign === campaign).result)
  }

  def findExpiredMessages(): Future[Seq[Int]] = {
    db.run(
      (for {
        (m, _) <- PushMessages.filter(m => m.state === Status.activated) join
          PushRules.filter(r => r.endAt < UTCNow) on (_.id === _.messageId)
      } yield m.id).result)
  }

  def findDailyPushMessage(): Future[Seq[PushMessage]] = {
    val status = Seq(Status.activated, Status.planning)
    pushRuleDao.findDailyRuleMessageIds().flatMap { ids =>
      db.run(PushMessages.filter(m => (m.id inSet ids) && (m.state inSet status)).result)
    }
  }

  def findByProjectId(projectId: Int): Future[List[PushMessage]] = {
    db.run(PushMessages.filter(m => m.projectId === projectId
      && (m.typ inSet List(MessageType.push, MessageType.popupWindow))
      && (m.state inSet List(Status.activated, Status.draft, Status.stop, Status.planning)))
      .sortBy(_.updatedAt.desc).result).map(_.toList)
  }

  def countOnlineByType(projectId: Int, typ: MessageType): Future[Int] = {
    db.run(PushMessages.filter(m => m.projectId === projectId && m.typ === typ && m.state === Status.activated).size.result)
  }

  def findByMsgId(id: Int): Future[PushMessage] = {
    db.run(PushMessages.filter(_.id === id).result.head)
  }

  def findMessagesByMsgIds(projectId: Int, ids: Seq[Int]): Future[Seq[PushMessage]] = {
    db.run(PushMessages.filter(m => m.projectId === projectId && (m.id inSet ids) && (!m.state.inSet(List(Status.archived, Status.deleted)))).result)
  }

  def findPlanningPushMessagesByMsgIds(ids: Seq[Int]): Future[Seq[PushMessage]] = {
    db.run(PushMessages.filter(m => (m.id inSet ids) && (m.state === Status.planning) && (m.typ === MessageType.push)).result)
  }

  def findByName(name: String, projectId: Int): Future[Option[PushMessage]] = {
    db.run(PushMessages.filter(m => m.name === name && m.projectId === projectId && m.state =!= Status.archived).result.headOption)
  }

  def findByNameInOthers(id: Int, name: String, projectId: Int): Future[Option[PushMessage]] = {
    db.run(PushMessages.filter(m => m.id =!= id && m.name === name && m.projectId === projectId && m.state =!= Status.archived).result.headOption)
  }

  def updateMsgMetricsRules(
    pushMessage: PushMessage,
    rulesToCreate: Seq[PushRule],
    rulesToDelete: Seq[PushRule],
    rulesToUpdate: Seq[PushRule],
    oldRules: Option[Seq[PushRule]]): Future[Unit] = {

    val createRules = if (rulesToCreate.nonEmpty) PushRules ++= rulesToCreate else DBIO.successful()
    val deleteRules = if (rulesToDelete.nonEmpty) {
      PushRules.filter(_.id inSet rulesToDelete.map(_.id.get)).delete
    } else {
      DBIO.successful(-1)
    }
    val updateRules = DBIO.sequence {
      if (rulesToUpdate.nonEmpty && oldRules.isDefined) {
        (for {
          ur <- rulesToUpdate
          or <- oldRules.get; if or.productId == ur.productId
        } yield (or, ur)).map(e => pushRuleDao.updateAction(e._1, e._2))
      } else {
        Seq(DBIO.successful(-1))
      }
    }
    val updateMsg = {
      val msg = PushMessages.filter(m => m.id === pushMessage.dbid && m.projectId === pushMessage.projectId)
      msg.update(pushMessage)
    }
    db.run(DBIO.seq(createRules, deleteRules, updateRules, updateMsg).transactionally)
  }
}
