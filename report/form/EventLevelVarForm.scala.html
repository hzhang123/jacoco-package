case class EventLevelVarForm(

  name: String,
  description: Option[String],
  valueType: String,
  isSystem: Boolean)

  implicit val eventLevelVarFormWrites: Writes[EventLevelVarForm] =
    (eventLevelVarForm: EventLevelVarForm) =>
      Json.obj(

        "name" -> eventLevelVarForm.name,
        "description" -> eventLevelVarForm.description,
        "valueType" -> eventLevelVarForm.valueType,
        "isSystem" -> eventLevelVarForm.isSystem)
}
