package actors.flow

import actors.Injector
import actors.protocol.JobProtocols.CheckPointedTasks
import akka.actor.{ ActorRef, Props }
import dtos.{ JobContext, PushTaskDto }
import javax.inject.Inject

import scala.concurrent.ExecutionContext.Implicits.global
import scala.concurrent.Future

/**
 * PushTaskActor
 *
 * @author damon lin
 *         2019/5/11
 */
class CheckpointTaskActor @Inject() (injector: Injector) extends AbstractActor {

  private val pushTaskDao = injector.pushTaskDao

  override def process(jobContext: JobContext, sender: ActorRef): Future[Any] = {
    if (jobContext.tasks.isEmpty) {
      logger.warn(s"${this.getClass.getSimpleName} found empty tasks messages: ${jobContext.id} .")
    }
    logger.info(s"checkpoint tasks of message: ${jobContext.message.uid}")
    checkpointTasks(jobContext.tasks).map { _ =>
      sender ! CheckPointedTasks(jobContext)
    }
  }

  /**
   * 落盘任务
   *
   * @param tasks
   * @return
   */
  def checkpointTasks(tasks: Seq[PushTaskDto]): Future[Any] = {
    pushTaskDao.insert(tasks.map(_.task))
  }

}

object CheckpointTaskActor {

  def props(injector: Injector): Props = Props[CheckpointTaskActor](new CheckpointTaskActor(injector))

}
