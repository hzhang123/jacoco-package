case class UserLifeJourneyStage(
  id: Option[Int] = None,
  projectId: Int,
  journeyId: Int,
  description: String,
  name: String,
  tagId: Option[Int],
  rank: Int,
  creatorId: Int,
  updaterId: Int,
  state: Status = Status.activated,
  createdAt: DateTime = UTCNow,
  updatedAt: DateTime = UTCNow)
  extends BaseModel[UserLifeJourneyStage] {
  self =>

}
