package service

import com.typesafe.scalalogging.LazyLogging
import javax.inject.{ Inject, Singleton }

import scala.concurrent.Future
import dao._
import io.growing.marketing.common.model.{ EditionStatus, EditionType, MarketingEdition, MarketingLead }
import io.growing.micros.play.mvc.FutureUtils
import io.growing.micros.utils.time.DateTimeUtil
import play.api.libs.ws.WSClient
import utils.ConfigParams

import scala.concurrent.ExecutionContext.Implicits.global

@Singleton
class MarketingEditionService @Inject() (
  ws: WSClient,
  configParams: ConfigParams,
  marketingEditionDao: MarketingEditionDao,
  marketingEditionLeadDao: MarketingEditionLeadDao) extends FutureUtils with LazyLogging {

  private val trialDay = 14

  def findByOrganizationId(organizationId: Int): Future[Option[MarketingEdition]] = {
    marketingEditionDao.findByOrganizationId(organizationId)
  }

  def insert(marketingEdition: MarketingEdition): Future[MarketingEdition] = {
    marketingEditionDao.insert(marketingEdition)
  }

  def findOrCreate(organizationId: Int, userId: Int): Future[MarketingEdition] = {
    findByOrganizationId(organizationId).flatMap(me ⇒ {
      if (me.isEmpty) {
        insert(MarketingEdition.default(organizationId, userId))
      } else {
        Future.successful(me.get)
      }
    })
  }

  def applyTrial(marketingEdition: MarketingEdition, userId: Int): Future[MarketingEdition] = {
    val payload = marketingEdition.copy(
      applyAt = Some(DateTimeUtil.UTCNow),
      expiredAt = Some(DateTimeUtil.UTCNow.plusDays(trialDay)),
      status = Some(EditionStatus.Activated),
      `type` = Some(EditionType.Trial),
      updaterId = userId,
      updatedAt = Some(DateTimeUtil.UTCNow))
    marketingEditionDao.update(payload).map(_ ⇒ payload)
  }

  def delayTrail(organizationId: Int): Future[Option[MarketingEdition]] = {
    for {
      maybeEdition <- marketingEditionDao.findByOrganizationId(organizationId)
      newEdition = maybeEdition.map(updateExpired)
      _ <- marketingEditionDao.update(newEdition.toSeq)
    } yield newEdition
  }

  private[this] def updateExpired(edition: MarketingEdition) = {
    val applyAt = edition.applyAt.getOrElse(DateTimeUtil.now)
    val expired = edition.expiredAt.getOrElse(DateTimeUtil.now)
    if (expired.minusDays(trialDay).withTimeAtStartOfDay().getMillis == applyAt.withTimeAtStartOfDay().getMillis) {
      edition.copy(
        expiredAt = Some(DateTimeUtil.UTCNow.plusDays(trialDay)),
        status = Some(EditionStatus.Activated),
        updatedAt = Some(DateTimeUtil.UTCNow))
    } else {
      edition
    }

  }

  def createLead(lead: MarketingLead) = {
    marketingEditionLeadDao.insertLead(lead)
  }

}
