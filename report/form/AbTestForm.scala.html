package form

import io.growing.marketing.common.model.AbTest
import io.growing.micros.business.model.Project
import io.growing.micros.business.model.Status.Status
import play.api.libs.functional.syntax._
import play.api.libs.json.{ JsPath, Reads }

case class AbTestForm(
  messages: List[AbTestMessageForm],
  state: Status,
  name: String,
  compareType: String) {

  def buildAbTest(project: Project, creatorId: Int, updaterId: Int, messages: List[Int], latestMessages: List[Int]): AbTest = {
    AbTest(
      None,
      project.keyid,
      project.dbid,
      creatorId,
      updaterId,
      state,
      name,
      messages,
      latestMessages,
      None,
      compareType)
  }
}

object AbTestForm {
  implicit val reads: Reads[AbTestForm] = (
    (JsPath \ "messages").read[List[AbTestMessageForm]] and
    (JsPath \ "state").read[Status] and
    (JsPath \ "name").read[String] and
    (JsPath \ "compareType").read[String])(AbTestForm.apply _)
}
