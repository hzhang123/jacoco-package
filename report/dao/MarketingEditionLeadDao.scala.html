package dao

import javax.inject.{ Inject, Singleton }
import slick.lifted.{ TableQuery, Tag }
import io.growing.marketing.common.model.{ MarketingLead }
import io.growing.micros.play.slickpg.GrowingPostgresDriver
import io.growing.micros.play.slickpg.GrowingPostgresDriver.api._
import org.joda.time.DateTime
import play.api.db.slick.{ DatabaseConfigProvider, HasDatabaseConfigProvider }

import scala.concurrent.Future

class MarketingLeadTable(tag: Tag) extends Table[MarketingLead](tag, "marketing_leads") {
  def id = column[Int]("id", O.PrimaryKey, O.AutoInc)
  def source = column[String]("source")
  def detail = column[String]("detail")
  def organizationId = column[Int]("organization_id")
  def projectId = column[Int]("project_id")
  def userId = column[Int]("user_id")
  def userPhone = column[Option[String]]("user_phone")
  def userName = column[Option[String]]("user_name")
  def pageUrl = column[String]("page_url")
  def comment = column[Option[String]]("comment")
  def createdAt = column[DateTime]("created_at")
  override def * = (id.?, source, detail, organizationId, projectId, userId, userPhone, userName, pageUrl, comment, createdAt) <> ((MarketingLead.apply _).tupled, MarketingLead.unapply)
}

@Singleton
class MarketingEditionLeadDao @Inject() (protected val dbConfigProvider: DatabaseConfigProvider)
  extends HasDatabaseConfigProvider[GrowingPostgresDriver] { self: HasDatabaseConfigProvider[GrowingPostgresDriver] ⇒
  val marketingLeadTable = TableQuery[MarketingLeadTable]

  def insertLead(marketingLead: MarketingLead): Future[MarketingLead] = {
    db.run(marketingLeadTable returning marketingLeadTable.map(_.id) into ((payload, id) ⇒ payload.withId(id)) += marketingLead)
  }
}
