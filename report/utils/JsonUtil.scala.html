package utils

import play.api.libs.json.{ JsObject, JsValue }

/**
 * JsonUtil
 *
 * @author damon lin
 *         2018/12/12
 */
object JsonUtil {

  def toCss(fields: JsValue) = {
    fields.as[JsObject].fields.foldLeft("")((l, r) =>
      l + s"${r._1}:${r._2};").replaceAll("\"", "")
  }

}
