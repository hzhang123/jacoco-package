package dao

import io.growing.micros.play.slickpg.GrowingPostgresDriver
import io.growing.micros.play.slickpg.GrowingPostgresDriver.api._
import javax.inject.{ Inject, Singleton }
import model.insight
import model.insight.TagType.TagType
import model.insight.{ ComputeDefinition, Tag, TagType }
import play.api.db.slick.{ DatabaseConfigProvider, HasDatabaseConfigProvider }
import play.api.libs.json.{ JsValue, Json }
import slick.lifted.{ Rep, TableQuery, Tag => SlickTag }

import scala.concurrent.ExecutionContext.Implicits.global
import scala.concurrent.Future

/**
 * @author : xhldtc
 * @since : 2019/11/18
 */
class TagsTable(tag: SlickTag) extends BaseTable[insight.Tag](tag, "marketing_tags") {

  implicit val TagTypeE = MappedColumnType.base[TagType, String](
    e => e.toString,
    s => TagType.withName(s))

  implicit val conversionComputeDefinition = MappedColumnType.base[Seq[ComputeDefinition], JsValue](
    e => Json.toJson(e),
    s => s.as[Seq[ComputeDefinition]])

  def name: Rep[String] = column[String]("name")

  def description: Rep[Option[String]] = column[Option[String]]("description")

  def `type`: Rep[TagType] = column[TagType]("type")

  def computes: Rep[Seq[ComputeDefinition]] = column[Seq[ComputeDefinition]]("computes")

  override def * = (id.?, projectId, name, description, `type`, computes, creatorId, updaterId, state, createdAt, updatedAt) <> ((Tag.apply _).tupled, Tag.unapply)
}

@Singleton
class TagDao @Inject() (override protected val dbConfigProvider: DatabaseConfigProvider)
  extends BaseDao[TagsTable, insight.Tag] with HasDatabaseConfigProvider[GrowingPostgresDriver] {

  override val table = TableQuery[TagsTable]

  override def findById(projectId: Int, id: Int): Future[Option[Tag]] = {
    db.run(table.filter(t => t.projectId === projectId && t.id === id).result).map(_.toList.headOption)
  }

  def updateById(projectId: Int, id: Int, tag: Tag): Future[Int] = {
    db.run(table.filter(t => t.projectId === projectId && t.id === id).update(tag))
  }
}
