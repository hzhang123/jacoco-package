package form

import io.growing.marketing.common.model.AudienceType.AudienceType
import io.growing.marketing.common.model.ShortMessage
import io.growing.marketing.common.model.SmsProvider.SmsProvider
import io.growing.micros.business.model.Project
import io.growing.micros.business.model.Status.Status
import io.growing.micros.utils.conversion.GrowingConversions.UidWrapper
import org.joda.time.{ DateTime, DateTimeZone }
import play.api.libs.functional.syntax._
import play.api.libs.json.{ JsPath, Reads }

case class ShortMessageForm(
  name: String,
  configId: String,
  audienceType: Option[AudienceType],
  audienceId: Option[String],
  state: Status,
  signName: Option[String],
  templateCode: Option[String],
  templateContent: Option[String],
  templateParams: List[String],
  segAttrs: List[String],
  testValues: List[String],
  testPhoneNum: Option[String],
  startAt: Long,
  smsProvider: SmsProvider) {

  def buildShortMessage(project: Project, creatorId: Int): ShortMessage = {
    val startTime: DateTime = new DateTime(startAt).withZone(DateTimeZone.UTC)
    ShortMessage(
      None,
      project.dbid,
      project.keyid,
      creatorId,
      creatorId,
      name,
      configId.toId,
      audienceType,
      audienceId,
      state,
      signName,
      templateCode,
      templateContent,
      templateParams,
      segAttrs,
      testValues,
      testPhoneNum,
      startTime,
      None,
      smsProvider)
  }
}

object ShortMessageForm {
  implicit val reads: Reads[ShortMessageForm] = (
    (JsPath \ "name").read[String] and
    (JsPath \ "configId").read[String] and
    (JsPath \ "audienceType").readNullable[AudienceType] and
    (JsPath \ "audienceId").readNullable[String] and
    (JsPath \ "state").read[Status] and
    (JsPath \ "signName").readNullable[String] and
    (JsPath \ "templateCode").readNullable[String] and
    (JsPath \ "templateContent").readNullable[String] and
    (JsPath \ "templateParams").read[List[String]] and
    (JsPath \ "segAttrs").read[List[String]] and
    (JsPath \ "testValues").read[List[String]] and
    (JsPath \ "testPhoneNum").readNullable[String] and
    (JsPath \ "startAt").read[Long] and
    (JsPath \ "smsProvider").read[SmsProvider])(ShortMessageForm.apply _)
}
