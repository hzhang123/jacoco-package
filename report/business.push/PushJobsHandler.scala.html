package business.push

import com.google.inject.ImplementedBy
import dtos.JobContext
import dtos.Messages.PushTaskRequest
import io.growing.marketing.common.model._
import model.context.AppMessageContext

import scala.concurrent.Future

/**
 * PushJobsHandler
 *
 * @author damon lin
 *         2019/5/8
 */
@ImplementedBy(classOf[PushJobsHandlerImpl])
abstract class PushJobsHandler {

  def publishPushJob(context: AppMessageContext): Future[_]

  def publishPushJobBySegment(messages: PushMessage): Future[_]

  /**
   * 根据推送 token 下发消息
   * 没有 u 和 token 的映射
   *
   * @param message
   * @param rule
   * @param channel
   * @param tokens
   * @return
   */
  def publishPushJobByTokens(message: PushMessage, rule: PushRule, channel: PushChannel, tokens: List[String]): Unit

  /**
   * 将消费到的push request发给actor进行后续处理
   *
   * @param pushTaskRequest
   */
  def publishPushRequests(pushTaskRequest: PushTaskRequest): Unit

  /**
   * 以task级别发布推送任务
   *
   * @param pushTask
   */
  def publishPushTask(pushTask: PushTask): Unit

  /**
   * 根据 push job 生成任务上下文
   *
   * @param job
   * @return
   */
  def buildEmptyJobContext(job: PushJob): Future[Seq[JobContext]]

}
