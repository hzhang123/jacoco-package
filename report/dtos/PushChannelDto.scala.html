package dtos

import io.growing.marketing.common.model.PushChannel
import io.growing.micros.utils.conversion.GrowingConversions._
import play.api.libs.json.JodaWrites._
import play.api.libs.json.{ Json, Writes }

case class PushChannelDto(channel: PushChannel)

object PushChannelDto {
  implicit val read: Writes[PushChannelDto] = (o: PushChannelDto) => {
    val channel = o.channel
    val res = Json.obj(
      "id" -> channel.id.map(_.toUid),
      "projectId" -> channel.projectId.toUid,
      "productId" -> channel.productId.toUid,
      "name" -> channel.name,
      "channelName" -> channel.channelName,
      "packageName" -> channel.packageName,
      "appId" -> channel.appId,
      "appKey" -> channel.appKey,
      "appSecret" -> channel.appSecret,
      "creatorId" -> channel.creatorId.toUid,
      "updaterId" -> channel.updaterId.toUid,
      "state" -> channel.state,
      "createdAt" -> channel.createdAt,
      "updatedAt" -> channel.updatedAt)
    res
  }
}
