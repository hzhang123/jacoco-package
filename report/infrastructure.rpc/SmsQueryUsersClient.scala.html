package infrastructure.rpc

import com.typesafe.scalalogging.LazyLogging
import dtos.{ GroupUsersRequest, SmsUserInfo }
import javax.inject.Inject
import play.api.http.Status
import play.api.libs.json.{ JsError, JsSuccess, Json }
import play.api.libs.ws.WSClient
import utils.ConfigParams

import scala.concurrent.ExecutionContext.Implicits.global
import scala.concurrent.Future

/**
 * 根据分群id和分群属性向数据端查询用户信息
 *
 * @param ws
 * @param configParams
 */
class SmsQueryUsersClient @Inject() (ws: WSClient, configParams: ConfigParams) extends LazyLogging {

  def usersInfo(request: GroupUsersRequest): Future[Seq[SmsUserInfo]] = {
    if (request.attrs.slice(0, 3).equals("ppl")) loginUsersInfo(request)
    else visitorsInfo(request)
  }

  private def loginUsersInfo(request: GroupUsersRequest): Future[Seq[SmsUserInfo]] = {
    val pplProps = configParams.qsHost + configParams.qsPort + configParams.loginUserPropsPath
    requestInsight(request, pplProps)
  }

  private def visitorsInfo(request: GroupUsersRequest): Future[Seq[SmsUserInfo]] = {
    val viewRange = configParams.qsHost + configParams.qsPort + configParams.insightPath
    requestInsight(request, viewRange)
  }

  private def requestInsight(request: GroupUsersRequest, path: String): Future[Seq[SmsUserInfo]] = {
    ws.url(path)
      .post(Json.toJson(request))
      .map { response =>
        response.status match {
          case Status.OK =>
            Json.fromJson[Seq[SmsUserInfo]](Json.parse(response.body)) match {
              case JsSuccess(values, _) =>
                values
              case e @ JsError(_) =>
                throw new Exception(JsError.toJson(e).toString())
            }
          case _ =>
            val errorMessage =
              s"""
                 path: ${configParams.qsHost + configParams.qsPort + configParams.loginUserPropsPath}
                 request: ${Json.toJson(request)}
                 response: ${response.toString}
              """
            throw new Exception(errorMessage)
        }
      }
  }
}
