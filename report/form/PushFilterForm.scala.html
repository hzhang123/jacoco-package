package form

import play.api.libs.functional.syntax._
import play.api.libs.json.{ JsPath, Json, Reads, Writes }

final case class PushFilterForm(
  op: String,
  exprsForms: Seq[ExpressionForm])

object PushFilterForm {
  implicit val reads: Reads[PushFilterForm] = (
    (JsPath \ "op").read[String] and
    (JsPath \ "exprs").read[Seq[ExpressionForm]])(PushFilterForm.apply _)

  implicit val writes: Writes[PushFilterForm] = (f: PushFilterForm) =>
    Json.obj(
      "op" -> f.op,
      "exprs" -> f.exprsForms)
}
