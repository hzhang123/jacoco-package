package controllers

import form.ShortMessageForm
import io.growing.micros.business.auth.{ AuthAction, AuthConstants, AuthRequest, PermissionCheckComponent }
import io.growing.micros.business.controller.{ AbstractController, ProjectActionComponent }
import io.growing.micros.play.mvc.TimeoutAction
import io.growing.micros.utils.conversion.GrowingConversions.UidWrapper
import javax.inject.{ Inject, Singleton }
import play.api.libs.json.{ JsValue, Json }
import play.api.mvc.Results.{ BadRequest, Ok }
import play.api.mvc.{ Action, ActionBuilder, BodyParser, PlayBodyParsers }
import service.ShortMessageService

import scala.concurrent.ExecutionContext.Implicits.global

@Singleton
class ShortMessageController @Inject() (
  ac: TimeoutAction,
  shortMessageService: ShortMessageService,
  parse: PlayBodyParsers)
  extends AbstractController
  with ProjectActionComponent
  with PermissionCheckComponent {

  val readPermission: String = AuthConstants.Marketing.Read
  val managePermission: String = AuthConstants.Marketing.Manage

  def Actions[B](auth: String, projectUid: String, parse: BodyParser[B]): ActionBuilder[AuthRequest, B] =
    WithTimeoutAction(parse) andThen
      new AuthAction(parse) andThen
      ProjectAction(projectUid.toId) andThen
      PermissionCheckAction(auth)

  def create(projectUid: String): Action[JsValue] = Actions(managePermission, projectUid, parse.json).async { implicit request =>
    request.body.validate[ShortMessageForm].fold(
      error => jsonBadRequest(error),
      form =>
        shortMessageService.create(form, request.project, request.userId).map {
          case Left(error) => BadRequest(jsonErrMsgGen(Seq(error)))
          case Right(dto) => Ok(Json.toJson(dto))
        })
  }

  def update(projectId: String, shortMsgId: String): Action[JsValue] = Actions(managePermission, projectId, parse.json).async { implicit request =>
    request.body.validate[ShortMessageForm].fold(
      error => jsonBadRequest(error),
      form =>
        shortMessageService.update(form, request.project, request.userId, shortMsgId.toId).map {
          case Left(error) => BadRequest(jsonErrMsgGen(Seq(error)))
          case Right(dto) => Ok(Json.toJson(dto))
        })
  }

  def findAllShortMsg(projectId: String): Action[Unit] = Actions(readPermission, projectId: String, parse.empty).async { implicit request =>
    shortMessageService.findAllShortMessages(request.projectId).map(dto => Ok(Json.toJson(dto)))
  }

  def findSingleShortMsg(projectId: String, shortMsgId: String): Action[Unit] = Actions(readPermission, projectId, parse.empty).async { implicit request =>
    shortMessageService.findSingleShortMessage(shortMsgId.toId).map {
      case Left(error) => BadRequest(jsonErrMsgGen(Seq(error)))
      case Right(dto) => Ok(Json.toJson(dto))
    }
  }

  def archiveShortMsg(projectId: String, shortMsgId: String): Action[Unit] = Actions(managePermission, projectId, parse.empty).async { implicit request =>
    shortMessageService.archiveShortMsg(request.projectId, shortMsgId.toId).map {
      case Left(error) => BadRequest(jsonErrMsgGen(Seq(error)))
      case Right(result) => Ok(Json.obj("message" -> result))
    }
  }

  def templateContent(projectId: String, templateCode: String, configId: String): Action[Unit] = Actions(managePermission, projectId, parse.empty).async { implicit request =>
    shortMessageService.getTemplateContent(templateCode, configId.toId).map {
      case Left(error) => BadRequest(jsonErrMsgGen(Seq(error)))
      case Right(dto) => Ok(Json.toJson(dto))
    }
  }
}
