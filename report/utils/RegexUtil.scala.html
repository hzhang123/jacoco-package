package utils

import io.growing.micros.utils.time.DateTimeUtil

import scala.util.matching.Regex

object RegexUtil {
  val nameRegex: Regex = """[！￥……（）——【】‘；：”“’。，、？]""".r

  val charsRegex = """[\[\]【】《》 。~!@#$%\^\+\*&\\\/\?\|:<>{}()';="]"""

  def genFileName(name: String): String = {
    s"${DateTimeUtil.now.toString("MMdd")}" + s"${name.replaceAll(charsRegex, "")}"
  }

  def genPrefix(id: Int): String = {
    s"${DateTimeUtil.now.toString("yyyyMMdd")}/${id}/${System.currentTimeMillis()}"
  }
}
