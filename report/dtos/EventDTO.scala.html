package dtos

import io.growing.micros.utils.hash.HashIdGenerator
import play.api.libs.json.{ JsPath, Json, OWrites, Reads }

/**
 * 事件rpc传输对象
 *
 * @param id     数据库自增 id
 * @param key    打点英文名称
 * @param name   打点中文名称
 * @param `type` 类型
 */
final case class EventDTO(
  id: String,
  key: String,
  name: String,
  `type`: String,
  eventType: String,
  scope: String,
  projectId: String,
  monitorStatus: String,
  dimensions: Option[Seq[EventLevelVarDTO]]) {
  val dbid: Int = HashIdGenerator.decodeId(id)
}

object EventDTO {

  import play.api.libs.functional.syntax._

  implicit val eventReads: Reads[EventDTO] = (
    (JsPath \ "id").read[String] and
    (JsPath \ "key").read[String] and
    (JsPath \ "name").read[String] and
    (JsPath \ "type").read[String] and
    (JsPath \ "eventType").read[String] and
    (JsPath \ "scope").read[String] and
    (JsPath \ "projectId").read[String] and
    (JsPath \ "monitorStatus").read[String] and
    (JsPath \ "attrs").readNullable[Seq[EventLevelVarDTO]])(EventDTO.apply _)

  implicit val eventWrites: OWrites[EventDTO] = Json.writes[EventDTO]

  def mock: EventDTO = {
    EventDTO(
      "id",
      "key",
      "name",
      "type",
      "eventType",
      "scope",
      "projectId",
      "monitorStatus",
      None)
  }
}
