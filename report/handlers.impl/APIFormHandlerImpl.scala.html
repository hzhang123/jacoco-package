package handlers.impl

import javax.inject.{ Inject, Singleton }
import dao.{ ProductDao, SegmentationDao }
import form.PushRuleTarget

import scala.concurrent.Future
import handlers.APIFormHandler
import io.growing.marketing.common.model.AudienceType
import io.growing.marketing.common.model.AudienceType.AudienceType
import io.growing.micros.utils.conversion.GrowingConversions._

import scala.concurrent.ExecutionContext.Implicits.global
/**
 * APIFormHandlerImpl
 *
 * @author damon lin
 *         2019/8/16
 */
@Singleton
class APIFormHandlerImpl @Inject() (segmentationDao: SegmentationDao, productDao: ProductDao) extends APIFormHandler {

  def extraAudienceType(audienceId: String): Option[AudienceType] = {
    if (AudienceType.StringTypes.contains(audienceId)) {
      Some(AudienceType.withName(audienceId))
    } else {
      val id = audienceId.toId
      if (id == -1) {
        None
      } else {
        Some(AudienceType.Segment)
      }
    }
  }

  def extraAudienceId(projectId: Int, audienceType: AudienceType, audienceId: String): Future[Option[String]] = {
    if (audienceType.equals(AudienceType.Segment)) {
      segmentationDao.findById(projectId, audienceId.toId).map { maybeSegment =>
        maybeSegment.map(s => segmentFieldMapping(s.userScope) + ":" + audienceId.toId)
      }
    } else {
      Future.successful(None)
    }
  }

  val segmentFieldMapping = Map("uv" -> "u", "usv" -> "cs1")

  def extraTargetMap(projectId: Int, targets: List[PushRuleTarget]): Future[Map[String, String]] = {
    val productIds = targets.map(_.productId.toId)
    productDao.findByIds(projectId, productIds).map { products =>
      targets.map { t =>
        val maybeProduct = products.find(_.dbid == t.productId.toId)
        val productKey = maybeProduct.flatMap(_.urlSchema.map(scheme => "growing." + scheme))
        productKey.getOrElse("") -> t.urlString
      }.filterNot(_._1.isEmpty).toMap
    }
  }
}
