package handlers.impl

import cache.RedisCaches
import handlers.VisitorIdDeviceIdMapper
import io.growing.micros.middleware.cache.CacheConfig
import javax.inject.{ Inject, Singleton }
import scala.concurrent.duration._
import scala.concurrent.Future

/**
 * VisitorIdDeviceMapper
 *
 * @author damon lin
 *         2019/6/17
 */
@Singleton
class VisitorIdDeviceIdMapperImpl @Inject() () extends VisitorIdDeviceIdMapper {

  private lazy val redis = RedisCaches.newRedisCache[String](CacheConfig("", Some(48.hours.toSeconds)), "udMapping")

  def insert(entries: Map[String, String]): Future[Boolean] = {
    redis.putAll(entries)
  }
}
