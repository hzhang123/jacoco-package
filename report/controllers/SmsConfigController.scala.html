package controllers

import form.SmsConfigForm
import io.growing.micros.business.auth._
import io.growing.micros.business.controller.{ AbstractController, ProjectActionComponent }
import io.growing.micros.play.mvc.TimeoutAction
import io.growing.micros.utils.conversion.GrowingConversions._
import javax.inject.{ Inject, Singleton }
import play.api.libs.json.{ JsValue, Json }
import play.api.mvc.Results.{ BadRequest, Ok }
import play.api.mvc.{ Action, ActionBuilder, BodyParser, PlayBodyParsers }
import service.SmsConfigService

import scala.concurrent.ExecutionContext.Implicits.global

@Singleton
class SmsConfigController @Inject() (
  ac: TimeoutAction,
  smsConfigService: SmsConfigService,
  parse: PlayBodyParsers) extends AbstractController with ProjectActionComponent with PermissionCheckComponent {

  val readPermission: String = AuthConstants.Marketing.Read
  val managePermission: String = AuthConstants.Marketing.Manage

  def Actions[B](auth: String, projectUid: String, parse: BodyParser[B]): ActionBuilder[AuthRequest, B] = {
    WithTimeoutAction(parse) andThen
      new AuthAction(parse) andThen
      ProjectAction(projectUid.toId) andThen
      PermissionCheckAction(auth)
  }

  def create(projectId: String): Action[JsValue] = Actions(managePermission, projectId, parse.json).async { implicit request =>
    request.body.validate[SmsConfigForm].fold(
      errors => jsonBadRequest(errors),
      form =>
        smsConfigService.create(form, request.projectId, request.userId).map(dto => Ok(Json.toJson(dto))))
  }

  def update(projectId: String, configId: String): Action[JsValue] = Actions(managePermission, projectId, parse.json).async { implicit request =>
    request.body.validate[SmsConfigForm].fold(
      errors => jsonBadRequest(errors),
      form => {
        smsConfigService.update(form, request.projectId, request.userId, configId.toId).map {
          case Left(error) => BadRequest(jsonErrMsgGen(Seq(error)))
          case Right(dto) => Ok(Json.toJson(dto))
        }
      })
  }

  def findAllConfig(projectId: String): Action[Unit] = Actions(readPermission, projectId, parse.empty).async { implicit request =>
    smsConfigService.findAllConfigs(request.projectId).map(dto => Ok(Json.toJson(dto)))
  }

  def findConfig(projectId: String, configId: String): Action[Unit] = Actions(readPermission, projectId, parse.empty).async { implicit request =>
    smsConfigService.findByConfigId(configId.toId).map {
      case Left(error) => BadRequest(jsonErrMsgGen(Seq(error)))
      case Right(dto) => Ok(Json.toJson(dto))
    }
  }

  def archiveConfig(projectId: String, configId: String): Action[Unit] = Actions(managePermission, projectId, parse.empty).async { implicit request =>
    smsConfigService.archiveConfig(request.projectId, configId.toId).map {
      case Left(error) => BadRequest(jsonErrMsgGen(Seq(error)))
      case Right(result) => Ok(Json.obj("message" -> result))
    }
  }
}
