package dtos

import io.growing.marketing.common.model.PushDevice
import io.growing.micros.utils.conversion.GrowingConversions._
import play.api.libs.json.JodaWrites._
import play.api.libs.json.{ Json, Writes }

case class PushDeviceDto(device: PushDevice)

object PushDeviceDto {
  implicit val writes: Writes[PushDeviceDto] = (d: PushDeviceDto) => {
    val device = d.device
    val res = Json.obj(
      "id" -> device.id.map(_.toUid),
      "projectId" -> device.projectId.toUid,
      "name" -> device.name,
      "productId" -> device.productId.toUid,
      "platform" -> device.platform,
      "brand" -> device.brand,
      "channelId" -> device.channelId.map(_.toUid),
      "channelName" -> device.channelName,
      "sdkVersion" -> device.sdkVersion,
      "appVersion" -> device.appVersion,
      "deviceId" -> device.deviceId,
      "regId" -> device.regId,
      "creatorId" -> device.creatorId.toUid,
      "updaterId" -> device.updaterId.toUid,
      "state" -> device.state,
      "createdAt" -> device.createdAt,
      "updatedAt" -> device.updatedAt)
    res
  }
}
