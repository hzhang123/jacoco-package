package controllers

import play.api.mvc._
import java.util.Locale

import play.api.mvc.Results.{ NoContent, Ok }
import form.MessageCampaignForm
import service.MessageCampaignService
import io.growing.micros.business.auth._
import javax.inject.{ Inject, Singleton }
import play.api.libs.json.{ JsValue, Json }
import com.typesafe.scalalogging.LazyLogging
import io.growing.micros.play.mvc.FutureUtils

import scala.concurrent.ExecutionContext.Implicits.global
import io.growing.micros.utils.conversion.GrowingConversions._
import io.growing.micros.business.controller.{ AbstractController, ProjectActionComponent }

@Singleton
class MessageCampaignController @Inject() (
  errorFormatterAction: ErrorFormatterAction,
  messageCampaignService: MessageCampaignService,
  parse: PlayBodyParsers)
  extends AbstractController with ProjectActionComponent with PermissionCheckComponent
  with ACLControlComponent with LazyLogging with FutureUtils {

  val readPermission: String = AuthConstants.Marketing.Read
  val managePermission: String = AuthConstants.Marketing.Manage

  def Actions[B](auth: String, projectUid: String, parse: BodyParser[B]): ActionBuilder[AuthRequest, B] = {
    WithTimeoutAction(parse) andThen errorFormatterAction andThen new AuthAction(parse) andThen
      ProjectAction(projectUid.toId) andThen PermissionCheckAction(auth)
  }

  def index(projectUid: String): Action[Unit] = Actions(readPermission, projectUid, parse.empty).async { implicit request =>
    messageCampaignService.findByProject(request.projectId).map { campaigns =>
      Ok(Json.toJson(campaigns))
    }
  }

  def detail(projectUid: String, id: String) = Actions(readPermission, projectUid, parse.empty).async { implicit request =>
    messageCampaignService.findDetail(request.projectId, id.toId).map { campaigns =>
      Ok(Json.toJson(campaigns))
    }
  }

  def create(projectUid: String): Action[JsValue] = Actions(managePermission, projectUid, parse.json).async { implicit request =>
    request.body.validate[MessageCampaignForm].fold(
      errors => jsonBadRequest(errors),
      form => {
        messageCampaignService.create(form, request.project, request.userId).map { message =>
          Ok(Json.toJson(message))
        }
      })
  }

  def update(projectUid: String, id: String): Action[JsValue] = Actions(managePermission, projectUid, parse.json).async { implicit request =>
    request.body.validate[MessageCampaignForm].fold(
      errors => jsonBadRequest(errors),
      form => {
        implicit val local: Locale = request.locale
        messageCampaignService.update(form, id.toId, request.project, request.userId).map { message =>
          Ok(Json.toJson(message))
        }
      })
  }

  def reset(projectUid: String, id: String) = Actions(managePermission, projectUid, parse.empty).async(parse.empty) { implicit request ⇒
    implicit val locale: Locale = request.locale
    messageCampaignService.reset(projectUid.toId, id.toId, request.userId).map { campaigns =>
      Ok(Json.toJson(campaigns))
    }
  }

  def archive(projectUid: String, id: String) = Actions(managePermission, projectUid, parse.empty).async(parse.empty) { implicit request ⇒
    implicit val locale: Locale = request.locale
    messageCampaignService.archive(projectUid.toId, id.toId, request.userId).map(_ => NoContent)
  }

}
