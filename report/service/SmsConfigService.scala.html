package service

import dao.SmsConfigDao
import dtos.SmsConfigDto
import form.SmsConfigForm
import handlers.ShortMessageHandler
import io.growing.marketing.common.model.SmsConfig
import io.growing.micros.utils.time.DateTimeUtil.UTCNow
import javax.inject.{ Inject, Singleton }

import scala.concurrent.ExecutionContext.Implicits.global
import scala.concurrent.Future

@Singleton
class SmsConfigService @Inject() (
  smsConfigDao: SmsConfigDao,
  shortMessageHandler: ShortMessageHandler) {

  def create(form: SmsConfigForm, projectId: Int, creatorId: Int): Future[SmsConfigDto] = {
    val smsConfig = form.buildSmsConfig(projectId, creatorId, creatorId)
    smsConfigDao.insert(smsConfig).map(config => SmsConfigDto(config))
  }

  def update(form: SmsConfigForm, projectId: Int, updaterId: Int, configId: Int): Future[Either[String, SmsConfigDto]] = {
    val oldConfig = smsConfigDao.findByConfigId(configId)
    oldConfig.map {
      case None => Left("未找到短信配置信息")
      case Some(config) =>
        val newConfig = updateSmsConfig(updaterId, form, config)
        smsConfigDao.updateConfig(configId, newConfig)
        Right(SmsConfigDto(newConfig))
    }
  }

  def findAllConfigs(projectId: Int): Future[List[SmsConfigDto]] = {
    smsConfigDao.findByProject(projectId).flatMap { configs =>
      Future.traverse(configs)(config => Future.successful(SmsConfigDto(config)))
    }
  }

  def findByConfigId(configId: Int): Future[Either[String, SmsConfigDto]] = {
    smsConfigDao.findByConfigId(configId).map {
      case Some(config) => Right(SmsConfigDto(config))
      case _ => Left("未找到短信配置信息")
    }
  }

  def archiveConfig(projectId: Int, configId: Int): Future[Either[String, String]] = {
    smsConfigDao.findByConfigId(configId).map {
      case None => Left("未找到短信配置信息")
      case Some(_) =>
        smsConfigDao.deleteById(projectId, configId)
        Right("删除成功")
    }
  }

  private def updateSmsConfig(updaterId: Int, form: SmsConfigForm, config: SmsConfig): SmsConfig = {
    val state = form.state
    config.copy(
      updaterId = updaterId,
      state = state,
      providerKey = form.providerKey,
      providerValue = form.providerValue,
      signNames = form.signNames,
      numAttribute = form.numAttribute,
      updatedAt = UTCNow)
  }
}
