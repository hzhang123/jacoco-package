package dao

import io.growing.marketing.common.model.PushTask
import io.growing.micros.business.model.Status
import io.growing.micros.play.slickpg.GrowingPostgresDriver
import io.growing.micros.play.slickpg.GrowingPostgresDriver.api._
import javax.inject.Inject
import org.joda.time.DateTime
import play.api.db.slick.{ DatabaseConfigProvider, HasDatabaseConfigProvider }
import slick.lifted.{ ProvenShape, Rep, TableQuery, Tag }

import scala.concurrent.Future

/**
 * PushTaskDao
 *
 * @author damon lin
 *         2019/5/9
 */
class PushTasksTable(tag: Tag) extends BaseTable[PushTask](tag, "push_tasks") {

  def messageId: Rep[Int] = column[Int]("message_id")

  def jobId: Rep[Option[Int]] = column[Option[Int]]("job_id")

  def taskId: Rep[String] = column[String]("task_id")

  def audience: Rep[List[String]] = column[List[String]]("audience")

  def channel: Rep[Int] = column[Int]("channel")

  def reason: Rep[String] = column[String]("reason")

  def retryTimes: Rep[Int] = column[Int]("retry_times")

  override def * : ProvenShape[PushTask] = (id.?, taskId, projectId, messageId, jobId, audience, channel, creatorId, updaterId, state, reason, retryTimes, createdAt, updatedAt) <> ((PushTask.apply _).tupled, PushTask.unapply)
}

class PushTaskDao @Inject() (override protected val dbConfigProvider: DatabaseConfigProvider)
  extends BaseDao[PushTasksTable, PushTask] with HasDatabaseConfigProvider[GrowingPostgresDriver] {

  override val table = TableQuery[PushTasksTable]

  def findFailedTasks(): Future[Seq[PushTask]] = {
    db.run(table.filter(t => t.state === Status.failed).result)
  }

  def findNeedRetryTasks(): Future[Seq[PushTask]] = {
    db.run(table.filter(t => t.state === Status.needRetry).result)
  }

  def findPlanningMessageIds(messageIds: Seq[Int]): Future[Seq[Int]] = {
    db.run(table.filter(t => (t.state === Status.planning || t.state === Status.needRetry)
      && t.messageId.inSet(messageIds)).map(_.messageId).result)
  }

  def findByMessageId(messageId: Int): Future[Seq[PushTask]] = {
    db.run(table.filter(_.messageId === messageId).result)
  }

  def findByTaskId(taskId: String): Future[Option[PushTask]] = {
    db.run(table.filter(_.taskId === taskId).result.headOption)
  }

  def count(messageId: Int): Future[Int] = {
    db.run(table.filter(t => t.messageId === messageId).size.result)
  }

  def updateSuccessfulTasks(ids: Seq[String]): Future[Int] = {
    db.run(table.filter(t => t.taskId inSet ids)
      .map(t => (t.state, t.updatedAt))
      .update(Status.finished, DateTime.now()))
  }

  def updateFailedTasks(ids: Seq[String], reason: String): Future[Int] = {
    db.run(table.filter(t => t.taskId inSet ids)
      .map(t => (t.state, t.reason, t.updatedAt))
      .update((Status.failed, reason, DateTime.now())))
  }

  def updateTaskReason(taskId: String, reason: String): Future[Int] = {
    db.run(table.filter(t => t.taskId === taskId)
      .map(t => (t.reason, t.updatedAt))
      .update((reason, DateTime.now())))
  }

  def updateTimeoutTasks(deadline: DateTime, excludeTasks: Seq[String]): Future[Int] = {
    db.run(table.filter(t => t.state === Status.planning && t.updatedAt < deadline && !(t.taskId inSet excludeTasks))
      .map(t => (t.state, t.reason, t.updatedAt))
      .update(Status.failed, "timeout", DateTime.now()))
  }

  def updateNeedRetryTask(taskId: String): Future[Int] = {
    db.run(table.filter(t => t.taskId === taskId && t.state === Status.needRetry)
      .map(t => (t.state, t.updatedAt))
      .update(Status.planning, DateTime.now()))
  }
}
