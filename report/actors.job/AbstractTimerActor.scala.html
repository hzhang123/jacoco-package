package actors.job

import akka.actor.{ Actor, Timers }

import scala.concurrent.duration._
import com.typesafe.scalalogging.LazyLogging
import infrastructure.lock.LockManager

import scala.concurrent.Future
import scala.concurrent.ExecutionContext.Implicits.global
/**
 * TimerActor
 *
 * @author damon lin
 *         2019/5/24
 */
trait AbstractTimerActor extends Actor with Timers with LazyLogging {

  import AbstractTimerActor._

  val TimerKey: Any
  val delayMillis: Int
  val intervalSeconds: Int
  val lockManager: LockManager

  def initJob(): Unit
  def tickJob(): Future[Any]

  def other(): PartialFunction[Any, Unit] = {
    case _ => logger.warn("not implement !")
  }

  def timerJob: PartialFunction[Any, Unit] = {
    case FirstTick =>
      logger.info(s"Start ${this.getClass.getSimpleName}, delay mills: ${this.delayMillis}.")
      initJob()
      timers.startPeriodicTimer(TimerKey, Tick, intervalSeconds.seconds)
    case Tick =>
      logger.info(s"Tick job of ${this.getClass.getSimpleName}, interval seconds: ${this.intervalSeconds}.")
      lockManager.acquire(s"${this.getClass.getSimpleName}").map {
        case Some(lock) =>
          logger.info(s"${this.getClass.getSimpleName} acquire lock .")
          tickJob().flatMap { _ =>
            lockManager.release(lock).map {
              case true => logger.info(s"${this.getClass.getSimpleName} release lock .")
              case false => logger.warn(s"${this.getClass.getSimpleName} failed release lock !")
            }
          }
        case _ => // do nothing
      }
  }

  def receive: Receive = timerJob orElse other

  timers.startSingleTimer(TimerKey, FirstTick, delayMillis.millis)

}

object AbstractTimerActor {

  case object Tick
  case object FirstTick

}
