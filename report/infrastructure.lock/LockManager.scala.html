package infrastructure.lock

import com.google.inject.ImplementedBy

import scala.concurrent.Future

/**
 * @author wusi
 * @since 2019-05-28 17:33
 */
@ImplementedBy(classOf[RedisCacheManagerImpl])
trait LockManager {

  val DEFAULT_EXPIRY_TIME_MILLIS: Long = 60 * 1000L

  def acquire(key: String, expireTimeMillis: Long = DEFAULT_EXPIRY_TIME_MILLIS): Future[Option[Lock]]

  def release(lock: Lock): Future[Boolean]
}
