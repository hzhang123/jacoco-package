package dao

import scala.concurrent.Future
import javax.inject.{ Inject, Singleton }
import io.growing.micros.play.slickpg.GrowingPostgresDriver
import dao.component.PushMetricsComponent
import io.growing.marketing.common.model.PushMetrics
import io.growing.micros.play.slickpg.GrowingPostgresDriver.api._
import model.Statics
import play.api.db.slick.{ DatabaseConfigProvider, HasDatabaseConfigProvider }

/**
 * PushMetricsDao
 *
 * @author damon lin
 *         2018/12/13
 */
@Singleton
class PushMetricsDao @Inject() (protected val dbConfigProvider: DatabaseConfigProvider)
  extends PushMetricsComponent with HasDatabaseConfigProvider[GrowingPostgresDriver] {

  lazy val insertQuery = PushMetricsTable returning PushMetricsTable.map(_.id) into ((Metric, id) => Metric.copy(id = Option(id)))

  def insert(m: PushMetrics): Future[Int] = {
    db.run(PushMetricsTable returning PushMetricsTable.map(_.id) into ((_, id) => id) += m)
  }

  def findById(id: Int): Future[Option[PushMetrics]] = {
    db.run(PushMetricsTable.filter(_.id === id).result.headOption)
  }

  def findProjectMetrics(projectId: Int): Future[Option[PushMetrics]] = {
    db.run(PushMetricsTable.filter(metrics => metrics.typ === Statics.GLOBAL && metrics.projectId === projectId).result.headOption)
  }

  def findProjectPushMessageMetrics(projectId: Int): Future[Option[PushMetrics]] = {
    db.run(PushMetricsTable.filter(metrics => metrics.typ === Statics.PUSH_GLOBAL && metrics.projectId === projectId).result.headOption)
  }
}
