package utils

import java.io.{ IOException, InputStream, StringWriter }
import java.security._
import java.security.cert.{ Certificate, CertificateEncodingException }

import sun.misc.BASE64Encoder

/**
 * CertUtil
 *
 * @author damon lin
 *         2019/5/14
 */
object CertUtil {

  private lazy val keyStoreType = "JKS"

  @throws[UnrecoverableKeyException]
  @throws[NoSuchAlgorithmException]
  @throws[KeyStoreException]
  def exportPem(stream: InputStream, password: String): String = {
    val keystore = KeyStore.getInstance(keyStoreType)
    keystore.load(stream, password.toCharArray)
    val alias = keystore.aliases().nextElement()
    val keyPair = getPrivateKey(keystore, alias, password.toCharArray)
    val privateKey = keyPair.getPrivate
    val cert = keystore.getCertificate(alias)
    convertCertToPem(cert, privateKey)
  }

  @throws[IllegalArgumentException]
  private[this] def getPrivateKey(keystore: KeyStore, alias: String, password: Array[Char]): KeyPair = {
    val key = keystore.getKey(alias, password)
    key match {
      case k: PrivateKey =>
        val cert = keystore.getCertificate(alias)
        val publicKey = cert.getPublicKey
        new KeyPair(publicKey, k)
      case _ => throw new IllegalArgumentException("Key store load private key failed!")
    }
  }

  @throws[IOException]
  @throws[CertificateEncodingException]
  private[this] def convertCertToPem(cert: Certificate, privateKey: PrivateKey) = {
    val encoder = new BASE64Encoder
    val encoded = encoder.encode(privateKey.getEncoded)
    val fw = new StringWriter()
    fw.write("-----BEGIN PRIVATE KEY-----\n")
    fw.write(encoded)
    fw.write("\n")
    fw.write("-----END PRIVATE KEY-----\n")
    val encoded2 = encoder.encode(cert.getEncoded)
    fw.write("-----BEGIN CERTIFICATE-----\n")
    fw.write(encoded2)
    fw.write("\n")
    fw.write("-----END CERTIFICATE-----\n")
    fw.toString
  }

}
