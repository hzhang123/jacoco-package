package dao.component

import io.growing.marketing.common.model.PushApiRequest
import io.growing.micros.play.slickpg.GrowingPostgresDriver
import io.growing.micros.play.slickpg.GrowingPostgresDriver.api._
import org.joda.time.DateTime
import play.api.db.slick.HasDatabaseConfigProvider
import slick.lifted.{ Rep, Tag }

/**
 * @author : xhldtc
 * @since : 2019-09-23
 */
trait PushApiRequestComponent {

  self: HasDatabaseConfigProvider[GrowingPostgresDriver] =>

  class PushApiRequestTable(tag: Tag) extends Table[PushApiRequest](tag, "push_api_requests") {

    def id: Rep[Int] = column[Int]("id", O.PrimaryKey, O.AutoInc)

    def cid: Rep[String] = column[String]("cid")

    def md5: Rep[String] = column[String]("md5")

    def messageId: Rep[Option[Int]] = column[Int]("message_id")

    def audience: Rep[List[String]] = column[List[String]]("audience")

    def createdAt: Rep[DateTime] = column[DateTime]("created_at")

    def updateAt: Rep[DateTime] = column[DateTime]("updated_at")

    override def * =
      (id.?, cid, md5, messageId, audience, createdAt, updateAt) <> ((PushApiRequest.apply _).tupled, PushApiRequest.unapply)

  }

  val PushApiRequestTable = TableQuery[PushApiRequestTable]
}
