package dtos

import model.Statics
import play.api.libs.json.Json

/**
 * VisitorPropsRequest
 *
 * @author damon lin
 *         2019/7/27
 */
final case class VisitorPropsRequest(
  ai: String,
  userScope: String,
  attrs: String)

object VisitorPropsRequest {

  def fromTokens(ai: String, field: String, csList: List[String]): VisitorPropsRequest = {
    VisitorPropsRequest(
      ai,
      csList.map(cs => field + ":" + cs).mkString(","),
      Statics.PushVariables.mkString(","))
  }

  implicit val fmt = Json.format[VisitorPropsRequest]

}
