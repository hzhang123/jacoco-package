package actors.job

import actors.Injector
import akka.actor.Props
import business.push.PushJobsHandler
import javax.inject.Inject
import dao.{ AbTestDao, AbTestMessageDao, AbTestRuleDao, OnlineMessageDao, OnlineRuleDao, PushRuleDao }
import com.typesafe.scalalogging.LazyLogging
import handlers.AbTestHandler

import scala.concurrent.ExecutionContext.Implicits.global
import scala.concurrent.Future

/**
 * 检查定时发送
 * 标准：有 startAt - endAt 没有 scheduleAt 状态是 planning 是一次性计划发送
 * @param pushJobsHandler
 * @param onlineRuleDao
 * @param onlineMessageDao
 * @param pushRuleDao
 */
class PushMessageScheduleOnceActor @Inject() (
  injector: Injector,
  onlineRuleDao: OnlineRuleDao,
  onlineMessageDao: OnlineMessageDao,
  pushRuleDao: PushRuleDao,
  pushJobsHandler: PushJobsHandler,
  abTestMessageDao: AbTestMessageDao,
  abTestDao: AbTestDao,
  abTestRuleDao: AbTestRuleDao,
  abTestHandler: AbTestHandler) extends AbstractTimerActor with LazyLogging {

  import PushMessageScheduleOnceActor._

  private val jobKey = "PushMessageScheduleOnce"

  val lockManager = injector.lockManager
  val pushMetricsHandler = injector.pushMetricsHandler
  val pushMessageDao = injector.pushMessageDao
  val TimerKey: Any = ExpiredTickKey
  val delayMillis: Int = 500
  val intervalSeconds: Int = 30

  def initJob(): Unit = {
  }

  // fixme: 会重复查找
  def tickJob(): Future[Any] = {
    for {
      distinctMessageIds <- pushRuleDao.findPlanningMessageIds().map(_.distinct)
      abTestMessageIds <- abTestRuleDao.findPlanningMessageIds().map(_.distinct)
      onlineMessageIds <- onlineRuleDao.findPlanningMessageIds().map(_.distinct)
      pushMessages <- pushMessageDao.findPlanningPushMessagesByMsgIds(distinctMessageIds)
      _ <- pushMessageDao.updatePlanningToActivated(distinctMessageIds)
      _ <- onlineMessageDao.updatePlanningToActivated(onlineMessageIds)
      _ <- abTestMessageDao.updatePlanningToActivated(abTestMessageIds)
      _ <- abTestDao.updatePlanningToActivated(abTestMessageIds)
      _ <- abTestRuleDao.updatePlanningToActivated(abTestMessageIds)
      _ <- pushRuleDao.updateActivatedAt(distinctMessageIds)
    } yield {
      if (pushMessages.nonEmpty) {
        // 定时发送推送消息
        logger.warn(s"${this.getClass.getSimpleName} find timing message: ${pushMessages.map(_.uid).mkString(";")} !")
        pushMessages.foreach(m => pushJobsHandler.publishPushJobBySegment(m))
      }
    }
    for {
      distinctMessageIds <- pushRuleDao.findExpiredMessageIds().map(_.distinct)
      onlineMessageIds <- onlineRuleDao.findExpiredMessageIds().map(_.distinct)
      expiredAbTestMessageIds <- abTestRuleDao.findExpiredMessageIds().map(_.distinct)
      _ <- pushMessageDao.updateActivatedToStop(distinctMessageIds)
      _ <- onlineMessageDao.updateActivatedToStop(onlineMessageIds)
      _ <- abTestHandler.updateExpireAbTestMessages(expiredAbTestMessageIds)
    } yield {
      distinctMessageIds
    }

  }

}

object PushMessageScheduleOnceActor {

  private case object ExpiredTickKey

  def props(injector: Injector, onlineRuleDao: OnlineRuleDao,
    onlineMessageDao: OnlineMessageDao, pushRuleDao: PushRuleDao, pushJobsHandler: PushJobsHandler,
    abTestMessageDao: AbTestMessageDao, abTestDao: AbTestDao, abTestRuleDao: AbTestRuleDao, abTestHandler: AbTestHandler) =
    Props[PushMessageScheduleOnceActor](new PushMessageScheduleOnceActor(
      injector, onlineRuleDao,
      onlineMessageDao, pushRuleDao, pushJobsHandler,
      abTestMessageDao, abTestDao, abTestRuleDao, abTestHandler))
}
