package form

import play.api.libs.functional.syntax._
import play.api.libs.json.Reads.filter
import play.api.libs.json.{ JsPath, Json, JsonValidationError, Reads, Writes }

final case class ExpressionForm(
  symbol: String,
  `type`: String,
  key: Option[String],
  op: String,
  valueType: Option[String],
  value: Option[String],
  values: Seq[String])

object ExpressionForm {
  val typeSeq: Seq[String] = Seq("visitorFilter", "loginUserFilter", "eventFilter")
  val opSeq: Seq[String] = Seq("<", ">", "<=", ">=", "=", "==", "!=", "in", "not in", "between")

  implicit val read: Reads[ExpressionForm] = (
    (JsPath \ "symbol").read[String] and
    (JsPath \ "type").read[String](filter[String](JsonValidationError("表达式类型错误"))(typeSeq.contains(_))) and
    (JsPath \ "key").readNullable[String] and
    (JsPath \ "op").read[String](filter[String](JsonValidationError("表达式操作符类型错误"))(opSeq.contains(_))) and
    (JsPath \ "valueType").readNullable[String] and
    (JsPath \ "value").readNullable[String] and
    (JsPath \ "values").read[Seq[String]])(ExpressionForm.apply _)

  implicit val writes: Writes[ExpressionForm] = (e: ExpressionForm) => {
    val value = if (e.value.isDefined) e.value.get else e.values.headOption.getOrElse("")
    var base = Json.obj(
      "symbol" -> e.symbol,
      "type" -> e.`type`,
      "op" -> e.op,
      "valueType" -> e.valueType,
      "values" -> e.values,
      "value" -> value)

    // 前端要求没有定义key或value, 不传回去
    if (e.key.isDefined)
      base ++= Json.obj(
        "key" -> e.key)

    base
  }
}
