package dao

import dao.component.PushApiRequestComponent
import io.growing.marketing.common.model.PushApiRequest
import io.growing.micros.play.slickpg.GrowingPostgresDriver
import io.growing.micros.play.slickpg.GrowingPostgresDriver.api._
import javax.inject.{ Inject, Singleton }
import play.api.db.slick.{ DatabaseConfigProvider, HasDatabaseConfigProvider }

import scala.concurrent.Future

/**
 * @author : xhldtc
 * @since : 2019-09-23
 */

@Singleton
class PushApiRequestDao @Inject() (protected val dbConfigProvider: DatabaseConfigProvider)
  extends PushApiRequestComponent with HasDatabaseConfigProvider[GrowingPostgresDriver] {

  lazy val insertQuery = PushApiRequestTable returning PushApiRequestTable.map(_.id) into ((ApiRequest, id) => ApiRequest.copy(id = Option(id)))

  def insert(pushApiRequest: PushApiRequest): Future[PushApiRequest] = {
    db.run(insertQuery += pushApiRequest)
  }

  //查看库里是否已经有除了当前插入的request之外已存在的cid request,如果有的话就不用创建message了，直接用已存在cid request的messageId查message表
  def findExistCidRequest(cid: String): Future[Option[PushApiRequest]] = {
    db.run(PushApiRequestTable.filter(r => r.cid === cid).result.headOption)
  }

  def findByMessageIds(messageIds: Seq[Int]): Future[Seq[PushApiRequest]] = {
    db.run(PushApiRequestTable.filter(r => r.messageId inSet messageIds).result)
  }

  def deleteById(id: Int): Future[Int] = {
    db.run(PushApiRequestTable.filter(r => r.id === id).delete)
  }
}
