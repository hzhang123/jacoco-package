package infrastructure.kafka

import java.util.{ Properties, UUID }

import com.google.protobuf.Message
import com.typesafe.config.Config
import com.typesafe.scalalogging.LazyLogging
import org.apache.kafka.clients.producer.{ KafkaProducer, Producer, ProducerRecord, RecordMetadata }
import org.apache.kafka.common.serialization.StringSerializer
import utils.FutureUtil

import scala.collection.JavaConverters._
import scala.concurrent.ExecutionContext.Implicits.global
import scala.concurrent.Future
import scala.util.Try

/**
 * @author : xhldtc
 * @since : 2019-09-11
 */
class KafkaCommonClient[T <: Message](topic: String, properties: Properties) extends LazyLogging {

  private[this] lazy val producer: Producer[String, Message] = {
    val keySerializer = new StringSerializer
    val valueSerializer = new MessageSerializer
    new KafkaProducer[String, Message](properties, keySerializer, valueSerializer)
  }

  def send(message: T): Future[Either[Throwable, RecordMetadata]] = {
    val key = UUID.randomUUID().toString
    val record = new ProducerRecord[String, Message](topic, key, message)
    FutureUtil.toScalaFuture(producer.send(record)).map(Right(_)).recover {
      case e: Throwable =>
        logger.error("send to kafka failed!", e)
        Left(e)
    }
  }

  def close(): Unit = Try(producer.close())
}

object KafkaCommonClient {

  def apply[T <: Message](topic: String, properties: Properties) = new KafkaCommonClient[T](topic, properties)

  def apply[T <: Message](topic: String, config: Config): KafkaCommonClient[T] = {
    val properties = new Properties()
    properties.put("bootstrap.servers", config.getStringList("channel.kafka.brokers").asScala.mkString(","))
    properties.put("acks", "1")
    new KafkaCommonClient[T](topic, properties)
  }
}
