package dao

import io.growing.marketing.common.model.PushMessage
import javax.inject.Inject
import play.api.db.slick.DatabaseConfigProvider
import scala.concurrent.Future
import io.growing.micros.play.slickpg.GrowingPostgresDriver.api._

/**
 * OnlineMessageDao
 *
 * @author damon lin
 *         2019/8/26
 */
class OnlineMessageDao @Inject() (
  override protected val dbConfigProvider: DatabaseConfigProvider,
  pushRuleDao: OnlineRuleDao) extends PushMessageDao(dbConfigProvider, pushRuleDao) {

  override val messageTableName = "online_messages"
  override val ruleTableName = "online_rules"

  def batchDelete(projectId: Int, ids: Seq[Int], userId: Int): Future[Int] = {
    db.run(PushMessages.filter(m => m.id.inSet(ids) && m.projectId === projectId).delete)
  }

}
