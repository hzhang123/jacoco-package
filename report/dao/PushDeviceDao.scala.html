package dao

import io.growing.marketing.common.model.PushDevice
import io.growing.micros.business.model.Status
import io.growing.micros.play.slickpg.GrowingPostgresDriver
import io.growing.micros.play.slickpg.GrowingPostgresDriver.api._
import javax.inject.Inject
import play.api.db.slick.{ DatabaseConfigProvider, HasDatabaseConfigProvider }
import slick.lifted.{ Rep, Tag }

import scala.concurrent.Future

class PushDevicesTable(tag: Tag) extends BaseTable[PushDevice](tag, "push_devices") {
  def name: Rep[String] = column[String]("name")

  def productId: Rep[Int] = column[Int]("product_id")

  def platform: Rep[String] = column[String]("platform")

  def brand: Rep[String] = column[String]("brand")

  def channelId: Rep[Option[Int]] = column[Option[Int]]("channel_id")

  def channelName: Rep[String] = column[String]("channel_name")

  def sdkVersion: Rep[String] = column[String]("sdk_version")

  def appVersion: Rep[String] = column[String]("app_version")

  def deviceId: Rep[String] = column[String]("device_id")

  def regId: Rep[String] = column[String]("reg_id")

  override def * = (id.?, projectId, name, productId, platform, brand, channelId, channelName, sdkVersion, appVersion,
    deviceId, regId, creatorId, updaterId, state, createdAt, updatedAt) <> ((PushDevice.apply _).tupled, PushDevice.unapply)
}

class PushDeviceDao @Inject() (override protected val dbConfigProvider: DatabaseConfigProvider)
  extends BaseDao[PushDevicesTable, PushDevice] with HasDatabaseConfigProvider[GrowingPostgresDriver] {

  override val table = TableQuery[PushDevicesTable]

  override lazy val insertQuery = table returning table.map(_.id) into ((Device, id) => Device.copy(id = Option(id)))

  override def insert(models: Seq[PushDevice]): Future[Seq[PushDevice]] = super.insert(models)

  override def update(projectId: Int, id: Int, model: PushDevice): Future[Int] = super.update(projectId, id, model)

  override def deleteById(projectId: Int, id: Int): Future[Int] = super.deleteById(projectId, id)

  def findByName(projectId: Int, name: String): Future[Option[PushDevice]] = {
    db.run(table.filter(d => d.projectId === projectId && d.name === name && d.state =!= Status.archived).result.headOption)
  }

  def findByNameInOthers(projectId: Int, deviceId: Int, name: String): Future[Option[PushDevice]] = {
    db.run(table.filter(d => d.projectId === projectId && d.id =!= deviceId && d.name === name && d.state =!= Status.archived).result.headOption)
  }

  def findDeviceWithoutChannelId(projectId: Int, productId: Int, channelName: String): Future[Seq[PushDevice]] = {
    db.run(table.filter(d => d.projectId === projectId && d.productId === productId && d.channelName === channelName).result)
  }

  def updateChannelId(deviceId: Int, channelId: Option[Int]): Future[Int] = {
    db.run(table.filter(d => d.id === deviceId).map(d => d.channelId).update(channelId))
  }
}
