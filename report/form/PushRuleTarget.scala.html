package form

import io.growing.marketing.common.model.PushRule
import play.api.libs.json.Json
import io.growing.micros.utils.conversion.GrowingConversions._
import model.Statics.ActionTargetProto._
/**
 * PushRuleTarget
 *
 * @author damon lin
 *         2019/6/3
 */
case class PushRuleTarget(
  productId: String,
  `type`: String,
  url: String,
  parameters: Map[String, String]) {

  def urlString: String = {
    `type` match {
      case OpenApp =>
        "GInAPP://" + url + queryString
      case _ =>
        url + queryString
    }
  }

  def queryString: String = "?" + parameters.map(e => s"${e._1}=${e._2}").mkString("&")

}

object PushRuleTarget {

  implicit val fmt = Json.format[PushRuleTarget]

  def fromRule(rule: PushRule): Option[PushRuleTarget] = {
    rule.actionType.map { typ =>
      PushRuleTarget(
        //web弹窗的rule没有product,但是自己构造了pushRuleTarget，type不为空会进来这个逻辑，因此设置了默认的productId,
        //否则序列化会空指针异常
        rule.productId.map(_.toUid).getOrElse(""),
        typ,
        rule.actionTarget.get,
        rule.actionParameters)
    }
  }
}
