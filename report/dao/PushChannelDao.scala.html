package dao

import io.growing.marketing.common.model.PushChannel

import io.growing.micros.business.model.Status.Status

import io.growing.micros.play.slickpg.GrowingPostgresDriver.api._
import javax.inject.Inject
import model.Statics
import org.joda.time.DateTime
import play.api.db.slick.{ DatabaseConfigProvider, HasDatabaseConfigProvider }
import slick.lifted.{ Rep, TableQuery, Tag }
import utils.AESUtils

import scala.concurrent.ExecutionContext.Implicits.global
import scala.concurrent.Future

/**
 * PushChannelDao
 *
 * @author damon lin
 *         2019/5/9
 */

class PushChannelsTable(tag: Tag) extends BaseTable[PushChannel](tag, "push_channels") {

  def productId: Rep[Int] = column[Int]("product_id")

  def name: Rep[String] = column[String]("name")

  def channelName: Rep[String] = column[String]("channel_name")

  def packageName: Rep[String] = column[String]("package_name")

  def appId: Rep[Option[String]] = column[Option[String]]("app_id")

  def appKey: Rep[String] = column[String]("app_key")

  def appSecret: Rep[String] = column[String]("app_secret")

  def decrypt(
    id: Option[Int],
    projectId: Int,
    productId: Int,
    name: String,
    channelName: String,
    packageName: String,
    appId: Option[String],
    appKey: String,
    appSecret: String,
    creatorId: Int,
    updaterId: Int,
    state: Status,
    createdAt: DateTime,
    updatedAt: DateTime): PushChannel = {
    val decryptedAppSecret = AESUtils.decrypt(appSecret)
    PushChannel(id, projectId, productId, name, channelName, packageName, appId, appKey, decryptedAppSecret, creatorId, updaterId, state, createdAt, updatedAt)
  }

  def encrypt(pc: PushChannel) = {
    val encryptedAppSecret = AESUtils.encrypt(pc.appSecret)
    PushChannel.unapply(pc.copy(appSecret = encryptedAppSecret))
  }

  override def * = (id.?, projectId, productId, name, channelName, packageName,
    appId, appKey, appSecret, creatorId, updaterId, state, createdAt, updatedAt) <> ((decrypt _).tupled, encrypt)

}

class PushChannelDao @Inject() (override protected val dbConfigProvider: DatabaseConfigProvider)
  extends BaseDao[PushChannelsTable, PushChannel] with HasDatabaseConfigProvider[GrowingPostgresDriver] {

  override val table = TableQuery[PushChannelsTable]

  override lazy val insertQuery = table returning table.map(_.id) into ((Channel, id) => Channel.copy(id = Option(id)))

  override def findByProject(projectId: Int): Future[List[PushChannel]] = {
    db.run(table.filter(c => c.projectId === projectId && c.state === Status.activated).result).map(_.toList)
  }

  def findByPackageName(projectId: Int, packageName: String): Future[List[PushChannel]] = {
    db.run(table.filter(c => c.projectId === projectId && c.packageName === packageName && c.state === Status.activated).result).map(_.toList)
  }

  def findByName(name: String, projectId: Int): Future[Option[PushChannel]] = {
    db.run(table.filter(c => c.name === name && c.projectId === projectId && c.state =!= Status.archived).result.headOption)
  }

  def findByNameInOthers(id: Int, name: String, projectId: Int): Future[Option[PushChannel]] = {
    db.run(table.filter(c => c.projectId === projectId && c.name === name && c.id =!= id && c.state =!= Status.archived).result.headOption)
  }

  def findByProductAndName(projectId: Int, productId: Int, name: String): Future[Option[PushChannel]] = {
    db.run(table.filter(c => c.channelName === name && c.projectId === projectId && c.state =!= Status.archived && c.productId === productId).result.headOption)
  }
}
