package dtos

import form.PushFilterForm
import io.growing.marketing.common.model._
import presenters.PushMessagePresenter

final case class PushMessageDto(
  message: PushMessage,
  rules: Seq[PushRule],
  metrics: Option[PushMetrics],
  audienceFilterForm: Option[PushFilterForm],
  from: String = "other") {

  def toPresenter: PushMessagePresenter =
    PushMessagePresenter(message, rules, metrics, audienceFilterForm)

  def joinSeq(seqs: List[Sequencing]): PushMessageDto = {
    val realPriority = seqs.find(_.messageId == message.dbid).map(_.priority).getOrElse(1)
    PushMessageDto(
      message.copy(priority = realPriority),
      rules,
      metrics,
      audienceFilterForm,
      from)
  }

}
