package actors.flow

import actors.protocol.JobProtocols.FailureTask
import akka.actor.{ Actor, ActorRef }
import com.typesafe.scalalogging.LazyLogging
import dtos.JobContext
import akka.pattern.pipe
import scala.concurrent.Future
import scala.concurrent.ExecutionContext.Implicits.global
/**
 * AbstractActor
 *
 * @author damon lin
 *         2019/7/2
 */
trait AbstractActor extends Actor with LazyLogging {

  /**
   * actor 业务主逻辑
   * @param jobContext
   * @param sender
   * @return
   */
  def process(jobContext: JobContext, sender: ActorRef): Future[Any]

  def receive = {
    case jobContext: JobContext =>
      logger.info(s"${this.getClass.getSimpleName} receive job message: ${jobContext.id} .")
      process(jobContext, sender()).recover(wrapThrowable(jobContext)) pipeTo self
    case failureTask: FailureTask =>
      val errorInfo = s"${failureTask.throwable.getMessage} + ${failureTask.throwable.getStackTrace}"
      logger.error(s"${this.getClass.getSimpleName} message: ${failureTask.context.id} catch exception: $errorInfo .")
    case otherMessage =>
      logger.warn(s"${this.getClass.getSimpleName} received other message: " + otherMessage.toString)
  }

  def wrapThrowable(jobContext: JobContext): PartialFunction[Throwable, FailureTask] = {
    case t: Throwable =>
      FailureTask(jobContext, t)
  }

}
