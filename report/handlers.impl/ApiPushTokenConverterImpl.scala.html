package handlers.impl

import dtos.UserInfo.reads
import dtos.{ UserInfo, VisitorPropsRequest }
import handlers.ApiPushTokenConverter
import javax.inject.{ Inject, Singleton }
import play.api.http.Status
import play.api.libs.json.{ JsError, JsSuccess, Json }
import play.api.libs.ws.WSClient
import utils.ConfigParams

import scala.concurrent.ExecutionContext.Implicits.global
import scala.concurrent.Future

/**
 * ApiPushTokenConverterImpl
 *
 * @author damon lin
 *         2019/7/27
 */
@Singleton
class ApiPushTokenConverterImpl @Inject() (ws: WSClient, configParams: ConfigParams) extends ApiPushTokenConverter {

  override def convertCs1ToPushTokens(ai: String, field: String, csList: List[String]): Future[List[UserInfo]] = {
    val request = VisitorPropsRequest.fromTokens(ai, field, csList)
    val url = configParams.qsHost + configParams.qsPort + configParams.visitorPropsPath
    ws.url(url)
      .post(Json.toJson(request))
      .map { response =>
        response.status match {
          case Status.OK =>
            Json.fromJson[Seq[UserInfo]](Json.parse(response.body)) match {
              case JsSuccess(values, _) => values.toList
              case e @ JsError(_) =>
                throw new Exception(JsError.toJson(e).toString())
            }
          case _ =>
            val errorMessage =
              s"""
                 path: $url
                 request: ${Json.toJson(request)}
                 response: ${response.toString}
              """
            throw new Exception(errorMessage)
        }
      }
  }
}
