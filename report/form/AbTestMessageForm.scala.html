package form

import io.growing.marketing.common.model.AudienceType.AudienceType
import io.growing.marketing.common.model.{ AbTestMessage, MessagePlatform, MessageType, PushContent, PushFilter }
import io.growing.micros.business.model.Status
import io.growing.micros.business.model.Status.Status
import play.api.libs.functional.syntax._
import play.api.libs.json.{ JsPath, Reads }
import utils.MappingUtil
import io.growing.micros.utils.conversion.GrowingConversions._

final case class AbTestMessageForm(
  id: Option[String],
  symbol: String,
  rate: Option[Int],
  state: Option[Status],
  audienceType: Option[AudienceType],
  audienceId: Option[String],
  audienceFilter: Option[PushFilterForm],
  rule: Option[PushRuleForm],
  content: Option[PushContent],
  ctrlGroup: Boolean,
  metric: Option[ConversionTargetForm],
  totalActivatedTimes: Int) extends BaseMessageForm {

  def buildMessage(projectId: Int, contentUrl: Option[String], creatorId: Int, updaterId: Int,
    filter: PushFilter, name: String): AbTestMessage = {

    val n = if (ctrlGroup) name + "-控制组" + symbol else name + "-实验组" + symbol
    val m = metric match {
      case Some(v) => Some(v.build)
      case _ => None
    }
    AbTestMessage(
      None,
      projectId,
      creatorId,
      updaterId,
      state.getOrElse(Status.draft),
      MessagePlatform.App,
      MessageType.abTest,
      audienceType,
      audienceId,
      filter,
      contentUrl,
      content,
      n,
      symbol,
      AbTestMessageForm.DefaultAbTestVersion,
      rate,
      AbTestMessageForm.EmptyMetricRef,
      ctrlGroup,
      m,
      None,
      totalActivatedTimes)
  }
}

object AbTestMessageForm {
  val mapping: Map[String, String] = Map("uv" -> "u", "usv" -> "cs1")
  val EmptyMetricRef = 1
  val DefaultAbTestVersion = 1

  implicit val reads: Reads[AbTestMessageForm] = (
    (JsPath \ "id").readNullable[String] and
    (JsPath \ "symbol").read[String] and
    (JsPath \ "rate").readNullable[Int] and
    (JsPath \ "state").readNullable[Status] and
    (JsPath \ "audienceType").readNullable[AudienceType] and
    (JsPath \ "audienceId").readNullable[String].map(id => MappingUtil.segmentIdMapping(id, mapping, id => id.toId.toString)) and
    (JsPath \ "audienceFilter").readNullable[PushFilterForm] and
    (JsPath \ "rule").readNullable[PushRuleForm] and
    (JsPath \ "content").readNullable[PushContent] and
    (JsPath \ "ctrlGroup").read[Boolean] and
    (JsPath \ "metric").readNullable[ConversionTargetForm] and
    (JsPath \ "totalActivatedTimes").read[Int])(AbTestMessageForm.apply _)
}
