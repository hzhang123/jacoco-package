package dtos

import io.growing.micros.utils.conversion.GrowingConversions._
import model.UserLifeJourneyStage
import model.insight.Tag
import play.api.libs.json.{ Json, Writes }

/**
 * @author : xhldtc
 * @since : 2019/11/27
 */
case class UserLifeJourneyStageDTO(
  userLifeJourneyStage: UserLifeJourneyStage,
  tag: Option[Tag])

object UserLifeJourneyStageDTO {
  implicit val writes: Writes[UserLifeJourneyStageDTO] = (dto: UserLifeJourneyStageDTO) => {
    val journeyStage = dto.userLifeJourneyStage
    Json.obj(
      "stageUid" -> journeyStage.id.map(_.toUid),
      "description" -> journeyStage.description,
      "name" -> journeyStage.name,
      "rank" -> journeyStage.rank,
      "tag" -> dto.tag,
      "creatorId" -> journeyStage.creatorId.toUid,
      "updaterId" -> journeyStage.updaterId.toUid,
      "state" -> journeyStage.state,
      "createdAt" -> journeyStage.createdAt.getMillis,
      "updatedAt" -> journeyStage.updatedAt.getMillis)
  }
}
