package model.auth

import io.growing.marketing.common.model.BaseModel
import io.growing.micros.business.model.Status
import io.growing.micros.business.model.Status.Status
import org.joda.time.DateTime

/**
 * Authentication
 *
 * @author damon lin
 *         2019/7/16
 */
case class Authentication(
  id: Option[Int] = None,
  key: String,
  userId: Option[Int] = None,
  provider: String,
  override val uid: String,
  email: Option[String] = None,
  token: String,
  createdAt: DateTime = DateTime.now,
  updatedAt: DateTime = DateTime.now) extends BaseModel[Authentication] { self =>

  def withId(id: Int): Authentication = self.copy(id = Some(id))

  override val projectId: Int = -1
  override val creatorId: Int = -1
  override val updaterId: Int = -1
  override val state: Status = Status.activated

}
