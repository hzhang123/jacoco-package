package dao

import io.growing.marketing.common.model.{ SmsConfig, SmsProvider }
import io.growing.marketing.common.model.SmsProvider.SmsProvider
import io.growing.micros.play.slickpg.GrowingPostgresDriver
import io.growing.micros.play.slickpg.GrowingPostgresDriver.api._
import javax.inject.Inject
import play.api.db.slick.{ DatabaseConfigProvider, HasDatabaseConfigProvider }
import slick.lifted.{ ProvenShape, Rep, Tag }

import scala.concurrent.Future

class SmsConfigTable(tag: Tag) extends BaseTable[SmsConfig](tag, "sms_configurations") {

  implicit val SmsProviderE = MappedColumnType.base[SmsProvider, String](
    l => l.toString,
    r => SmsProvider.withName(r))

  def smsProvider: Rep[SmsProvider] = column[SmsProvider]("sms_provider")

  def providerKey: Rep[String] = column[String]("provider_key")

  def providerValue: Rep[Option[String]] = column[Option[String]]("provider_value")

  def signNames: Rep[List[String]] = column[List[String]]("sign_names")

  def numAttribute: Rep[Option[String]] = column[Option[String]]("num_attribute")

  def sendPath: Rep[Option[String]] = column[Option[String]]("send_path")

  override def * : ProvenShape[SmsConfig] = (id.?, projectId, creatorId, updaterId, state, smsProvider, providerKey,
    providerValue, signNames, numAttribute, sendPath, createdAt, updatedAt) <> ((SmsConfig.apply _).tupled, SmsConfig.unapply)
}

class SmsConfigDao @Inject() (override protected val dbConfigProvider: DatabaseConfigProvider)
  extends BaseDao[SmsConfigTable, SmsConfig] with HasDatabaseConfigProvider[GrowingPostgresDriver] {

  override val table = TableQuery[SmsConfigTable]

  override lazy val insertQuery = table returning table.map(_.id) into ((SmsConfig, id) => SmsConfig.copy(id = Some(id)))

  override def insert(model: SmsConfig): Future[SmsConfig] = super.insert(model)

  override def deleteById(projectId: Int, id: Int): Future[Int] = super.deleteById(projectId, id)

  def updateConfig(id: Int, model: SmsConfig): Future[Int] = {
    db.run(table.filter(c => c.id === id).update(model))
  }

  def findByConfigId(configId: Int): Future[Option[SmsConfig]] = {
    db.run(table.filter(config => config.id === configId).result.headOption)
  }
}
