package actors.job

import actors.Injector
import actors.job.PushJobScheduleActor.PushJobTickKey
import akka.actor.Props
import com.typesafe.scalalogging.LazyLogging
import dao.PushJobDao
import infrastructure.lock.LockManager
import io.growing.micros.business.model.Status
import io.growing.micros.utils.time.DateTimeUtil
import javax.inject.Inject

import scala.concurrent.ExecutionContext.Implicits.global
import scala.concurrent.Future
import scala.concurrent.duration.{ Duration, _ }
import scala.util.{ Failure, Success }

/**
 * @author : xhldtc
 * @since : 2019/10/30
 */
class PushJobScheduleActor @Inject() (injector: Injector) extends AbstractTimerActor with LazyLogging {

  override val TimerKey: Any = PushJobTickKey
  override val delayMillis: Int = 500
  override val intervalSeconds: Int = 30
  override val lockManager: LockManager = injector.lockManager
  val pushJobDao: PushJobDao = injector.pushJobDao
  val pushTaskDao = injector.pushTaskDao

  private val timeoutJobCheckDuration: Duration = 1.hour

  override def initJob(): Unit = {
    logger.info("start to schedule push job background checking timer")
  }

  /**
   * 1小时之前update的job,状态还是planning,并且对应messageId下所有的task都finished或者fail了，或者没有task(说明没有符合条件的推送设备)
   * 把这种job状态更新成finished，message的状态也更新
   *
   * @return
   */
  override def tickJob(): Future[Unit] = {
    for {
      messageIds <- pushJobDao.findPlanningJobMessageIdsPeriodAgo(timeoutJobCheckDuration)
      dailyPushMessageIds <- injector.pushRuleDao.findActiveDailyRuleMessageIds()
      _ <- pushTaskDao.updateTimeoutTasks(DateTimeUtil.UTCNow.minus(timeoutJobCheckDuration.toMillis), Seq())
    } yield {
      Future.traverse(messageIds) { messageId =>
        injector.pushTaskDao.findByMessageId(messageId).andThen {
          case Success(pushTasks) if pushTasks.forall(task => task.state == Status.finished || task.state == Status.failed) => updateStateToFinish(messageId, dailyPushMessageIds)
          case Failure(ex) => logger.error(s"query error for message: $messageId", ex)
        }
      }
    }
  }

  private[this] def updateStateToFinish(messageId: Int, dailyPushMessageIds: Seq[Int]): Future[_] = {
    if (!dailyPushMessageIds.contains(messageId)) {
      logger.info(s"updating once push message to stop, message id: $messageId")
      injector.pushMessageDao.updateActivatedToStop(Seq(messageId))
    }
    logger.info(s"updating push job to finish, message id: $messageId")
    pushJobDao.updateToFinishedByMessageIds(Seq(messageId))
  }
}

object PushJobScheduleActor {

  private case object PushJobTickKey

  def props(injector: Injector) = Props[PushJobScheduleActor](
    new PushJobScheduleActor(injector))
}
