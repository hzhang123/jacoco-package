package model.insight

import form.TagForm
import io.growing.marketing.common.model.BaseModel
import io.growing.micros.business.model.Status
import io.growing.micros.business.model.Status.Status
import io.growing.micros.utils.time.DateTimeUtil.UTCNow
import model.insight.TagType.TagType
import org.joda.time.DateTime
import play.api.libs.json.{ Json, Writes }

/**
 *
 * @author AI
 *         2019/9/30
 */
final case class Tag(
  id: Option[Int] = None,
  projectId: Int,
  name: String,
  description: Option[String],
  `type`: TagType,
  computes: Seq[ComputeDefinition],
  creatorId: Int,
  updaterId: Int,
  state: Status = Status.activated,
  createdAt: DateTime = UTCNow,
  updatedAt: DateTime = UTCNow) extends BaseModel[Tag] {
  self =>
  override def withId(id: Int): Tag = this.copy(id = Some(id))
}

object Tag {

  def buildByTagForm(projectId: Int, tagForm: TagForm, userId: Int, state: Status = Status.activated): Tag = {
    Tag(None, projectId, tagForm.name, tagForm.description, tagForm.`type`, tagForm.computes, userId, userId, state)
  }

  implicit val writes: Writes[Tag] = (tag: Tag) =>
    Json.obj(
      "name" -> tag.name,
      "description" -> tag.description,
      "type" -> tag.`type`,
      "computes" -> tag.computes)
}
