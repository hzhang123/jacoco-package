package service.impl

import dao.ProductDao
import javax.inject.{ Inject, Singleton }
import service.ProductService
import io.growing.micros.business.model.Product

import scala.concurrent.Future

@Singleton
class ProductServiceImpl @Inject() (productDao: ProductDao) extends ProductService {

  override def findById(id: Int): Future[Option[Product]] = productDao.findById(id)

  override def all(): Future[List[Product]] = productDao.all()

  def findByIds(projectId: Int, ids: Seq[Int]): Future[Seq[Product]] =
    productDao.findByIds(projectId, ids)
}
