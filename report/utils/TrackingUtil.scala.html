package utils

import io.growing.sdk.java.GrowingAPI
import com.typesafe.scalalogging.LazyLogging
import io.growing.marketing.common.model.PushMessage
import io.growing.micros.business.model.Project
import io.growing.sdk.java.dto.GIOEventMessage
import model.Statics

/**
 * TrackingUtil
 *
 * @author damon lin
 *         2019/5/21
 */
object TrackingUtil extends LazyLogging {

  // 发送到 growing io 的数据
  val ai = "0a1b4118dd954ec3bcc69da5138bdb96"

  def trackCreatedPopup(userId: Int, message: PushMessage): Unit = {
    traceForGIO("create_popup", userId.toString, buildProperties(message))
  }

  private[this] def buildProperties(message: PushMessage): Map[String, String] = {
    Map("popup_id" -> message.dbid.toString, "popup_name" -> message.name, "popup_state" -> message.state.toString)
  }

  def trackPushMessageSent(ai: String, visitorId: String, name: String): Unit = {
    _traceVisitor(Statics.STATIC_PUSH_MESSAGE_SENT_EVENT_KEY, ai, visitorId, Map(Statics.EVENT_LEVEL_VAR_PUSH_MESSAGE_NAME_KEY -> name))
  }

  private def _traceVisitor(eventKey: String, ai: String, userId: String, properties: Map[String, String] = Map.empty[String, String]) = {
    val builder = new GIOEventMessage.Builder()
      .eventKey(eventKey)
      .visitorId(userId)
    properties.foreach(kv ⇒ builder.addEventVariable(kv._1, kv._2))
    val message = builder.build()
    logger.info("send message by java sdk: " + message.getMapResult)
    GrowingAPI.send(ai, message)
  }

  def traceForGIO(eventKey: String, userId: String, properties: Map[String, String] = Map.empty[String, String]) = {
    val builder = new GIOEventMessage.Builder()
      .eventKey(eventKey)
      .loginUserId(userId)
    properties.foreach(kv ⇒ builder.addEventVariable(kv._1, kv._2))
    val message = builder.build()
    logger.info("send message by java sdk: " + message.getMapResult)
    GrowingAPI.send(ai, message)
  }
}
