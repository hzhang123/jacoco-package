package controllers

import play.api.libs.json._
import play.api.mvc.Results._
import java.util.Locale

import io.growing.micros.utils.conversion.GrowingConversions._
import service.{ ProductService, PushMessageService, PushRuleService }
import javax.inject.{ Inject, Singleton }
import form.{ MediaForm, PushMessageForm }
import com.typesafe.scalalogging.LazyLogging
import handlers.MessageContentHandler

import scala.concurrent.ExecutionContext.Implicits.global
import io.growing.micros.business.controller.{ AbstractController, ProjectActionComponent }
import io.growing.micros.business.auth._
import io.growing.micros.play.module.FileService
import io.growing.micros.play.mvc.FutureUtils
import play.api.mvc.{ BodyParser, PlayBodyParsers, Result }
import utils.TrackingManager

import scala.concurrent.Future

@Singleton
class PushMessageController @Inject() (
  trackingManager: TrackingManager,
  fileService: FileService,
  productService: ProductService,
  pushRuleService: PushRuleService,
  errorFormatterAction: ErrorFormatterAction,
  pushMessageService: PushMessageService,
  messageContentHandler: MessageContentHandler,
  parse: PlayBodyParsers) extends AbstractController with ProjectActionComponent with PermissionCheckComponent
  with ACLControlComponent with LazyLogging with FutureUtils {

  val readPermission: String = AuthConstants.Marketing.Read
  val managePermission: String = AuthConstants.Marketing.Manage

  def Actions[B](auth: String, projectUid: String, parse: BodyParser[B]) =
    WithTimeoutAction(parse) andThen errorFormatterAction andThen
      new AuthAction(parse) andThen
      ProjectAction(projectUid.toId) andThen
      PermissionCheckAction(auth)

  def index(projectUid: String) = Actions(readPermission, projectUid, parse.empty).async { implicit request =>
    pushMessageService.findByProject(request.projectId).map { messages =>
      Ok(Json.toJson(messages.map(_.toPresenter)))
    }
  }

  def find(projectUid: String, messageId: String) = Actions(readPermission, projectUid, parse.empty).async { implicit request =>
    pushMessageService.findByProjectAndId(request.projectId, messageId.toId).map { maybeMessageDto =>
      Ok(Json.toJson(maybeMessageDto.map(_.toPresenter)))
    }
  }

  def create(projectUid: String) = Actions(managePermission, projectUid, parse.tolerantJson).async(parse.tolerantJson) { implicit request ⇒
    request.body.validate[PushMessageForm].fold(
      errors => jsonBadRequest()(errors),
      form ⇒ {
        pushMessageService.create(form, request.project, request.userId)
          .map { r ⇒
            if (r.message.isActive && !r.message.isPush) {
              trackingManager.trackingPopupActive(r.message, r.rules.headOption, "直接上线")
            }
            Ok(Json.toJson(r.toPresenter))
          }
      })
  }

  def update(projectUid: String, id: String) = Actions(managePermission, projectUid, parse.tolerantJson).async(parse.tolerantJson) { implicit request =>
    request.body.validate[PushMessageForm].fold(
      errors => jsonBadRequest()(errors),
      form => {
        pushMessageService.update(form, id.toId, request.project, request.userId)
          .map { r ⇒
            if (r.message.isActive && !r.message.isPush) {
              trackingManager.trackingPopupActive(r.message, r.rules.headOption, r.from)
            }
            Ok(Json.toJson(r.toPresenter))
          }
      })
  }

  def archive(projectUid: String, id: String) = Actions(managePermission, projectUid, parse.empty).async(parse.empty) { implicit request ⇒
    implicit val locale: Locale = request.locale
    pushMessageService.deleteById(projectUid.toId, id.toId, request.userId).map(_ => NoContent)
  }

  // todo: api 接口限制大小
  // todo: 前端传文件名称过来
  def uploadFile(projectUid: String) = Actions(readPermission, projectUid, parse.tolerantJson).async(parse.tolerantJson) { implicit request =>
    request.body.validate[MediaForm].fold(
      errors => jsonBadRequest()(errors),
      form => {
        messageContentHandler.handleUpload(request.project, form)
      })
  }

  def updateWxQrCode(projectUid: String, messageUid: String, productUid: String) = Actions(managePermission, projectUid, parse.empty).async { implicit request =>
    implicit val locale: Locale = request.locale
    pushMessageService.updateMinpPreviewUrl(projectUid.toId, messageUid.toId, productUid.toId, request.userId)
      .map {
        case Left(error) => UnprocessableEntity(jsonErrMsgGen(Seq(error)))
        case Right(url) => Ok(Json.obj("wxQrCode" -> url))
      }
  }

  private[this] def jsonBadRequest()(errors: Seq[(JsPath, Seq[JsonValidationError])]): Future[Result] = {
    Future.successful(BadRequest(jsonErrMsgGen(errors.flatMap(_._2.map(_.message)))))
  }
}
