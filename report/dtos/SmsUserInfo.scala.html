package dtos

import com.typesafe.scalalogging.LazyLogging
import play.api.libs.json.{ JsError, JsSuccess, JsValue, Reads }

case class SmsUserInfo(info: Seq[String])

object SmsUserInfo extends LazyLogging {
  implicit val reads: Reads[Seq[SmsUserInfo]] = (json: JsValue) => {
    val data = (json \ "data").asOpt[Seq[Seq[String]]]
    if (data.isEmpty)
      JsError("")
    else
      JsSuccess(valueReads(data.get))
  }

  private def valueReads(values: Seq[Seq[String]]) = {
    values.map(value => SmsUserInfo(value))
  }
}
