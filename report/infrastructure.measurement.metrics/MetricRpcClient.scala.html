package infrastructure.measurement.metrics

import com.google.inject.{ Inject, Singleton }
import com.google.protobuf.{ Any, Empty, StringValue }
import form.EventForm
import io.growing.events.grpc.v1.InsightServiceGrpc.InsightServiceStub
import io.growing.events.grpc.v1.MeasureServiceGrpc.MeasureServiceStub
import io.growing.events.grpc.v1.RealtimeServiceGrpc.RealtimeServiceStub
import io.growing.events.grpc.v1.{ BatchCreateMeasureRequest, GetMetricsRequest, RealtimeRequest, SearchCustomEventKeysRequest }
import io.growing.events.protobuf._
import io.growing.micros.middleware.rpc.grpc.GrpcClient
import io.growing.micros.utils.conversion.GrowingConversions._
import io.growing.micros.utils.hash.HashIdGenerator
import io.grpc.{ CallOptions, StatusRuntimeException }
import model.insight.Measurement
import org.slf4j.LoggerFactory
import play.api.libs.json.Json
import play.api.libs.ws.WSClient
import utils.ConfigParams

import scala.collection.JavaConverters._
import scala.concurrent.ExecutionContext.Implicits.global
import scala.concurrent.Future
import scala.util.control.NonFatal

@Singleton
class MetricRpcClient @Inject() (ws: WSClient, configParams: ConfigParams, grpcClient: GrpcClient) {

  private lazy val logger = LoggerFactory.getLogger(getClass)
  private val REMOTE_EVENT_SERVICE_NAME = "events-service"
  // 为了数据管理中看不到，所以设置userId为-1
  private val DEFAULT_USER_ID = -1

  def createEventsBatch(entityForms: Seq[EventForm], projectId: Int, userId: Int, ignoreQuota: Boolean = false): Future[Either[String, Seq[CustomEventDto]]] = {
    logger.info("CreateEventsBatchV2 form: " + Json.toJson(entityForms))

    val request = BatchCreateMeasureRequest.newBuilder()
      .setCreatorId(DEFAULT_USER_ID)
      .setType(Measures.MeasureType.CUSTOM_EVENT)
      .setIsSystem(true)
      .setProjectId(projectId)

    buildCustomEventDto(entityForms).foreach {
      event => request.addMeasures(com.google.protobuf.Any.pack(event))
    }

    grpcClient
      .target[MeasureServiceStub, BatchCreateMeasureRequest, Any](REMOTE_EVENT_SERVICE_NAME)
      .withOptions(CallOptions.DEFAULT)
      .withRequest(request.build())
      .stream(stub ⇒ stub.batchCreate)
      .map { result =>
        logger.info("createEventsBatchV2 result" + Json.toJson(entityForms))
        val events = result.map(_.unpack(classOf[CustomEventDto]))
        events.foreach(each => {
          enableCustomEventRealtime(projectId, each.getId.toId, DEFAULT_USER_ID)
        })
        Right(events)
      }.recover {
        case e: Exception =>
          logger.error("grpc failed to change realtime, cause exception is ", e)
          Left("grpc failed to change realtime, cause exception is " + e.toString)
      }
  }

  def queryCustomEvents(eventKeys: Seq[String], projectId: Int): Future[Seq[CustomEventDto]] = {
    val request = SearchCustomEventKeysRequest.newBuilder()
      .setProjectId(projectId)
      .addAllKeys(eventKeys.asJava)

    grpcClient
      .target[MeasureServiceStub, SearchCustomEventKeysRequest, CustomEventDto](REMOTE_EVENT_SERVICE_NAME)
      .withOptions(CallOptions.DEFAULT)
      .withRequest(request.build())
      .stream(stub ⇒ stub.searchCustomEventKeys)
      .recover {
        case e: Exception =>
          logger.error("grpc failed to query events, cause exception is ", e)
          Seq.empty
      }
  }

  def enableCustomEventRealtime(projectId: Int, measureId: Int, operatorId: Int): Unit = {
    logger.info(s"EnableCustomEventRealtime projectId $projectId, measureId $measureId, operatorId $operatorId .")

    val request = RealtimeRequest.newBuilder()
      .setProjectId(projectId)
      .setMeasureId(measureId)
      .setOperatorId(operatorId)
      .setMeasureType(RealtimeRequest.MeasureType.CUSTOM)
      .build()
    grpcClient
      .target[RealtimeServiceStub, RealtimeRequest, Empty](REMOTE_EVENT_SERVICE_NAME)
      .withOptions(CallOptions.DEFAULT)
      .withRequest(request)
      .call(stub ⇒ stub.enabled)
      .recover {
        case e: StatusRuntimeException ⇒
          logger.warn(s"Grpc failed to change realtime, caused by: " + e.getLocalizedMessage)
        case NonFatal(e) ⇒
          logger.error("Grpc failed to change realtime, caused by: " + e.getLocalizedMessage)
        case e: Exception =>
          logger.error("Grpc failed to change realtime, caused by: " + e.getLocalizedMessage)
      }
  }

  //TODO hashId的转换events服务用的salt可能和我们不一样！！！
  def getMetrics(projectId: Int, measurements: Seq[Measurement]): Future[Seq[MetricDto]] = {
    val request = GetMetricsRequest.newBuilder().setProjectId(projectId)
    measurements.foreach { measurement =>
      val builder = MeasurementDto.newBuilder().setType(measurement.`type`)
      measurement.key match {
        case None ⇒ builder.setId(measurement.id.get)
        case Some(key) ⇒ builder.setId(key)
      }
      measurement.attribute.foreach(attribute => builder.setAttribute(StringValue.of(attribute)))
      request.addMeasurements(builder.build())
    }

    grpcClient
      .target[InsightServiceStub, GetMetricsRequest, MetricDto](REMOTE_EVENT_SERVICE_NAME)
      .withOptions(CallOptions.DEFAULT)
      .withRequest(request.build)
      .stream(stub => stub.getMetrics)
      .map { metrics =>
        logger.info(s"getMetric result: $metrics")
        metrics
      }
    //Future.successful(MetricDto.newBuilder().setExpression("testExpression").build())
  }

  private def buildCustomEventDto(entityForms: Seq[EventForm]): Seq[CustomEventDto] = {
    entityForms.map { eventForm =>
      val grpcAttributes = eventForm.attributes.map { attribute =>
        CustomEventAttributeDto.newBuilder().setId(attribute.id).build()
      }
      val builder = CustomEventDto.newBuilder()
        .setCreatorId(DEFAULT_USER_ID.toString)
        .setKey(eventForm.key)
        .setName(eventForm.name)
        .setValueType(eventForm.valueType)
        .setIsSystem(true)

      grpcAttributes.foreach(a => builder.addAttributes(a))
      builder.build()
    }
  }

}
