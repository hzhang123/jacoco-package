package controllers

import com.typesafe.scalalogging.LazyLogging
import form.AbTestForm
import io.growing.micros.business.auth._
import io.growing.micros.business.controller.{ AbstractController, ProjectActionComponent }
import io.growing.micros.play.mvc.{ FutureUtils, TimeoutAction }
import io.growing.micros.utils.conversion.GrowingConversions._
import javax.inject.{ Inject, Singleton }
import play.api.libs.json.{ JsValue, Json }
import play.api.mvc.Results.{ BadRequest, Ok }
import play.api.mvc.{ Action, ActionBuilder, BodyParser, PlayBodyParsers }
import service.{ AbTestService, TokenService }

import scala.concurrent.ExecutionContext.Implicits.global

@Singleton
class AbTestController @Inject() (
  ac: TimeoutAction,
  errorFormatterAction: ErrorFormatterAction,
  abTestService: AbTestService,
  parse: PlayBodyParsers,
  tokenService: TokenService)
  extends AbstractController with ProjectActionComponent with PermissionCheckComponent
  with ACLControlComponent with LazyLogging with FutureUtils {

  val readPermission: String = AuthConstants.Marketing.Read
  val managePermission: String = AuthConstants.Marketing.Manage

  def Actions[B](auth: String, projectUid: String, parse: BodyParser[B]): ActionBuilder[AuthRequest, B] = {
    WithTimeoutAction(parse) andThen errorFormatterAction andThen new AuthAction(parse) andThen
      ProjectAction(projectUid.toId) andThen PermissionCheckAction(auth)
  }

  def createAbTest(projectUid: String): Action[JsValue] = Actions(managePermission, projectUid, parse.json).async { implicit request =>
    request.body.validate[AbTestForm].fold(
      errors => jsonBadRequest(errors),
      form =>
        abTestService.create(form, request.project, request.userId).map(dto => Ok(Json.toJson(dto))))
  }

  def updateAbTest(projectUid: String, abTestId: String): Action[JsValue] = Actions(managePermission, projectUid, parse.json).async { implicit request =>
    request.body.validate[AbTestForm].fold(
      errors => jsonBadRequest(errors),
      form =>
        abTestService.update(form, request.project, abTestId.toId, request.userId).map(dto => Ok(Json.toJson(dto))))
  }

  /**
   * 查找一条AB测试最新的一对弹窗,用于编辑页,详情页.
   * @param projectUid 项目id
   * @param abTestId AB测试id
   * @return
   */
  def findLatestAbTest(projectUid: String, abTestId: String): Action[Unit] = Actions(readPermission, projectUid, parse.empty).async { implicit request =>
    abTestService.findLatestMessages(projectUid.toId, abTestId.toId).map {
      case Left(value) => BadRequest(jsonErrMsgGen(Seq(value)))
      case Right(value) => Ok(Json.toJson(value))
    }
  }

  /**
   * 查找项目下所有AB测试最新一对弹窗,用于列表页.
   * @param projectUid 项目id
   * @return
   */
  def findAllLatestAbTest(projectUid: String): Action[Unit] = Actions(readPermission, projectUid, parse.empty).async { implicit request =>
    abTestService.findAllLatestMessages(request.projectId).map(dto => Ok(Json.toJson(dto)))
  }

  /**
   * 查找一条AB测试所有历史版本弹窗,用于前端查数据.
   * @param projectUid 项目id
   * @param abTestId AB测试id
   * @return
   */
  def findHistoryAbTest(projectUid: String, abTestId: String): Action[Unit] = Actions(readPermission, projectUid, parse.empty).async { implicit request =>
    abTestService.findHistoryMessages(projectUid.toId, abTestId.toId).map {
      case Left(value) => BadRequest(jsonErrMsgGen(Seq(value)))
      case Right(value) => Ok(Json.toJson(value))
    }
  }

  def deleteAbTest(projectUid: String, abTestId: String): Action[Unit] = Actions(readPermission, projectUid, parse.empty).async { implicit request =>
    abTestService.archive(projectUid.toId, abTestId.toId).map {
      case None => BadRequest(jsonErrMsgGen(Seq("删除失败")))
      case _ => Ok(Json.obj("message" -> "删除成功"))
    }
  }
}
