package controllers

import com.typesafe.scalalogging.LazyLogging
import io.growing.micros.business.exceptions.{ EntityNotFoundException, PermissionDeniedException }
import io.growing.micros.play.exception.{ BadRequestException, DataRequestInvokeException, NotFoundException }
import io.growing.micros.play.json.CaseClassJsonSupport
import javax.inject.Inject
import org.postgresql.util.PSQLException
import play.api.http.HttpErrorHandler
import play.api.libs.json.{ JsValue, Json }
import play.api.mvc.Results._
import play.api.mvc.{ RequestHeader, Result, Results }

import scala.concurrent.{ ExecutionContext, Future }

/**
 * @author damon lin
 *         2019/7/18
 */
class MarketingErrorHandler @Inject() (implicit ec: ExecutionContext)
  extends HttpErrorHandler with LazyLogging with CaseClassJsonSupport {

  override def onClientError(request: RequestHeader, statusCode: Int, message: String): Future[Result] = {
    Future.successful(Results.Status(statusCode)(s"A client error occurred: $message, uri: ${request.uri}"))
  }

  override def onServerError(request: RequestHeader, exception: Throwable): Future[Result] = {
    logger.error(s"Server error for (${request.method}) [${request.uri}], because ${exception.getLocalizedMessage}", exception)
    Future.successful(handleException(exception))
  }

  val handleException: PartialFunction[Throwable, Result] = {
    case e: BadRequestException => {
      BadRequest(buildError("Illegal argument", "Illegal argument" -> e.getMessage))
    }
    case _: EntityNotFoundException => NotFound(buildError("Entity not found", "entity_not_found" -> "Entity not found"))
    case e: NotFoundException => NotFound(buildError("Resource not found", "resource_not_found" -> e.errorMsg.message))
    case e: IllegalArgumentException => {
      logger.warn(e.getMessage)
      BadRequest(buildError("Illegal argument", "illegal_argument" -> e.getLocalizedMessage))
    }
    case e: UnsupportedOperationException => {
      logger.error(e.getMessage)
      NotImplemented(buildError("Not Implemented", "unsupported_operation" -> "Unsupported operation"))
    }
    case _: DataRequestInvokeException => InternalServerError(buildError("数据查询请求错误, 请稍后重试!", "qs_error" -> "数据查询请求错误, 请稍后重试!"))
    case e: PSQLException if e.getSQLState == "23505" => Conflict(buildError("Entity already exists", "duplicate_key" -> "Entity already exists"))
    case _: PermissionDeniedException => Forbidden(buildError("Permission denied", "permission_denied" -> "Permission denied"))
    case t: Throwable => {
      logger.error(t.getMessage)
      InternalServerError(buildError("Unknown error", "unknown_error" -> "系统内部错误，请稍后重试！"))
    }
  }

  def buildError(title: String, errors: (String, String)*): JsValue = {
    Json.obj("message" -> title, "errors" -> errors.map(entry => Json.obj("code" -> entry._1, "message" -> entry._2)))
  }

}
