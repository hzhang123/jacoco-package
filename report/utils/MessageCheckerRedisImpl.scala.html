package utils

import akka.actor.ActorSystem
import com.typesafe.scalalogging.LazyLogging
import io.growing.micros.middleware.cache.RedisConfigs
import javax.inject.Inject
import redis.{ RedisClientPool, RedisServer }

import scala.concurrent.Await
import scala.concurrent.duration._

/**
 * @author : xhldtc
 * @since : 2019-09-09
 */
class MessageCheckerRedisImpl @Inject() (redisConfigs: RedisConfigs) extends MessageChecker with LazyLogging {

  implicit val akkaSystem: ActorSystem = akka.actor.ActorSystem()
  private val redisClient = RedisClientPool(Seq(RedisServer(redisConfigs.getDefaultRedisConfig.host, redisConfigs.getDefaultRedisConfig.port)))
  private val DEFAULT_KEY_TIMEOUT = 24 * 60 * 60L //一天，单位秒
  private val DEFAULT_VALUE = ""

  /**
   * 判断这个key是否被消费过，如果发生异常，默认没消费过
   *
   * @param key
   * @param prefix
   * @return
   */
  override def consumed(key: String, prefix: String): Boolean = {
    val redisKey = prefix + key
    try {
      val setSuccess = Await.result(redisClient.set(redisKey, DEFAULT_VALUE, exSeconds = Some(DEFAULT_KEY_TIMEOUT), NX = true), 1.seconds)
      if (!setSuccess) {
        logger.warn(s"the key $redisKey maybe already consumed!")
      }
      !setSuccess
    } catch {
      case e: Throwable =>
        logger.error("error write redis", e)
        false
    }
  }
}
