package model.insight

import com.fasterxml.jackson.databind.annotation.JsonDeserialize
import play.api.libs.json.{ Json, Reads }

/**
 *
 * @author AI
 *         2019/11/11
 */
final case class Measurement(
  id: Option[String], //id和attribute前端传的都是uid，后端rpc调用的时候无需再转整形
  key: Option[String],
  `type`: String,
  attribute: Option[String],
  aggregator: Option[String],
  name: Option[String])

object Measurement {

  implicit val reads: Reads[Measurement] = Json.reads[Measurement]

  implicit val writes = Json.writes[Measurement]
}
