package utils

import java.net.URLDecoder
import java.nio.charset.StandardCharsets

import com.typesafe.scalalogging.LazyLogging

object ParseUtil extends LazyLogging {

  def parseUrlParameters(url: String): Map[String, String] = {
    try {
      val index = url.lastIndexOf('?')
      if (index != -1) {
        val params = url.substring(index + 1).split("&").filter(_ contains "=")
        params.map(param => {
          val pair = param.split("=", 2).map(URLDecoder.decode(_, StandardCharsets.UTF_8.name()))
          pair(0) -> pair(1)
        }).toMap
      } else {
        Map.empty
      }
    } catch {
      case e: Throwable => {
        logger.warn(s"parse url parameters failed: $url", e)
        Map.empty
      }
    }
  }
}
