package utils

import com.google.inject.Inject
import model.Statics
import play.api.Configuration

/**
 *
 * 配置读取类
 *
 * @author wusi
 * @since 2019-03-28 17:32
 */
class ConfigParams @Inject() (configuration: Configuration) {
  val isLocal: Boolean = configuration.getOptional[Boolean]("isLocal").getOrElse(false)

  val host: String = configuration.getOptional[String]("rpc.event.service.host").getOrElse("http://localhost")
  val port: String = configuration.getOptional[String]("rpc.event.service.port").map(port => ":" + port).getOrElse("")
  val token: String = configuration.getOptional[String]("rpc.event.service.token").getOrElse("3b266b3a10e9627de1acaf39801aa410")

  val qsHost: String = configuration.getOptional[String]("rpc.qs.host").getOrElse("http://qs0")
  val qsPort: String = configuration.getOptional[String]("rpc.qs.port").map(port => ":" + port).getOrElse(":6060")
  val insightPath: String = configuration.getOptional[String]("rpc.qs.insight.path").getOrElse("/stat/groupUser/viewRange")
  val visitorPropsPath: String = configuration.getOptional[String]("rpc.qs.visitorProps.path").getOrElse("/stat/userList/visitorProps")
  val loginUserPropsPath: String = configuration.getOptional[String]("rpc.qs.userInfo.path").getOrElse("/stat/groupUser/pplProps")
  val computeNumPath: String = configuration.getOptional[String]("rpc.qs.com.path").getOrElse("/stat/userJourney/rule")
  val qsRequestDelayMinMillis: Int = configuration.getOptional[Int]("rpc.qs.request.delay.min").getOrElse(1000)
  val qsRequestDelayMaxMillis: Int = configuration.getOptional[Int]("rpc.qs.request.delay.max").getOrElse(5000)
  val qsRequestTimeoutMillis: Int = configuration.getOptional[Int]("rpc.qs.request.timeout").getOrElse(30000)
  val qsRequestInsightBatchSize: Int = configuration.getOptional[Int]("rpc.qs.request.insight.batch.size").getOrElse(1000)

  val pushRetries: Int = configuration.getOptional[Int]("push.retries").getOrElse(3)
  val pushChannelExpires: Int = configuration.getOptional[Int]("push.channel.expires").getOrElse(3)

  val pushResultConsumerGroupId: String = configuration.getOptional[String]("push.result.consumer.groupId").getOrElse("push-result-consumer")
  val pushRequestConsumerGroupId: String = configuration.getOptional[String]("push.request.consumer.groupId").getOrElse("push-request-consumer")
  val pushRequestConsumerThrottleElements: Option[Int] = configuration.getOptional[Int]("push.request.consumer.throttle.elements")

  val pushChannelTopic: String = configuration.getOptional[String]("push.channel.topic").getOrElse(Statics.TOPIC_CHANNEL_PUSH)
  val pushMessageCaptionLimit: Int = configuration.getOptional[Int]("push.message.caption.limit").getOrElse(20)
  val pushMessageParagraphLimit: Int = configuration.getOptional[Int]("push.message.paragraph.limit").getOrElse(50)
  val pushTokenQueryBatchSize: Int = configuration.getOptional[Int]("push.token.query.batch.size").getOrElse(50000)

  val huaWeiCallback: String = configuration.getOptional[String]("callback.url.huawei").getOrElse("https://messages.growingio.com/v1/callback/huawei")
  val meiZuCallback: String = configuration.getOptional[String]("callback.url.meizu").getOrElse("https://messages.growingio.com/v1/callback/meizu")
  val xiaoMiCallback: String = configuration.getOptional[String]("callback.url.xiaomi").getOrElse("https://messages.growingio.com/v1/callback/xiaomi")
  val oppoCallback: String = configuration.getOptional[String]("callback.url.oppo").getOrElse("https://messages.growingio.com/v1/callback/oppo")
  val vivoCallback: String = configuration.getOptional[String]("callback.url.vivo").getOrElse("https://messages.growingio.com/v1/callback/vivo")

  val smsSubTaskConsumerGroupId: String = configuration.getOptional[String]("sms.task.consumer.groupId").getOrElse("sms-task-consumer")

  val mockHost: String = configuration.getOptional[String]("mock.path").getOrElse("http://localhost")
