package handlers.impl

import com.typesafe.scalalogging.LazyLogging
import com.yunpian.sdk.YunpianClient
import com.yunpian.sdk.constant.YunpianConstant
import io.growing.marketing.common.model.{ ShortMessage, SmsConfig }
import javax.inject.Inject

class YunpianSmsHandlerImpl @Inject() extends LazyLogging {
  def queryTemplate(templateCode: String, config: SmsConfig): Either[String, String] = {
    val client = getClient(config)
    val param = client.newParam(1)
    param.put(YunpianConstant.TPL_ID, templateCode)
    if (client.tpl().get(param).isSucc) {
      val template = client.tpl().get(param).getData.get(0)
      template.getCheck_status match {
        case "FAIL" => Left("短信模板未经审核")
        case _ => Right(template.getTpl_content)
      }
    } else {
      Left("请检查模板id和APIKEY是否正确.")
    }
  }

  def querySignName(signName: String, config: SmsConfig): Either[String, String] = {
    val client = getClient(config)
    val param = client.newParam(1)
    param.put(YunpianConstant.SIGN, signName)
    val signs = client.sign().get(param).getData.getSign
    if (signs.size() > 0) {
      Right(signs.get(0).getSign)
    } else {
      Left("not found")
    }
  }

  def sendSingleMessage(shortMsg: ShortMessage, phoneNum: String, config: SmsConfig): Either[String, String] = {
    val client = getClient(config)
    val param = client.newParam(2)
    val templateContent = queryTemplate(shortMsg.templateCode.getOrElse(""), config) match {
      case Right(v) => v
      case Left(_) => ""
    }
    val msg = templateContent.replaceAll("\\#(.*?)\\#", "%s")
      .format(shortMsg.testValues: _*)
    param.put(YunpianConstant.MOBILE, phoneNum)
    param.put(YunpianConstant.TEXT, msg)
    val response = client.sms().single_send(param)
    if (response.getCode > 0) {
      Left(response.getDetail)
    } else {
      Right(response.getDetail)
    }
  }

  private def getClient(smsConfig: SmsConfig) = {
    new YunpianClient(smsConfig.providerKey).init()
  }
}
