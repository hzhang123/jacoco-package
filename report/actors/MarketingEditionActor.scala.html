package actors

import actors.job.AbstractTimerActor
import actors.job.PushMessageScheduleDailyActor.OnTime
import actors.job.PushResultListenerActor.ResultListenerTickKey
import com.google.inject.Inject
import com.google.common.base.Throwables
import akka.actor.SupervisorStrategy.Resume
import com.typesafe.scalalogging.LazyLogging
import akka.actor.{Actor, OneForOneStrategy, Props, SupervisorStrategy}
import io.growing.marketing.common.model.{EditionStatus, MarketingEdition}
import infrastructure.lock.EmptyLockManagerImpl

import scala.concurrent.Future


class MarketingEditionActor @Inject() (
	injector: Injector,
) extends AbstractTimerActor with LazyLogging {
	import MarketingEditionActor._
	import context.dispatcher
	override val lockManager = new EmptyLockManagerImpl()
	override val TimerKey: Any = ResultListenerTickKey
	val delayMillis: Int = 600
	val intervalSeconds: Int = 60 * 10
	override val supervisorStrategy: SupervisorStrategy = OneForOneStrategy() {
		case e: Exception =>
			logger.error(Throwables.getStackTraceAsString(e))
			Resume
	}
	override def initJob(): Unit = {
	}
	override def tickJob(): Future[Any] = {
		for {
			editions <- injector.marketingEditionDao.findExpiredTrail
			_ ← injector.marketingEditionDao.update(editions.map(e ⇒ e.copy(status = Some(EditionStatus.Expired))))
		} yield 1
	}

}

object MarketingEditionActor {
	private case object ResultListenerTickKey
	def props(injector: Injector): Props = Props[MarketingEditionActor](new MarketingEditionActor(injector))
}
