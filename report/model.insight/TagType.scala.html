package model.insight

import play.api.libs.json.{ JsString, Reads, Writes }

/**
 *
 * @author AI
 *         2019/10/18
 */
object TagType extends Enumeration {

  type TagType = Value

  val SEGMENT: Value = Value("segment")
  val AGGREGATED: Value = Value("aggregated")
  val TOP_N_ATTRIBUTE: Value = Value("top_n_attribute")
  val DATA_SET_ATTRIBUTE: Value = Value("data_set_attribute")
  val ATTRIBUTION_ATTRIBUTE: Value = Value("attribution_attribute")

  implicit val reads: Reads[TagType.Value] = Reads.enumNameReads(TagType)

  implicit val writes: Writes[TagType.Value] = (p: TagType) => JsString(p.toString)

}
