package dao

import io.growing.micros.play.slickpg.GrowingPostgresDriver
import io.growing.micros.play.slickpg.GrowingPostgresDriver.api._
import javax.inject.{ Inject, Singleton }
import model.insight.TagCompute
import play.api.db.slick.{ DatabaseConfigProvider, HasDatabaseConfigProvider }
import slick.lifted.{ Rep, TableQuery, Tag }

import scala.concurrent.Future

/**
 * @author : xhldtc
 * @since : 2019/11/18
 */
class TagComputesTable(tag: Tag) extends BaseTable[TagCompute](tag, "marketing_tags_computes") {

  def tagId: Rep[Int] = column[Int]("tag_id")

  def computeId: Rep[Int] = column[Int]("compute_id")

  def computeName: Rep[Option[String]] = column[Option[String]]("compute_name")

  override def * = (id.?, projectId, tagId, computeId, computeName, creatorId, updaterId, state, createdAt, updatedAt) <> ((TagCompute.apply _).tupled, TagCompute.unapply)
}

@Singleton
class TagComputeDao @Inject() (override protected val dbConfigProvider: DatabaseConfigProvider)
  extends BaseDao[TagComputesTable, TagCompute] with HasDatabaseConfigProvider[GrowingPostgresDriver] {

  override val table = TableQuery[TagComputesTable]

  def modify(projectId: Int, tagId: Int, tagComputes: Seq[TagCompute]): Future[Unit] = {
    val deleteAction = table.filter(t => t.projectId === projectId && t.tagId === tagId).delete
    val insertAction = insertQuery ++= tagComputes
    db.run(DBIO.seq(deleteAction, insertAction))
  }

  def findByTagId(projectId: Int, tagId: Int): Future[Seq[TagCompute]] = {
    db.run(table.filter(t => t.projectId === projectId && t.tagId === tagId).result)
  }
}
