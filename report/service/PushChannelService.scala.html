package service

import java.io.ByteArrayInputStream
import java.util.Locale

import dao.{ PushChannelDao, PushDeviceDao }
import dtos.PushChannelDto
import form.PushChannelForm
import io.growing.marketing.common.model.PushChannel
import io.growing.micros.business.model.Project
import io.growing.micros.play.mvc.FutureUtils
import io.growing.micros.utils.conversion.GrowingConversions._
import javax.inject.{ Inject, Singleton }
import org.jmotor.i18n.Messages
import org.joda.time.DateTime
import utils.CertUtil

import scala.concurrent.ExecutionContext.Implicits.global
import scala.concurrent.Future

@Singleton
class PushChannelService @Inject() (pushChannelDao: PushChannelDao, pushDeviceDao: PushDeviceDao) extends FutureUtils {

  private val errorMessages = Messages("i18n/common")

  def findByProject(projectId: Int): Future[List[PushChannelDto]] = {
    pushChannelDao.findByProject(projectId).flatMap { channels =>
      Future.traverse(channels) { channel =>
        Future.successful(PushChannelDto(channel))
      }
    }
  }

  def create(pushChannelForm: PushChannelForm, project: Project, userId: Int): Future[Either[String, PushChannelDto]] = {
    val tempChannel = buildChannel(project, pushChannelForm, userId)
    pushChannelDao.insert(Seq(tempChannel)).flatMap { channels =>
      val c = channels.head
      pushDeviceDao.findDeviceWithoutChannelId(c.projectId, c.productId, c.channelName).flatMap { devices =>
        Future.sequence(devices.map(d => pushDeviceDao.updateChannelId(d.id.get, c.id)))
      }
      Future.successful(Right(PushChannelDto(channels.head)))
    }
  }

  def update(pushChannelForm: PushChannelForm, channelId: Int, project: Project, userId: Int)(implicit locale: Locale): Future[Either[String, PushChannelDto]] = {
    pushChannelDao.findById(project.dbid, channelId).flatMap {
      case None => Future.successful(Left(errorMessages.format("not found")))
      case Some(pushChannel) =>
        val newChannel = copyNewChannel(userId, pushChannel, pushChannelForm)
        pushChannelDao.update(project.dbid, channelId, newChannel)
        Future.successful(Right(PushChannelDto(newChannel)))
    }
  }

  def upload(bytes: Array[Byte], password: String): Future[Either[String, String]] = {
    try {
      Future.successful(Right(CertUtil.exportPem(new ByteArrayInputStream(bytes), password)))
    } catch {
      case _: Exception => Future.successful(Left("证书密码错误"))
    }
  }

  private[this] def buildChannel(project: Project, pushChannelForm: PushChannelForm, userId: Int): PushChannel = {
    PushChannel(
      None,
      project.dbid,
      pushChannelForm.productId.toId,
      pushChannelForm.name,
      pushChannelForm.channelName,
      pushChannelForm.packageName,
      pushChannelForm.appId,
      pushChannelForm.appKey,
      pushChannelForm.appSecret,
      userId,
      userId)
  }

  private[this] def copyNewChannel(userId: Int, channel: PushChannel, f: PushChannelForm): PushChannel = {
    channel.copy(
      name = f.name,
      appId = f.appId,
      appKey = f.appKey,
      appSecret = f.appSecret,
      updatedAt = DateTime.now(),
      updaterId = userId)
  }
}
